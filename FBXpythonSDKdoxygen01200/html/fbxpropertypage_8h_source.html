460  {

462  T* lItem = GetPropertyItem(pItemType, pIndex, &lReferencePage);
463  if( pInheritType == FbxPropertyFlags::eOverride )
464  {

467  return lItem;
468  }
469  else if( lItem )

472  lEntry->Set(lItem->Clone(this));

475  }
476  else
477  {

485  lEntry->Set((T*)0);
