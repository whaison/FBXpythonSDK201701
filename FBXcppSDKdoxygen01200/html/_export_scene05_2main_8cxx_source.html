14 // The scene created in this example is a skeleton made of 3 segments.
15 // The position of a node in a .FBX file is expressed in coordinates
16 // relative to it's parent. This example shows how to convert to and
17 // from a global position.
18 //
19 // The example illustrates how to:
20 // 1) create a skeleton segment
21 // 2) get a node's global default position
22 // 3) set a node's global default position
23 // 4) set limits, rotation order and pre/post pivots
24 // 5) export a scene in a .FBX file

68

74  else if( !lSampleFileName ) lSampleFileName = argv[i];

79  if(lResult == false)

159 bool CreateScene(FbxScene* pScene)
160 {
161  FbxVector4 lT, lR, lS;
162  FbxAMatrix lGM;
163 
164  // Create nodes.
165  FbxNode* pNodeA = FbxNode::Create(pScene,"A");
166  FbxNode* pNodeB = FbxNode::Create(pScene,"B");
167  FbxNode* pNodeC = FbxNode::Create(pScene,"C");
168  FbxNode* pNodeD = FbxNode::Create(pScene,"D");
169 
170  // Create node attributes.
171  FbxSkeleton* lSkeletonA = FbxSkeleton::Create(pScene,"");
172  lSkeletonA->SetSkeletonType(FbxSkeleton::eRoot);
173  pNodeA->SetNodeAttribute(lSkeletonA);
174  FbxSkeleton* lSkeletonB = FbxSkeleton::Create(pScene,"");
175  lSkeletonB->SetSkeletonType(FbxSkeleton::eLimbNode);
176  pNodeB->SetNodeAttribute(lSkeletonB);
177  FbxSkeleton* lSkeletonC = FbxSkeleton::Create(pScene,"");
178  lSkeletonC->SetSkeletonType(FbxSkeleton::eLimbNode);
179  pNodeC->SetNodeAttribute(lSkeletonC);
180  FbxSkeleton* lSkeletonD = FbxSkeleton::Create(pScene,"");
181  lSkeletonD->SetSkeletonType(FbxSkeleton::eLimbNode);
182  pNodeD->SetNodeAttribute(lSkeletonD);
183 
184 
185  // On node A we set translation limits
186  pNodeA->TranslationActive = true;
187  pNodeA->TranslationMinX = true;
188  pNodeA->TranslationMinY = true;
189  pNodeA->TranslationMinZ = true;
190  pNodeA->TranslationMin = FbxVector4(0.1, 0.2, 0.3);
191  pNodeA->TranslationMaxX = true;
192  pNodeA->TranslationMaxY = true;
193  pNodeA->TranslationMaxZ = true;
194  pNodeA->TranslationMax = FbxVector4(5.0, 1.0, 0.0);
195  pNodeA->UpdatePivotsAndLimitsFromProperties();
196 
197  // On node B we set the rotation order and the pre/post pivots
198  // (for these value to have an effect, we need to enable the RotationActive flag)
199  pNodeB->RotationActive = true;
200  pNodeB->RotationMaxX = true;
201  pNodeB->RotationMaxY = false;
202  pNodeB->RotationMaxZ = false;
203  pNodeB->RotationMax = FbxVector4(33.3, 0.0, 0.0);
204  pNodeB->UpdatePivotsAndLimitsFromProperties();
205 
206  pNodeB->SetPivotState(FbxNode::eSourcePivot, FbxNode::ePivotActive);
207  pNodeB->SetRotationOrder(FbxNode::eSourcePivot, eSphericXYZ);
208  pNodeB->SetUseRotationSpaceForLimitOnly(FbxNode::eSourcePivot, false);
209  pNodeB->SetQuaternionInterpolation(FbxNode::eSourcePivot, eQuatInterpClassic);
210  pNodeB->SetRotationPivot(FbxNode::eSourcePivot, FbxVector4(11.1, 22.2, 33.3));
211  pNodeB->SetPreRotation(FbxNode::eSourcePivot, FbxVector4(15.0, 30.0, 45.0));
212  pNodeB->SetPostRotation(FbxNode::eSourcePivot, FbxVector4(-45.0, -30.0, -15.0));
213 
214  // Set node hierarchy.
215  pScene->GetRootNode()->AddChild(pNodeA);
216  pNodeA->AddChild(pNodeB);
217  pNodeB->AddChild(pNodeC);
218  pNodeC->AddChild(pNodeD);
219 
220  // Set global position of node A.
221  lT.Set(0.0, 0.0, 0.0); lGM.SetT(lT);
222  lR.Set(0.0, 0.0, 45.0); lGM.SetR(lR);
223  SetGlobalDefaultPosition(pNodeA, lGM);
224 
225  // Set global position of node B.
226  lT.Set(30.0, 20.0, 0.0); lGM.SetT(lT);
227  lR.Set(0.0, 0.0, 0.0); lGM.SetR(lR);
228  SetGlobalDefaultPosition(pNodeB, lGM);
229 
230  // Set global position of node C.
231  lT.Set(55.0, 20.0, 0.0); lGM.SetT(lT);
232  lR.Set(0.0, 0.0, -40.0); lGM.SetR(lR);
233  SetGlobalDefaultPosition(pNodeC, lGM);
234 
235  // Set global position of node D.
236  lT.Set(70.0, 10.0, 0.0); lGM.SetT(lT);
237  lR.Set(0.0, 0.0, 0.0); lGM.SetR(lR);
238  SetGlobalDefaultPosition(pNodeD, lGM);
239 
240  // Set meta-data on some of the nodes.
241  //
242  // For this sample, we'll use a hiearchical set of meta-data:
243  //
244  // Family
245  // Type
246  // Instance
247  //
248  // Family contains all the common properties, and the lower levels override various
249  // values.
250  //
251  FbxObjectMetaData* pFamilyMetaData = FbxObjectMetaData::Create(pScene, "Family");
252  FbxProperty::Create(pFamilyMetaData, FbxStringDT, "Level", "Level").Set(FbxString("Family"));
253  FbxProperty::Create(pFamilyMetaData, FbxStringDT, "Type", "Type").Set(FbxString("Wall"));
254  FbxProperty::Create(pFamilyMetaData, FbxFloatDT, "Width", "Width").Set(10.0f);
255  FbxProperty::Create(pFamilyMetaData, FbxDoubleDT, "Weight", "Weight").Set(25.0);
256  FbxProperty::Create(pFamilyMetaData, FbxDoubleDT, "Cost", "Cost").Set(1.25);
257 
258  FbxObjectMetaData* pTypeMetaData = FbxCast<FbxObjectMetaData>(pFamilyMetaData->Clone(FbxObject::eReferenceClone, pScene));
259 
260  pTypeMetaData->SetName("Type");
261 
262  // On this level we'll just override two properties
263  pTypeMetaData->FindProperty("Cost").Set(2500.0);
264  pTypeMetaData->FindProperty("Level").Set(FbxString("Type"));
265 
266  FbxObjectMetaData* pInstanceMetaData = FbxCast<FbxObjectMetaData>(pTypeMetaData->Clone(FbxObject::eReferenceClone, pScene));
267 
268  pInstanceMetaData->SetName("Instance");
269 
270  // And on this level, we'll go in and add a brand new property, too.
271  FbxProperty::Create(pInstanceMetaData, FbxStringDT, "Sku", "Sku#").Set(FbxString("143914-10"));
272  pInstanceMetaData->FindProperty("Width").Set(1100.50f);
273  pInstanceMetaData->FindProperty("Type").Set(FbxString("Super Heavy Duty Wall"));
274  pInstanceMetaData->FindProperty("Level").Set(FbxString("Instance"));
275 
276  // Finally connect metadata information to some of our nodes.
277  pNodeA->ConnectSrcObject(pInstanceMetaData);
278  pNodeC->ConnectSrcObject(pInstanceMetaData); // Share the same object
279 
280  pNodeD->ConnectSrcObject(pTypeMetaData);
281 
282  return true;
283 }

285 // Function to get a node's global default position.
286 // As a prerequisite, parent node's default local position must be already set.
287 void SetGlobalDefaultPosition(FbxNode* pNode, FbxAMatrix pGlobalPosition)
288 {
289  FbxAMatrix lLocalPosition;
290  FbxAMatrix lParentGlobalPosition;
291 
292  if (pNode->GetParent())
293  {
294  lParentGlobalPosition = GetGlobalDefaultPosition(pNode->GetParent());
295  lLocalPosition = lParentGlobalPosition.Inverse() * pGlobalPosition;
296  }
297  else
298  {
299  lLocalPosition = pGlobalPosition;
300  }
301 
302  pNode->LclTranslation.Set(lLocalPosition.GetT());
303  pNode->LclRotation.Set(lLocalPosition.GetR());
304  pNode->LclScaling.Set(lLocalPosition.GetS());
305 }
306 
307 // Recursive function to get a node's global default position.
308 // As a prerequisite, parent node's default local position must be already set.
309 FbxAMatrix GetGlobalDefaultPosition(FbxNode* pNode)
310 {
311  FbxAMatrix lLocalPosition;
312  FbxAMatrix lGlobalPosition;
313  FbxAMatrix lParentGlobalPosition;
314 
315  lLocalPosition.SetT(pNode->LclTranslation.Get());
316  lLocalPosition.SetR(pNode->LclRotation.Get());
317  lLocalPosition.SetS(pNode->LclScaling.Get());
318 
319  if (pNode->GetParent())
320  {
321  lParentGlobalPosition = GetGlobalDefaultPosition(pNode->GetParent());
322  lGlobalPosition = lParentGlobalPosition * lLocalPosition;
323  }
324  else
325  {
326  lGlobalPosition = lLocalPosition;
327  }
328 
329  return lGlobalPosition;
330 }
