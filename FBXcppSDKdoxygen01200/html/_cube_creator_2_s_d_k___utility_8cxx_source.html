36 bool InitializeSdkObjects(FbxManager*& pManager, FbxScene*& pScene)
37 {
38  //The first thing to do is to create the FBX Manager which is the object allocator for almost all the classes in the SDK
39  pManager = FbxManager::Create();
40  if( !pManager )
41  {
42  FBXSDK_printf("Error: Unable to create FBX Manager!\n");
43  exit(1);
44  }
45  else FBXSDK_printf("Autodesk FBX SDK version %s\n", pManager->GetVersion());
46 
47  //Create an IOSettings object. This object holds all import/export settings.
48  FbxIOSettings* ios = FbxIOSettings::Create(pManager, IOSROOT);
49  pManager->SetIOSettings(ios);
50 
51  //Create an FBX scene. This object holds most objects imported/exported from/to files.
52  pScene = FbxScene::Create(pManager, "My Scene");
53  if( !pScene )
54  {
55  FBXSDK_printf("Error: Unable to create FBX scene!\n");
56  exit(1);
57  }
58  return true;

69 // to create a basic scene
70 bool CreateScene()
71 {
72  // Initialize the FbxManager and the FbxScene
73  if(InitializeSdkObjects(gSdkManager, gScene) == false)
74  {
75  return false;
76  }
77 
78  // set the animation stack and use the unique AnimLayer to support all the animation
79  FbxAnimStack* lAnimStack = FbxAnimStack::Create(gScene, "Animation stack camera animation");
80  gAnimLayer = FbxAnimLayer::Create(gScene, "Base Layer");
81  lAnimStack->AddMember(gAnimLayer);
82 
83  // create a marker
84  FbxNode* lMarker = CreateMarker(gScene, "Marker");
85 
86  // create a camera
87  FbxNode* lCamera = CreateCamera(gScene, "Camera");
88 
89  // create a single texture shared by all cubes
90  CreateTexture(gScene);
91 
92  // create a material shared by all faces of all cubes
93  CreateMaterial(gScene);
94 
95  // set the camera point of interest on the marker
96  SetCameraPointOfInterest(lCamera, lMarker);
97 
98  // set the marker position
99  SetMarkerDefaultPosition(lMarker);
100 
101  // set the camera position
102  SetCameraDefaultPosition(lCamera);
103 
104  // animate the camera
105  AnimateCamera(lCamera, gAnimLayer);
106 
107  // build a minimum scene graph
108  FbxNode* lRootNode = gScene->GetRootNode();
109  lRootNode->AddChild(lMarker);
110  lRootNode->AddChild(lCamera);
111 
112  // set camera switcher as the default camera
113  gScene->GetGlobalSettings().SetDefaultCamera((char *)lCamera->GetName());
114 
115  return true;

157 void RemoveCubes()
158 {
159  if(gSdkManager == NULL) return;
160 
161  // get the node count
162  int nc = gScene->GetNodeCount();
163 
164  // we want to keep the root node, the marker node and the camera node
165  if(nc <= 3) return;
166 
167  // remove other nodes (cube nodes)
168  // start from the end
169  for(int i=nc-1; i >= 3; i--)
170  {
171  FbxNode *node = gScene->GetNode(i);
172  gScene->RemoveNode(node);
173 
174  // remove animation
175  FbxAnimCurveNode* lCurveNode = node->LclRotation.GetCurveNode(gAnimLayer);
176  if(lCurveNode != NULL)
177  {
178  lCurveNode->Destroy(true);
179  }
180 
181  // remove from memory
182  node->Destroy(true);
183  }
184 
185  // reset cube data
186  SetInitialCubeData();
187 }

229 // to save a scene to a FBX file
230 bool SaveScene(FbxManager* pSdkManager, FbxDocument* pScene, const char* pFilename, int pFileFormat, bool pEmbedMedia)
231 {
232  if(pSdkManager == NULL) return false;
233  if(pScene == NULL) return false;
234  if(pFilename == NULL) return false;
235 
236  bool lStatus = true;
237 
238  // Create an exporter.
239  FbxExporter* lExporter = FbxExporter::Create(pSdkManager, "");
240 
241  if( pFileFormat < 0 || pFileFormat >= pSdkManager->GetIOPluginRegistry()->GetWriterFormatCount() )
242  {
243  // Write in fall back format if pEmbedMedia is true
244  pFileFormat = pSdkManager->GetIOPluginRegistry()->GetNativeWriterFormat();
245 
246  if (!pEmbedMedia)
247  {
248  //Try to export in ASCII if possible
249  int lFormatIndex, lFormatCount = pSdkManager->GetIOPluginRegistry()->GetWriterFormatCount();
250 
251  for (lFormatIndex=0; lFormatIndex<lFormatCount; lFormatIndex++)
252  {
253  if (pSdkManager->GetIOPluginRegistry()->WriterIsFBX(lFormatIndex))
254  {
255  FbxString lDesc =pSdkManager->GetIOPluginRegistry()->GetWriterFormatDescription(lFormatIndex);
256  char *lASCII = "ascii";
257  if (lDesc.Find(lASCII)>=0)
258  {
259  pFileFormat = lFormatIndex;
260  break;
261  }
262  }
263  }
264  }
265  }
266 
267  // Initialize the exporter by providing a filename.
268  if(lExporter->Initialize(pFilename, pFileFormat, pSdkManager->GetIOSettings() ) == false)
269  {
270  return false;
271  }
272 
273  // Set the export states. By default, the export states are always set to
274  // true except for the option eEXPORT_TEXTURE_AS_EMBEDDED. The code below
275  // shows how to change these states.
276  IOS_REF.SetBoolProp(EXP_FBX_MATERIAL, true);
277  IOS_REF.SetBoolProp(EXP_FBX_TEXTURE, true);
278  IOS_REF.SetBoolProp(EXP_FBX_EMBEDDED, pEmbedMedia);
279  IOS_REF.SetBoolProp(EXP_FBX_SHAPE, true);
280  IOS_REF.SetBoolProp(EXP_FBX_GOBO, true);
281  IOS_REF.SetBoolProp(EXP_FBX_ANIMATION, true);
282  IOS_REF.SetBoolProp(EXP_FBX_GLOBAL_SETTINGS, true);
283 
284  // Export the scene.
285  lStatus = lExporter->Export(pScene);
286 
287  // Destroy the exporter.
288  lExporter->Destroy();
289 
290  return lStatus;
291 }

293 // to get the filters for the <Save file> dialog (description + file extention)
294 const char *GetWriterSFNFilters()
295 {
296  int nbWriters = gSdkManager->GetIOPluginRegistry()->GetWriterFormatCount();
297 
298  FbxString s;
299  int i=0;
300 
301  for(i=0; i < nbWriters; i++)
302  {
303  s += gSdkManager->GetIOPluginRegistry()->
304  GetWriterFormatDescription(i);
305  s += "|*.";
306  s += gSdkManager->GetIOPluginRegistry()->
307  GetWriterFormatExtension(i);
308  s += "|";
309  }
310 
311  // replace | by \0
312  int nbChar = int(strlen(s.Buffer())) + 1;
313  char *filter = new char[ nbChar ];
314  memset(filter, 0, nbChar);
315 
316  FBXSDK_strcpy(filter, nbChar, s.Buffer());
317 
318  for(i=0; i < int(strlen(s.Buffer())); i++)
319  {
320  if(filter[i] == '|')
321  {
322  filter[i] = 0;
323  }
324  }
325 
326  // the caller must delete this allocated memory
327  return filter;
328 }

410 // to get a string with info about material, texture, animation
411 FbxString GetNodeInfo( const FbxNode* pNode )
412 {

414 
415  // check for texture
416  int lMaterialCount = pNode->GetMaterialCount();
417  FbxSurfacePhong* lMaterial = NULL;
418  bool lTextureExist = false;
419  if( lMaterialCount > 0)
420  {
421  for( int i = 0; i < lMaterialCount; ++i)
422  {
423  lMaterial = (FbxSurfacePhong*)pNode->GetMaterial( i);
424  if( lMaterial->Diffuse.GetSrcObjectCount<FbxFileTexture>() > 0)

427  break;

429  }
430  }
431 
432  if( lTextureExist) s+= "[Texture: Yes] ";
433  else s+= "[Texture: No] ";
434 
435  // check for animation
436  bool anim = false;
437  FbxAnimCurveNode* lCurveNode = NULL;
438 
439  // check rotation FCurve node
440  lCurveNode =((FbxNode* )pNode)->LclRotation.GetCurveNode(gAnimLayer);
441  if(lCurveNode != NULL) anim = true;
442 
443  // check Translation FCurve node
444  lCurveNode = ((FbxNode* )pNode)->LclTranslation.GetCurveNode(gAnimLayer);
445  if(lCurveNode != NULL) anim = true;
446 
447  if(anim == true)
448  {
449  s+= "[Animation: Yes] ";
450  }
451  else
452  {
453  s+= "[Animation: No] ";
454  }
455 
456  return s;
