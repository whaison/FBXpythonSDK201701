67 
68  if(r) UI_Printf("------- Export succeeded -------------------------");
69  else UI_Printf("------- Export failed ----------------------------");

104 // Creates an importer object, and uses it to
105 // import a file into a scene.
106 bool LoadScene(
107  FbxManager* pSdkManager, // Use this memory manager...
108  FbxScene* pScene, // to import into this scene
109  const char* pFilename // the data from this file.
110  )
111 {
112  int lFileMajor, lFileMinor, lFileRevision;
113  int lSDKMajor, lSDKMinor, lSDKRevision;
114  int i, lAnimStackCount;
115  bool lStatus;
116  char lPassword[1024];
117 
118  // Get the version number of the FBX files generated by the
119  // version of FBX SDK that you are using.
120  FbxManager::GetFileFormatVersion(lSDKMajor, lSDKMinor, lSDKRevision);
121 
122  // Create an importer.
123  FbxImporter* lImporter = FbxImporter::Create(pSdkManager,"");
124 
125  // Initialize the importer by providing a filename.
126  const bool lImportStatus = lImporter->Initialize(pFilename, -1, pSdkManager->GetIOSettings() );
127 
128  // Get the version number of the FBX file format.
129  lImporter->GetFileVersion(lFileMajor, lFileMinor, lFileRevision);
130 
131  if( !lImportStatus ) // Problem with the file to be imported
132  {
133  FbxString error = lImporter->GetStatus().GetErrorString();
134  UI_Printf("Call to FbxImporter::Initialize() failed.");
135  UI_Printf("Error returned: %s", error.Buffer());
136 
137  if (lImporter->GetStatus().GetCode() == FbxStatus::eInvalidFileVersion)
138  {
139  UI_Printf("FBX version number for this FBX SDK is %d.%d.%d",
140  lSDKMajor, lSDKMinor, lSDKRevision);
141  UI_Printf("FBX version number for file %s is %d.%d.%d",
142  pFilename, lFileMajor, lFileMinor, lFileRevision);
143  }
144 
145  return false;
146  }
147 
148  UI_Printf("FBX version number for this FBX SDK is %d.%d.%d",
149  lSDKMajor, lSDKMinor, lSDKRevision);
150 
151  if (lImporter->IsFBX())
152  {
153  UI_Printf("FBX version number for file %s is %d.%d.%d",
154  pFilename, lFileMajor, lFileMinor, lFileRevision);
155 
156  // In FBX, a scene can have one or more "animation stack". An animation stack is a
157  // container for animation data.
158  // You can access a file's animation stack information without
159  // the overhead of loading the entire file into the scene.
160 
161  UI_Printf("Animation Stack Information");
162 
163  lAnimStackCount = lImporter->GetAnimStackCount();
164 
165  UI_Printf(" Number of animation stacks: %d", lAnimStackCount);
166  UI_Printf(" Active animation stack: \"%s\"",
167  lImporter->GetActiveAnimStackName());
168 
169  for(i = 0; i < lAnimStackCount; i++)
170  {
171  FbxTakeInfo* lTakeInfo = lImporter->GetTakeInfo(i);
172 
173  UI_Printf(" Animation Stack %d", i);
174  UI_Printf(" Name: \"%s\"", lTakeInfo->mName.Buffer());
175  UI_Printf(" Description: \"%s\"",
176  lTakeInfo->mDescription.Buffer());
177 
178  // Change the value of the import name if the animation stack should
179  // be imported under a different name.
180  UI_Printf(" Import Name: \"%s\"", lTakeInfo->mImportName.Buffer());
181 
182  // Set the value of the import state to false
183  // if the animation stack should be not be imported.
184  UI_Printf(" Import State: %s", lTakeInfo->mSelect ? "true" : "false");
185  }
186 
187  // Import options determine what kind of data is to be imported.
188  // The default is true, but here we set the options explictly.
189 
190  IOS_REF.SetBoolProp(IMP_FBX_MATERIAL, true);
191  IOS_REF.SetBoolProp(IMP_FBX_TEXTURE, true);
192  IOS_REF.SetBoolProp(IMP_FBX_LINK, true);
193  IOS_REF.SetBoolProp(IMP_FBX_SHAPE, true);
194  IOS_REF.SetBoolProp(IMP_FBX_GOBO, true);
195  IOS_REF.SetBoolProp(IMP_FBX_ANIMATION, true);
196  IOS_REF.SetBoolProp(IMP_FBX_GLOBAL_SETTINGS, true);
197  }
198 
199  // Import the scene.
200  lStatus = lImporter->Import(pScene);
201 
202  if(lStatus == false && // The import file may have a password
203  lImporter->GetStatus().GetCode() == FbxStatus::ePasswordError)
204  {
205  UI_Printf("Please enter password: ");
206 
207  lPassword[0] = '\0';
208 
209  FBXSDK_CRT_SECURE_NO_WARNING_BEGIN
210  scanf("%s", lPassword);
211  FBXSDK_CRT_SECURE_NO_WARNING_END
212  FbxString lString(lPassword);
213 
214  IOS_REF.SetStringProp(IMP_FBX_PASSWORD, lString);
215  IOS_REF.SetBoolProp(IMP_FBX_PASSWORD_ENABLE, true);
216 
217 
218  lStatus = lImporter->Import(pScene);
219 
220  if(lStatus == false && lImporter->GetStatus().GetCode() == FbxStatus::ePasswordError)
221  {
222  UI_Printf("Incorrect password: file not imported.");
223  }
224  }
225 
226  // Destroy the importer
227  lImporter->Destroy();
228 
229  return lStatus;
230 }

232 // Exports a scene to a file
233 bool SaveScene(
234  FbxManager* pSdkManager,
235  FbxScene* pScene,
236  const char* pFilename,
237  int pFileFormat,
238  bool pEmbedMedia
239  )
240 {
241  int lMajor, lMinor, lRevision;
242  bool lStatus = true;
243 
244  // Create an exporter.
245  FbxExporter* lExporter = FbxExporter::Create(pSdkManager, "");
246 
247  if( pFileFormat < 0 ||
248  pFileFormat >=
249  pSdkManager->GetIOPluginRegistry()->GetWriterFormatCount() )
250  {
251  // Write in fall back format if pEmbedMedia is true
252  pFileFormat =
253  pSdkManager->GetIOPluginRegistry()->GetNativeWriterFormat();
254 
255  if (!pEmbedMedia)
256  {
257  //Try to export in ASCII if possible
258  int lFormatIndex, lFormatCount =
259  pSdkManager->GetIOPluginRegistry()->
260  GetWriterFormatCount();
261 
262  for (lFormatIndex=0; lFormatIndex<lFormatCount; lFormatIndex++)
263  {
264  if (pSdkManager->GetIOPluginRegistry()->
265  WriterIsFBX(lFormatIndex))
266  {
267  FbxString lDesc = pSdkManager->GetIOPluginRegistry()->
268  GetWriterFormatDescription(lFormatIndex);
269  if (lDesc.Find("ascii")>=0)
270  {
271  pFileFormat = lFormatIndex;
272  break;
273  }
274  }
275  }
276  }
277  }
278 
279  // Initialize the exporter by providing a filename.
280  if(lExporter->Initialize(pFilename, pFileFormat, pSdkManager->GetIOSettings()) == false)
281  {
282  UI_Printf("Call to FbxExporter::Initialize() failed.");
283  UI_Printf("Error returned: %s", lExporter->GetStatus().GetErrorString());
284  return false;
285  }
286 
287  FbxManager::GetFileFormatVersion(lMajor, lMinor, lRevision);
288  UI_Printf("FBX version number for this FBX SDK is %d.%d.%d",
289  lMajor, lMinor, lRevision);
290 
291  if (pSdkManager->GetIOPluginRegistry()->WriterIsFBX(pFileFormat))
292  {
293  // Export options determine what kind of data is to be imported.
294  // The default (except for the option eEXPORT_TEXTURE_AS_EMBEDDED)
295  // is true, but here we set the options explicitly.
296  IOS_REF.SetBoolProp(EXP_FBX_MATERIAL, true);
297  IOS_REF.SetBoolProp(EXP_FBX_TEXTURE, true);
298  IOS_REF.SetBoolProp(EXP_FBX_EMBEDDED, pEmbedMedia);
299  IOS_REF.SetBoolProp(EXP_FBX_SHAPE, true);
300  IOS_REF.SetBoolProp(EXP_FBX_GOBO, true);
301  IOS_REF.SetBoolProp(EXP_FBX_ANIMATION, true);
302  IOS_REF.SetBoolProp(EXP_FBX_GLOBAL_SETTINGS, true);
303  }
304 
305  // Export the scene.
306  lStatus = lExporter->Export(pScene);
307 
308  // Destroy the exporter.
309  lExporter->Destroy();
310 
311  return lStatus;
312 }

314 // Get the filters for the <Open file> dialog
315 // (description + file extention)
316 const char *GetReaderOFNFilters()
317 {
318  int nbReaders =
319  gSdkManager->GetIOPluginRegistry()->GetReaderFormatCount();
320 
321  FbxString s;
322  int i = 0;
323 
324  for(i=0; i < nbReaders; i++)
325  {
326  s += gSdkManager->GetIOPluginRegistry()->
327  GetReaderFormatDescription(i);
328  s += "|*.";
329  s += gSdkManager->GetIOPluginRegistry()->
330  GetReaderFormatExtension(i);
331  s += "|";
332  }
333 
334  // replace | by \0
335  int nbChar = int(strlen(s.Buffer())) + 1;
336  char *filter = new char[ nbChar ];
337  memset(filter, 0, nbChar);
338 
339  FBXSDK_strcpy(filter, nbChar, s.Buffer());
340 
341  for(i=0; i < int(strlen(s.Buffer())); i++)
342  {
343  if(filter[i] == '|')
344  {
345  filter[i] = 0;
346  }
347  }
348 
349  // the caller must delete this allocated memory
350  return filter;
351 }
