94 void MyOwnWriter::PrintHierarchy(FbxNode* pStartNode)
95 {
96  FbxNode* lChildNode;
97  const char* lParentName = pStartNode->GetName();
98  for(int i = 0; i<pStartNode->GetChildCount(); i++)
99  {
100  lChildNode = pStartNode->GetChild(i);
101  const char* lChildName = lChildNode->GetName();
102  FBXSDK_fprintf(mFilePointer,"%s%s%s%s%s%s%s","\"",lChildName,"\"",", parent is ","\"",lParentName,"\"\n");
103  }
104 
105  int lNodeChildCount = pStartNode->GetChildCount ();
106  while (lNodeChildCount > 0)
107  {
108  lNodeChildCount--;
109  lChildNode = pStartNode->GetChild (lNodeChildCount);
110  PrintHierarchy(lChildNode);
111  }
112 }
