13 //
14 // This example shows to the user how to get UV information from mesh,
15 // and how to modify UV information on a specific UV set.
16 // Steps:
17 // 1. Initialize FBX SDK Manager and FBX Scene
18 // 2. Load the input file to scene
19 // 3. Access Node and its mesh
20 // 4. Get the UVs Information from the mesh
21 // 5. Modify the UV information and apply to the mesh
22 // 6. Save the scene to output file
23 // 7. Destroy the FBX SDK Manager and FBX Scene

42 int main()
43 {
44 
45  //FBX SDK Default Manager
46  FbxManager* lSdkManager = NULL;
47 
48  //Scene to load from file
49  FbxScene* lScene = NULL;
50 
51  // Prepare the FBX SDK.
52  InitializeSdkObjects(lSdkManager, lScene);
53 
54  // Load the scene.
55  bool lResult = LoadScene(lSdkManager, lScene, sInputFile);
56 
57  if(lResult == false)
58  {
59  FBXSDK_printf("\n\nAn error occurred while loading the scene...");
60  }
61  else
62  {
63  //Get the first node in the scene
64  FbxNode* lNodeOfInterest = lScene->GetRootNode()->GetChild(0);
65  if(lNodeOfInterest)
66  {
67  FbxMesh* lMeshOFInterest = lNodeOfInterest->GetMesh();
68  if(lMeshOFInterest)
69  {
70  //first, load the UV information and display them
71  LoadUVInformation(lMeshOFInterest);
72 
73  //then, modify certain uv set and save it
74  SaveUVInformation(lMeshOFInterest);
75 
76  //save the modified scene to file
77  SaveScene(lSdkManager, lScene, sOutputFile);
78  }
79  }
80  }
81 
82  // Destroy all objects created by the FBX SDK.
83  DestroySdkObjects(lSdkManager, lResult);
84  return 0;
85 }

87 void LoadUVInformation(FbxMesh* pMesh)
88 {
89  //get all UV set names
90  FbxStringList lUVSetNameList;
91  pMesh->GetUVSetNames(lUVSetNameList);
92 
93  //iterating over all uv sets
94  for (int lUVSetIndex = 0; lUVSetIndex < lUVSetNameList.GetCount(); lUVSetIndex++)
95  {
96  //get lUVSetIndex-th uv set
97  const char* lUVSetName = lUVSetNameList.GetStringAt(lUVSetIndex);
98  const FbxGeometryElementUV* lUVElement = pMesh->GetElementUV(lUVSetName);
99 
100  if(!lUVElement)
101  continue;
102 
103  // only support mapping mode eByPolygonVertex and eByControlPoint
104  if( lUVElement->GetMappingMode() != FbxGeometryElement::eByPolygonVertex &&
105  lUVElement->GetMappingMode() != FbxGeometryElement::eByControlPoint )
106  return;
107 
108  //index array, where holds the index referenced to the uv data
109  const bool lUseIndex = lUVElement->GetReferenceMode() != FbxGeometryElement::eDirect;
110  const int lIndexCount= (lUseIndex) ? lUVElement->GetIndexArray().GetCount() : 0;
111 
112  //iterating through the data by polygon
113  const int lPolyCount = pMesh->GetPolygonCount();
114 
115  if( lUVElement->GetMappingMode() == FbxGeometryElement::eByControlPoint )
116  {
117  for( int lPolyIndex = 0; lPolyIndex < lPolyCount; ++lPolyIndex )
118  {
119  // build the max index array that we need to pass into MakePoly
120  const int lPolySize = pMesh->GetPolygonSize(lPolyIndex);
121  for( int lVertIndex = 0; lVertIndex < lPolySize; ++lVertIndex )
122  {
123  FbxVector2 lUVValue;
124 
125  //get the index of the current vertex in control points array
126  int lPolyVertIndex = pMesh->GetPolygonVertex(lPolyIndex,lVertIndex);
127 
128  //the UV index depends on the reference mode
129  int lUVIndex = lUseIndex ? lUVElement->GetIndexArray().GetAt(lPolyVertIndex) : lPolyVertIndex;
130 
131  lUVValue = lUVElement->GetDirectArray().GetAt(lUVIndex);
132 
133  //User TODO:
134  //Print out the value of UV(lUVValue) or log it to a file
135  }
136  }
137  }
138  else if (lUVElement->GetMappingMode() == FbxGeometryElement::eByPolygonVertex)
139  {
140  int lPolyIndexCounter = 0;
141  for( int lPolyIndex = 0; lPolyIndex < lPolyCount; ++lPolyIndex )
142  {
143  // build the max index array that we need to pass into MakePoly
144  const int lPolySize = pMesh->GetPolygonSize(lPolyIndex);
145  for( int lVertIndex = 0; lVertIndex < lPolySize; ++lVertIndex )
146  {
147  if (lPolyIndexCounter < lIndexCount)
148  {
149  FbxVector2 lUVValue;
150 
151  //the UV index depends on the reference mode
152  int lUVIndex = lUseIndex ? lUVElement->GetIndexArray().GetAt(lPolyIndexCounter) : lPolyIndexCounter;
153 
154  lUVValue = lUVElement->GetDirectArray().GetAt(lUVIndex);
155 
156  //User TODO:
157  //Print out the value of UV(lUVValue) or log it to a file
158 
159  lPolyIndexCounter++;
160  }
161  }
162  }
163  }
164  }
