13 //
14 // Illustrates the use of animation stacks, layers, curvenodes and curves.
15 //
16 // 1. Create a stack.
17 // 2. Add the mandatory base layer.
18 // 3. Show how to set the layer blend mode bypass.
19 // 4. Animate the layers's weight property.
20 // 5. Show how to use AddChannel and ResetChannels.
21 // 6. Create a 3 components curvenode and animate two of the three channels.
22 // 7. Add the curve node to the animation layer.
23 // 8. Create curves and add keys.
24 // 9. Create an extra layer with a different blend mode.
25 // 10. Show how to share the same animation curve on two layers (different channels)
26 // 11. Evaluate the overall result.

57 bool CreateScene(FbxManager* /*pSdkManager*/, FbxScene* pScene)
58 {
59  int i;
60  FbxTime lTime;
61  FbxAnimCurveKey key;
62  FbxAnimCurve* lCurve = NULL;
63 
64  // Create one animation stack
65  FbxAnimStack* lAnimStack = FbxAnimStack::Create(pScene, "Stack001");
66 
67  // this stack animation range is limited from 0 to 1 second
68  lAnimStack->LocalStop = FBXSDK_TIME_ONE_SECOND;
69  lAnimStack->Description = "This is the animation stack description field.";
70 
71  // all animation stacks need, at least, one layer.
72  FbxAnimLayer* lAnimLayer = FbxAnimLayer::Create(pScene, "Base Layer"); // the AnimLayer object name is "Base Layer"
73  lAnimStack->AddMember(lAnimLayer); // add the layer to the stack
74 
75  // Set and get the blend mode bypass of the layer
76  bool val;
77  lAnimLayer->SetBlendModeBypass(eFbxTypeCount, true); // set the bypass to all the datatypes.
78  val = lAnimLayer->GetBlendModeBypass(eFbxBool); // val = true
79  lAnimLayer->SetBlendModeBypass(eFbxBool, false); // overwrite just for the bool datatype.
80  val = lAnimLayer->GetBlendModeBypass(eFbxBool); // val = false
81  val = lAnimLayer->GetBlendModeBypass(eFbxChar); // val = true
82  val = lAnimLayer->GetBlendModeBypass(eFbxDateTime); // val = true
83  val = lAnimLayer->GetBlendModeBypass((EFbxType)-1); // invalid type, val = false
84  val = lAnimLayer->GetBlendModeBypass((EFbxType)120); // invalid type (>MAX_TYPES), val = false
85 
86 
87  // we want to animate the layer's weight property.
88  FbxAnimCurveNode* wcn = lAnimLayer->CreateCurveNode(lAnimLayer->Weight);
89  if (wcn)
90  {
91  // the curve node from the Weight property already contains 1 channel (Weight).
92  i = wcn->GetChannelsCount(); // i = 1
93 
94  // Now, let's add a second channel to the animation node. Note that this code
95  // is useless and has only been provided to show the usage of the AddChannel and
96  // ResetChannels
97  bool ret;
98  ret = wcn->AddChannel<int>("MyAddedIntChannel", 99); // this call will succed
99  i = wcn->GetChannelsCount(); // i = 2
100  ret = wcn->AddChannel<int>("MyAddedIntChannel", 10); // this call will fail, since the channel already exists.
101  i = wcn->GetChannelsCount(); // i = 2
102  wcn->ResetChannels(); // remove any added channels
103  i = wcn->GetChannelsCount(); // i = 1
104  }
105 
106  // get the Weight curve (and create it if it does not exist, wich is the case!)
107  lCurve = lAnimLayer->Weight.GetCurve(lAnimLayer, true);
108  if (lCurve)
109  {
110  // add two keys at time 0 sec and 1 sec with values 0 and 100 respectively.
111  lCurve->KeyModifyBegin();
112  for (i = 0; i < 2; i++)
113  {
114  lTime.SetSecondDouble((float)i);
115  key.Set(lTime, i*100.0f);
116  lCurve->KeyAdd(lTime, key);
117  }
118  lCurve->KeyModifyEnd();
119  }
120 
121  //
122  // now create a 3 components curvenode and animate two of the three channels.
123  //
124  // first, we need a "dummy" property so we can call the CreateTypedCurveNode
125  FbxProperty p = FbxProperty::Create(pScene, FbxDouble3DT, "Vector3Property");
126  p.Set(FbxDouble3(1.1, 2.2, 3.3));
127  FbxAnimCurveNode* lCurveNode = FbxAnimCurveNode::CreateTypedCurveNode(p, pScene);
128 
129  // let's make sure the curveNode is added to the animation layer.
130  lAnimLayer->AddMember(lCurveNode);
131 
132  // and to the "Vector3Property" since CreateTypedCurveNode does not make any connection
133  p.ConnectSrcObject(lCurveNode);
134 
135  //Example of channel get value:
136  //double v1 = lCurveNode->GetChannelValue<double>(0U, 0.0); // v1 = 1.1
137  //float v2 = lCurveNode->GetChannelValue<float> (1U, 0.0f); // v2 = 2.2
138  //int v3 = lCurveNode->GetChannelValue<int> (2U, 0); // v3 = 3
139 
140  //
141  // create two free curves (not connected to anything)
142  //
143 
144  // first curve
145  lCurve = FbxAnimCurve::Create(pScene, "curve1");
146  if (lCurve)
147  {
148  // add two keys at time 0 sec and 1 sec with values 0 and 10 respectively.
149  lCurve->KeyModifyBegin();
150  for (i = 0; i < 2; i++)
151  {
152  lTime.SetSecondDouble((float)i);
153  key.Set(lTime, i*10.0f);
154  lCurve->KeyAdd(lTime, key);
155  }
156  lCurve->KeyModifyEnd();
157  }
158 
159  // connect it to the second channel
160  lCurveNode->ConnectToChannel(lCurve, 1);
161 
162  // second curve
163  lCurve = FbxAnimCurve::Create(pScene, "curve2");
164  if (lCurve)
165  {
166  // add three keys at time 1, 2 and 3 sec with values 3.33, 6.66 and 9.99 respectively
167  lCurve->KeyModifyBegin();
168  for (i = 1; i < 4; i++)
169  {
170  lTime.SetSecondDouble((float)i);
171  key.Set(lTime, i*3.33f);
172  lCurve->KeyAdd(lTime, key);
173  }
174  lCurve->KeyModifyEnd();
175  }
176  // connect it to the third channel
177  lCurveNode->ConnectToChannel(lCurve, "Z"); // for backward compatibility, string identifier are still
178  // allowed for the X,Y,Z and W components or "0", "1", ... "9", "A", "B", ... "F" for the Matrix44 datatype
179 
180 
181  // ======================================================================
182  //
183  // Add a second animation layer and evaluate using the FbxAnimEvaluator
184  //
185  // ======================================================================
186  lAnimLayer = FbxAnimLayer::Create(pScene, "Layer2");
187  lAnimStack->AddMember(lAnimLayer);
188 
189  // get the number of animation layers in the stack
190  // int nbLayers = lAnimStack->GetMemberCount<FbxAnimLayer>(); // nblayers = 2
191  lAnimLayer = lAnimStack->GetMember<FbxAnimLayer>(1); // get the second layer
192 
193  // set its blend mode to Additive
194  lAnimLayer->BlendMode.Set(FbxAnimLayer::eBlendAdditive);
195 
196  // Now, let's animate the first channel of the "Vector3Property" (remember, we animated the second and
197  // third ones on the base layer - when we connected "curve1" and "curve2" on lCurveNode above)
198  // but first, make sure the property is animatable otherwise the creation of the curveNode is prohibited.
199  p.ModifyFlag(FbxPropertyFlags::eAnimatable, true);
200  lCurveNode = p.GetCurveNode(lAnimLayer, true); // create it since it does not exist yet
201 
202  // use "curve2" to animate it on channel 0
203  lCurveNode->ConnectToChannel(lCurve, 0U);
204 
205  // and set the other two channels values
206  lCurveNode->SetChannelValue<double>(1U, 5.0);
207  lCurveNode->SetChannelValue<double>(2U, 0.0);
208 
209  // evaluate the "Vector3Property" value at three different times
210  // with the use of the FbxAnimEvaluator so we take into account the two layers
211 
212  // make sure the evaluator is using the correct context (context == animstack)
213  pScene->SetCurrentAnimationStack(lAnimStack);
214  for (i = 0; i < 3; i++)
215  {
216  lTime.SetSecondDouble((float)i);
217  FbxDouble3 value = p.EvaluateValue<FbxDouble3>(lTime);
218  }
219 
220  /*
221  The base layer has a weight curve:
222 
223  Time | 0 | 1 | 2 |
224  Weight +-------------+----------------+--------------|
225  Base Layer | 0.0 | 100.0% | (100.0%) |
226  Layer2 | <100.0%> | <100.0%> | <100.0%> |
227  +-------------+----------------+--------------|
228 
229  () Querying values outside the first and/or last keys in a curve will return
230  the first/last key defined.
231  <> Since it has never been set, it defaults to the multiplication neutral element (in percent).
232 
233  At the specified times each channel value on their respective layers is:
234 
235  Time | 0 | 1 | 2 |
236  Channel +-------------+----------------+--------------|
237  0 Base | 0* | 1.1* | 1.1* |
238  Layer2 | (3.33) | 3.33 | 6.66 | (curve2)
239  +-------------+----------------+--------------|
240  1 Base | 0* | 10.0* | (10.0) | (curve1)
241  Layer2 | 5.0 | 5.0 | 5.0 |
242  +-------------+----------------+--------------|
243  2 Base | 0* | 3.3* | 6.66* | (curve2)
244  Layer2 | 0.0 | 0.0 | 0.0 |
245  +-------------+----------------+--------------|
246 
247  * key (or property, if not animated) value multiplied by the weight.
248  () same as value at time 1 since there is no key here.
249 
250  therefore, considering that the second animation layer's blend mode is set
251  to additive, the evaluated values for v at 0, 1 and 2 seconds are:
252 
253  time | 0 | 1 | 2 |
254  v +-------------+----------------+--------------|
255  0 | 3.33 | 4.43 | 7.76 |
256  1 | 5.0 | 15.00 | 15.00 |
257  2 | 0.0 | 3.33 | 6.66 |
258  +-------------+----------------+--------------|
259  */
260  return true;
261 }
