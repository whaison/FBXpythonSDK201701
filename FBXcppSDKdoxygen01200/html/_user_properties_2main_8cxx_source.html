44 int main(int argc, char** argv)
45 {
46  FbxManager* lSdkManager = NULL;
47  FbxScene* lScene = NULL;
48  bool lResult;
49 
50  // Prepare the FBX SDK.
51  InitializeSdkObjects(lSdkManager, lScene);
52 
53  // Create the scene.
54  FbxNode* lCube = CreateCube(lScene, "Cube");
55  FbxNode* lPyramid = CreatePyramid(lScene, "Pyramid");
56 
57  // Build the node tree.
58  FbxNode* lRootNode = lScene->GetRootNode();
59  lRootNode->AddChild(lCube);
60  lRootNode->AddChild(lPyramid);
61 
62  // Create the user properties on the Cube.
63  CreateUserProperties(lCube);
64 
65  //
66 
67  // animate the list "MyList"
68  FbxProperty p6 = lCube->FindProperty("MyList", false);
69  AnimateList(lScene, &p6);
70 
71  // Constraint (position constraint) the pyramid to the cube.
72  FbxConstraintPosition* lPositionConstraint = (FbxConstraintPosition*)CreatePositionConstraint(lScene, lCube, lPyramid);
73  if( lPositionConstraint ) lPositionConstraint->ConnectDstObject(lScene);
74 
75  // Animate the cube: the pyramid will follow, because of the position constraint.
76  AnimateCube(lScene, lCube);
77 
78  // Animate the pyramid: it doesn't actually move, because it is constrained to the immobile cube.
79  AnimatePyramid(lScene, lPyramid);
80 
81  // Save the scene.
82 
83  // The example can take an output file name as an argument.
84  const char* lSampleFileName = NULL;
85  for( int i = 1; i < argc; ++i )
86  {
87  if( FBXSDK_stricmp(argv[i], "-test") == 0 ) continue;
88  else if( !lSampleFileName ) lSampleFileName = argv[i];
89  }
90  if( !lSampleFileName ) lSampleFileName = SAMPLE_FILENAME;
91 
92  lResult = SaveScene(lSdkManager, lScene, lSampleFileName, lSdkManager->GetIOPluginRegistry()->FindWriterIDByDescription("FBX ascii (*.fbx)") );
93  if(lResult == false)
94  {
95  FBXSDK_printf("\n\nAn error occurred while saving the scene...\n");
96  DestroySdkObjects(lSdkManager, lResult);
97  return 0;
98  }
99 
100  // Destroy all objects created by the FBX SDK.
101  DestroySdkObjects(lSdkManager, lResult);
102 
103  return 0;
104 }

108 FbxNode* CreateCube(FbxScene* pScene, const char* pName)
109 {
110  typedef double Vector4[4];
111  typedef double Vector2[2];
112 
113  // indices of the vertices per each polygon
114  static int vtxId[24] = {
115  0,1,2,3, // front face (Z+)
116  1,5,6,2, // right side (X+)
117  5,4,7,6, // back face (Z-)
118  4,0,3,7, // left side (X-)
119  0,4,5,1, // bottom face (Y-)
120  3,2,6,7 // top face (Y+)
121  };
122 
123  // control points
124  static Vector4 lControlPoints[8] = {
125  { -50.0, 0.0, 50.0, 1.0}, { 50.0, 0.0, 50.0, 1.0}, { 50.0,100.0, 50.0, 1.0}, { -50.0,100.0, 50.0, 1.0},
126  { -50.0, 0.0, -50.0, 1.0}, { 50.0, 0.0, -50.0, 1.0}, { 50.0,100.0, -50.0, 1.0}, { -50.0,100.0, -50.0, 1.0}
127  };
128 
129  // normals
130  static Vector4 lNormals[8] = {
131  {-0.577350258827209,-0.577350258827209, 0.577350258827209, 1.0},
132  { 0.577350258827209,-0.577350258827209, 0.577350258827209, 1.0},
133  { 0.577350258827209, 0.577350258827209, 0.577350258827209, 1.0},
134  {-0.577350258827209, 0.577350258827209, 0.577350258827209, 1.0},
135  {-0.577350258827209,-0.577350258827209,-0.577350258827209, 1.0},
136  { 0.577350258827209,-0.577350258827209,-0.577350258827209, 1.0},
137  { 0.577350258827209, 0.577350258827209,-0.577350258827209, 1.0},
138  {-0.577350258827209, 0.577350258827209,-0.577350258827209, 1.0}
139  };
140 
141  // create the main structure.
142  FbxMesh* lMesh = FbxMesh::Create(pScene,"");
143 
144  // Create control points.
145  lMesh->InitControlPoints(8);
146  FbxVector4* vertex = lMesh->GetControlPoints();
147  memcpy((void*)vertex, (void*)lControlPoints, 8*sizeof(FbxVector4));
148 
149  // create the polygons
150  int vId = 0;
151  for (int f=0; f<6; f++)
152  {
153  lMesh->BeginPolygon();
154  for (int v=0; v<4; v++)
155  lMesh->AddPolygon(vtxId[vId++]);
156  lMesh->EndPolygon();
157  }
158 
159  // specify normals per control point.
160 
161  FbxGeometryElementNormal* lNormlElement= lMesh->CreateElementNormal();
162  lNormlElement->SetMappingMode(FbxGeometryElement::eByControlPoint);
163  lNormlElement->SetReferenceMode(FbxGeometryElement::eDirect);
164 
165  for (int n=0; n<8; n++)
166  lNormlElement->GetDirectArray().Add(FbxVector4(lNormals[n][0], lNormals[n][1], lNormals[n][2]));
167 
168 
169  // Finally we create the node containing the mesh
170  FbxNode* lNode = FbxNode::Create(pScene,pName);
171  pScene->GetRootNode()->AddChild(lNode);
172  lNode->SetNodeAttribute(lMesh);
173 
174  return lNode;
175 }

178 // Create a pyramid.
179 FbxNode* CreatePyramid(FbxScene* pScene, const char* pName)
180 {
181  int i, j;
182  FbxMesh* lMesh = FbxMesh::Create(pScene,"");
183 
184  FbxVector4 lControlPoint0(-50, 0, 50);
185  FbxVector4 lControlPoint1(50, 0, 50);
186  FbxVector4 lControlPoint2(50, 0, -50);
187  FbxVector4 lControlPoint3(-50, 0, -50);
188  FbxVector4 lControlPoint4(0, 100, 0);
189 
190  FbxVector4 lNormalP0(0, 1, 0);
191  FbxVector4 lNormalP1(0, 0.447, 0.894);
192  FbxVector4 lNormalP2(0.894, 0.447, 0);
193  FbxVector4 lNormalP3(0, 0.447, -0.894);
194  FbxVector4 lNormalP4(-0.894, 0.447, 0);
195 
196  // Create control points.
197  lMesh->InitControlPoints(16);
198  FbxVector4* lControlPoints = lMesh->GetControlPoints();
199 
200  lControlPoints[0] = lControlPoint0;
201  lControlPoints[1] = lControlPoint1;
202  lControlPoints[2] = lControlPoint2;
203  lControlPoints[3] = lControlPoint3;
204  lControlPoints[4] = lControlPoint0;
205  lControlPoints[5] = lControlPoint1;
206  lControlPoints[6] = lControlPoint4;
207  lControlPoints[7] = lControlPoint1;
208  lControlPoints[8] = lControlPoint2;
209  lControlPoints[9] = lControlPoint4;
210  lControlPoints[10] = lControlPoint2;
211  lControlPoints[11] = lControlPoint3;
212  lControlPoints[12] = lControlPoint4;
213  lControlPoints[13] = lControlPoint3;
214  lControlPoints[14] = lControlPoint0;
215  lControlPoints[15] = lControlPoint4;
216 
217  // specify normals per control point.
218  FbxGeometryElementNormal* lNormlElement = lMesh->CreateElementNormal();
219  lNormlElement->SetMappingMode(FbxGeometryElement::eByControlPoint);
220  lNormlElement->SetReferenceMode(FbxGeometryElement::eDirect);
221 
222  lNormlElement->GetDirectArray().Add(lNormalP0);
223  lNormlElement->GetDirectArray().Add(lNormalP0);
224  lNormlElement->GetDirectArray().Add(lNormalP0);
225  lNormlElement->GetDirectArray().Add(lNormalP0);
226  lNormlElement->GetDirectArray().Add(lNormalP1);
227  lNormlElement->GetDirectArray().Add(lNormalP1);
228  lNormlElement->GetDirectArray().Add(lNormalP1);
229  lNormlElement->GetDirectArray().Add(lNormalP2);
230  lNormlElement->GetDirectArray().Add(lNormalP2);
231  lNormlElement->GetDirectArray().Add(lNormalP2);
232  lNormlElement->GetDirectArray().Add(lNormalP3);
233  lNormlElement->GetDirectArray().Add(lNormalP3);
234  lNormlElement->GetDirectArray().Add(lNormalP3);
235  lNormlElement->GetDirectArray().Add(lNormalP4);
236  lNormlElement->GetDirectArray().Add(lNormalP4);
237  lNormlElement->GetDirectArray().Add(lNormalP4);
238 
239  // Array of polygon vertices.
240  int lPolygonVertices[] = { 0, 3, 2, 1,
241  4, 5, 6,
242  7, 8, 9,
243  10, 11, 12,
244  13, 14, 15 };
245 
246  // Create polygons.
247 
248  // Pyramid base.
249  lMesh->BeginPolygon();
250  for(j = 0; j < 4; j++)
251  {
252  lMesh->AddPolygon(lPolygonVertices[j]); // Control point index.
253  }
254 
255  lMesh->EndPolygon ();
256 
257  // Pyramid sides.
258  for(i = 1; i < 5; i++)
259  {
260  lMesh->BeginPolygon();
261 
262  for(j = 0; j < 3; j++)
263  {
264  lMesh->AddPolygon(lPolygonVertices[4 + 3*(i - 1) + j]); // Control point index.
265  }
266 
267  lMesh->EndPolygon ();
268  }
269 
270  FbxNode* lNode = FbxNode::Create(pScene,pName);
271 
272  lNode->SetNodeAttribute(lMesh);
273  pScene->GetRootNode()->AddChild(lNode);
274 
275  // Translate the pyramid
276  FbxVector4 lTranslation(-150, 0, 0, 0);
277  lNode->LclTranslation.Set(lTranslation);
278 
279  return lNode;
280 }

283 void CreateUserProperties(FbxNode *pNode) {
284  // Now we create the user properties
285  FbxProperty p1 = FbxProperty::Create(pNode, FbxBoolDT, "MyBooleanProperty", "My Bool");
286  FbxProperty p2 = FbxProperty::Create(pNode, FbxFloatDT, "MyRealProperty", "My floating point number");
287  FbxProperty p3 = FbxProperty::Create(pNode, FbxColor3DT, "MyColorProperty", "My Color");
288  FbxProperty p4 = FbxProperty::Create(pNode, FbxIntDT, "MyInteger", "");
289  FbxProperty p5 = FbxProperty::Create(pNode, FbxDouble4DT, "MyVector", "");
290  FbxProperty p6 = FbxProperty::Create(pNode, FbxStringListDT, "MyList", "");
291 
292  /*
293  NOTE: The properties labels exists only while the property object is in memory.
294  The label is not saved in the FBX file. When loading properties from the FBX file
295  it will take the same value as the property name.
296  */
297 
298  // we now fill the properties. All the properties are user properties so we set the
299  // correct flag
300  p1.ModifyFlag(FbxPropertyFlags::eUserDefined, true);
301  p2.ModifyFlag(FbxPropertyFlags::eUserDefined, true);
302  p3.ModifyFlag(FbxPropertyFlags::eUserDefined, true);
303  p4.ModifyFlag(FbxPropertyFlags::eUserDefined, true);
304  p5.ModifyFlag(FbxPropertyFlags::eUserDefined, true);
305  p6.ModifyFlag(FbxPropertyFlags::eUserDefined, true);
306 
307  // let's make MyColorProperty, MyVector and MyList animatables
308  p3.ModifyFlag(FbxPropertyFlags::eAnimatable, true);
309  p5.ModifyFlag(FbxPropertyFlags::eAnimatable, true);
310  p6.ModifyFlag(FbxPropertyFlags::eAnimatable, true);
311 
312  // we set the default values
313  FbxColor lRed(1.0, 0.0, 0.0);
314  p1.Set(false);
315  p2.Set(3.33);
316  p3.Set(lRed);
317  p4.Set(11);
318  p5.Set(FbxDouble3(-1.1, 2.2, -3.3));
319  p6.Set(2);
320 
321  // and some limits
322  p4.SetLimits(-5.0, 9.0);
323  p5.SetLimits(0.0, 2.1);
324 
325  // add elements to the list
326  p6.AddEnumValue("one");
327  p6.AddEnumValue("two");
328  p6.AddEnumValue("three");
329  p6.AddEnumValue("Four");
330  p6.InsertEnumValue(0, "zero");

334 // Animate the user property given by pList.
335 void AnimateList(FbxScene* pScene, FbxProperty* pList)
336 {
337  // This is represented by the AnimStack object and in order to be able to add
338  // animation curves, we need to define at least one AnimLayer.
339  FbxAnimStack* lAnimStack = FbxAnimStack::Create(pScene, ANIM_STACK_ANIMATE_LIST);
340  FbxAnimLayer* lAnimLayer = FbxAnimLayer::Create(pScene, "Base Layer");
341  lAnimStack->AddMember(lAnimLayer);
342 
343  FbxAnimCurveNode* lFbxFCurveNode = pList->GetCurveNode(lAnimLayer, true);
344  if (lFbxFCurveNode)
345  {
346  FbxTime lKeyTime;
347  FbxAnimCurve* lFCurve = lFbxFCurveNode->GetCurve(0U);
348  if (lFCurve == NULL)
349  lFCurve = lFbxFCurveNode->CreateCurve(lFbxFCurveNode->GetName());
350 
351  lFCurve->KeyModifyBegin();
352  lFCurve->ResizeKeyBuffer(5);
353 
354  // One way of setting a keyframe value
355  lKeyTime.SetSecondDouble(0.0);
356  FbxAnimCurveKey lKey1(lKeyTime, 0.0);
357  lKey1.SetInterpolation(FbxAnimCurveDef::eInterpolationConstant);
358  lKey1.SetConstantMode(FbxAnimCurveDef::eConstantStandard);
359  lFCurve->KeySet(0, lKey1);
360 
361  lKeyTime.SetSecondDouble(1.0);
362  FbxAnimCurveKey lKey2(lKeyTime, 1.0);
363  lKey2.SetInterpolation(FbxAnimCurveDef::eInterpolationConstant);
364  lKey2.SetConstantMode(FbxAnimCurveDef::eConstantStandard);
365  lFCurve->KeySet(1, lKey2);
366 
367  // an other way of setting a keyframe value
368  lKeyTime.SetSecondDouble(2.0);
369 
370  // the cast to TangentMode is intended
371  lFCurve->KeySet(2, lKeyTime, 2.0, FbxAnimCurveDef::eInterpolationConstant, (FbxAnimCurveDef::ETangentMode)FbxAnimCurveDef::eConstantStandard);
372 
373  lKeyTime.SetSecondDouble(3.0);
374  lFCurve->KeySet(3, lKeyTime, 3.0, FbxAnimCurveDef::eInterpolationConstant, (FbxAnimCurveDef::ETangentMode)FbxAnimCurveDef::eConstantStandard);
375 
376  lKeyTime.SetSecondDouble(4.0);
377  lFCurve->KeySet(4, lKeyTime, 0.0, FbxAnimCurveDef::eInterpolationConstant, (FbxAnimCurveDef::ETangentMode)FbxAnimCurveDef::eConstantStandard);
378 
379  lFCurve->KeyModifyEnd();
380  }

384 // Create a position constraint whith pSourceNode as source node and pConstraintedNode as constrained node.
385 FbxConstraintPosition* CreatePositionConstraint(FbxScene* pScene, FbxNode* pSourceNode, FbxNode* pConstrainedNode)
386 {
387  FbxConstraintPosition *lPositionConstraint = FbxConstraintPosition::Create(pScene,"Position");
388 
389  // set constrained object
390  lPositionConstraint->SetConstrainedObject(pConstrainedNode);
391 
392  // set source
393  lPositionConstraint->AddConstraintSource(pSourceNode, 100.0);
394 
395  // Constrain the position in X, Y and Z
396  lPositionConstraint->AffectX = true;
397  lPositionConstraint->AffectY = true;
398  lPositionConstraint->AffectZ = true;
399 
400  // keep offset between source and constrained object
401  FbxVector4 lPositionSource;
402  FbxVector4 lPositionConstrainedObj;
403  lPositionSource = pSourceNode->LclTranslation.Get();
404  lPositionConstrainedObj = pConstrainedNode->LclTranslation.Get();
405  FbxVector4 lOffset = lPositionConstrainedObj - lPositionSource;
406  lPositionConstraint->Translation = FbxDouble3(lOffset);
407 
408  // activate property
409  FbxProperty lActive = lPositionConstraint->FindProperty("Active", false);
410  lActive.Set(true);
411 
412 
413  return lPositionConstraint;
414 }
