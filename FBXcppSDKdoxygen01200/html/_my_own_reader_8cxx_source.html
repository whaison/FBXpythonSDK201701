63 bool MyOwnReader::Read(FbxDocument* pDocument)
64 {
65  if (!pDocument)
66  {
67  GetStatus().SetCode(FbxStatus::eFailure, "Invalid document handle");
68  return false;
69  }
70  FbxScene* lScene = FbxCast<FbxScene>(pDocument);
71  bool lIsAScene = (lScene != NULL);
72  bool lResult = false;
73 
74  if(lIsAScene)
75  {
76  FbxNode* lRootNode = lScene->GetRootNode();
77  FbxNodeAttribute * lRootNodeAttribute = FbxNull::Create(lScene,"");
78  lRootNode->SetNodeAttribute(lRootNodeAttribute);
79 
80  int lSize;
81  char* lBuffer = NULL;
82  if(mFilePointer != NULL)
83  {
84  //To obtain file size
85  fseek (mFilePointer , 0 , SEEK_END);
86  lSize = ftell (mFilePointer);
87  rewind (mFilePointer);
88 
89  //Read file content to a string.
90  lBuffer = (char*) malloc (sizeof(char)*lSize + 1);
91  size_t lRead = fread(lBuffer, 1, lSize, mFilePointer);
92  lBuffer[lRead]='\0';
93  FbxString lString(lBuffer);
94 
95  //Parse the string to get name and relation of Nodes.
96  FbxString lSubString, lChildName, lParentName;
97  FbxNode* lChildNode;
98  FbxNode* lParentNode;
99  FbxNodeAttribute* lChildAttribute;
100  int lEndTokenCount = lString.GetTokenCount("\n");
101 
102  for (int i = 0; i < lEndTokenCount; i++)
103  {
104  lSubString = lString.GetToken(i, "\n");
105  FbxString lNodeString;
106  lChildName = lSubString.GetToken(0, "\"");
107  lParentName = lSubString.GetToken(2, "\"");
108 
109  //Build node hierarchy.
110  if(lParentName == "RootNode")
111  {
112  lChildNode = FbxNode::Create(lScene,lChildName.Buffer());
113  lChildAttribute = FbxNull::Create(mManager,"");
114  lChildNode->SetNodeAttribute(lChildAttribute);
115 
116  lRootNode->AddChild(lChildNode);
117  }
118  else
119  {
120  lChildNode = FbxNode::Create(lScene,lChildName.Buffer());
121  lChildAttribute = FbxNull::Create(lScene,"");
122  lChildNode->SetNodeAttribute(lChildAttribute);
123 
124  lParentNode = lRootNode->FindChild(lParentName.Buffer());
125  lParentNode->AddChild(lChildNode);
126  }
127  }
128  free(lBuffer);
129  }
130  lResult = true;
131  }
132  return lResult;
133 }
