85 void DisplayPolygons(FbxMesh* pMesh)
86 {
87  int i, j, lPolygonCount = pMesh->GetPolygonCount();
88  FbxVector4* lControlPoints = pMesh->GetControlPoints();
89  char header[100];
90 
91  DisplayString(" Polygons");
92 
93  int vertexId = 0;
94  for (i = 0; i < lPolygonCount; i++)
95  {
96  DisplayInt(" Polygon ", i);
97  int l;
98 
99  for (l = 0; l < pMesh->GetElementPolygonGroupCount(); l++)
100  {
101  FbxGeometryElementPolygonGroup* lePolgrp = pMesh->GetElementPolygonGroup(l);
102  switch (lePolgrp->GetMappingMode())
103  {
104  case FbxGeometryElement::eByPolygon:
105  if (lePolgrp->GetReferenceMode() == FbxGeometryElement::eIndex)
106  {
107  FBXSDK_sprintf(header, 100, " Assigned to group: ");
108  int polyGroupId = lePolgrp->GetIndexArray().GetAt(i);
109  DisplayInt(header, polyGroupId);
110  break;
111  }
112  default:
113  // any other mapping modes don't make sense
114  DisplayString(" \"unsupported group assignment\"");
115  break;
116  }
117  }
118 
119  int lPolygonSize = pMesh->GetPolygonSize(i);
120 
121  for (j = 0; j < lPolygonSize; j++)
122  {
123  int lControlPointIndex = pMesh->GetPolygonVertex(i, j);
124 
125  Display3DVector(" Coordinates: ", lControlPoints[lControlPointIndex]);
126 
127  for (l = 0; l < pMesh->GetElementVertexColorCount(); l++)
128  {
129  FbxGeometryElementVertexColor* leVtxc = pMesh->GetElementVertexColor( l);
130  FBXSDK_sprintf(header, 100, " Color vertex: ");
131 
132  switch (leVtxc->GetMappingMode())
133  {
134  default:
135  break;
136  case FbxGeometryElement::eByControlPoint:
137  switch (leVtxc->GetReferenceMode())
138  {
139  case FbxGeometryElement::eDirect:
140  DisplayColor(header, leVtxc->GetDirectArray().GetAt(lControlPointIndex));
141  break;
142  case FbxGeometryElement::eIndexToDirect:
143  {
144  int id = leVtxc->GetIndexArray().GetAt(lControlPointIndex);
145  DisplayColor(header, leVtxc->GetDirectArray().GetAt(id));
146  }
147  break;
148  default:
149  break; // other reference modes not shown here!
150  }
151  break;
152 
153  case FbxGeometryElement::eByPolygonVertex:
154  {
155  switch (leVtxc->GetReferenceMode())
156  {
157  case FbxGeometryElement::eDirect:
158  DisplayColor(header, leVtxc->GetDirectArray().GetAt(vertexId));
159  break;
160  case FbxGeometryElement::eIndexToDirect:
161  {
162  int id = leVtxc->GetIndexArray().GetAt(vertexId);
163  DisplayColor(header, leVtxc->GetDirectArray().GetAt(id));
164  }
165  break;
166  default:
167  break; // other reference modes not shown here!
168  }
169  }
170  break;
171 
172  case FbxGeometryElement::eByPolygon: // doesn't make much sense for UVs
173  case FbxGeometryElement::eAllSame: // doesn't make much sense for UVs
174  case FbxGeometryElement::eNone: // doesn't make much sense for UVs
175  break;
176  }
177  }
178  for (l = 0; l < pMesh->GetElementUVCount(); ++l)
179  {
180  FbxGeometryElementUV* leUV = pMesh->GetElementUV( l);
181  FBXSDK_sprintf(header, 100, " Texture UV: ");
182 
183  switch (leUV->GetMappingMode())
184  {
185  default:
186  break;
187  case FbxGeometryElement::eByControlPoint:
188  switch (leUV->GetReferenceMode())
189  {
190  case FbxGeometryElement::eDirect:
191  Display2DVector(header, leUV->GetDirectArray().GetAt(lControlPointIndex));
192  break;
193  case FbxGeometryElement::eIndexToDirect:
194  {
195  int id = leUV->GetIndexArray().GetAt(lControlPointIndex);
196  Display2DVector(header, leUV->GetDirectArray().GetAt(id));
197  }
198  break;
199  default:
200  break; // other reference modes not shown here!
201  }
202  break;
203 
204  case FbxGeometryElement::eByPolygonVertex:
205  {
206  int lTextureUVIndex = pMesh->GetTextureUVIndex(i, j);
207  switch (leUV->GetReferenceMode())
208  {
209  case FbxGeometryElement::eDirect:
210  case FbxGeometryElement::eIndexToDirect:
211  {
212  Display2DVector(header, leUV->GetDirectArray().GetAt(lTextureUVIndex));
213  }
214  break;
215  default:
216  break; // other reference modes not shown here!
217  }
218  }
219  break;
220 
221  case FbxGeometryElement::eByPolygon: // doesn't make much sense for UVs
222  case FbxGeometryElement::eAllSame: // doesn't make much sense for UVs
223  case FbxGeometryElement::eNone: // doesn't make much sense for UVs
224  break;
225  }
226  }
227  for( l = 0; l < pMesh->GetElementNormalCount(); ++l)
228  {
229  FbxGeometryElementNormal* leNormal = pMesh->GetElementNormal( l);
230  FBXSDK_sprintf(header, 100, " Normal: ");
231

236  case FbxGeometryElement::eDirect:
237  Display3DVector(header, leNormal->GetDirectArray().GetAt(vertexId));

239  case FbxGeometryElement::eIndexToDirect:
240  {

242  Display3DVector(header, leNormal->GetDirectArray().GetAt(id));
243  }
244  break;
245  default:
246  break; // other reference modes not shown here!
247  }
248  }
249 
250  }

252  {
253  FbxGeometryElementTangent* leTangent = pMesh->GetElementTangent( l);
254  FBXSDK_sprintf(header, 100, " Tangent: ");
255 
256  if(leTangent->GetMappingMode() == FbxGeometryElement::eByPolygonVertex)
257  {
258  switch (leTangent->GetReferenceMode())
259  {
260  case FbxGeometryElement::eDirect:
261  Display3DVector(header, leTangent->GetDirectArray().GetAt(vertexId));

277 
278  FbxGeometryElementBinormal* leBinormal = pMesh->GetElementBinormal( l);
279 
280  FBXSDK_sprintf(header, 100, " Binormal: ");

285  case FbxGeometryElement::eDirect:

303 
304  //check visibility for the edges of the mesh

332  if(lLayeredTextureCount > 0)
333  {
334  for(int j=0; j<lLayeredTextureCount; ++j)
335  {
336  FbxLayeredTexture *lLayeredTexture = pProperty.GetSrcObject<FbxLayeredTexture>(j);
337  int lNbTextures = lLayeredTexture->GetSrcObjectCount<FbxTexture>();
338  pConnectionString += " Texture ";
339 
340  for(int k =0; k<lNbTextures; ++k)
341  {
342  //lConnectionString += k;
343  pConnectionString += "\"";
344  pConnectionString += (char*)lLayeredTexture->GetName();
345  pConnectionString += "\"";
346  pConnectionString += " ";
347  }
348  pConnectionString += "of ";
349  pConnectionString += pProperty.GetName();
350  pConnectionString += " on layer ";
351  pConnectionString += j;
352  }
353  pConnectionString += " |";
354  }
355  else
356  {
357  //no layered texture simply get on the property
358  int lNbTextures = pProperty.GetSrcObjectCount<FbxTexture>();
359 
360  if(lNbTextures > 0)
361  {
362  pConnectionString += " Texture ";
363  pConnectionString += " ";
364 
365  for(int j =0; j<lNbTextures; ++j)
366  {
367  FbxTexture* lTexture = pProperty.GetSrcObject<FbxTexture>(j);
368  if(lTexture)
369  {
370  pConnectionString += "\"";
371  pConnectionString += (char*)lTexture->GetName();
372  pConnectionString += "\"";
373  pConnectionString += " ";
374  }
375  }
376  pConnectionString += "of ";
377  pConnectionString += pProperty.GetName();
378  pConnectionString += " |";
379  }

465 void DisplayMaterialConnections(FbxMesh* pMesh)
466 {
467  int i, l, lPolygonCount = pMesh->GetPolygonCount();
468 
469  char header[MAT_HEADER_LENGTH];
470 
471  DisplayString(" Polygons Material Connections");
472 
473  //check whether the material maps with only one mesh
474  bool lIsAllSame = true;
475  for (l = 0; l < pMesh->GetElementMaterialCount(); l++)
476  {
477 
478  FbxGeometryElementMaterial* lMaterialElement = pMesh->GetElementMaterial(l);
479  if( lMaterialElement->GetMappingMode() == FbxGeometryElement::eByPolygon)
480  {
481  lIsAllSame = false;
482  break;
483  }
484  }
485 
486  //For eAllSame mapping type, just out the material and texture mapping info once
487  if(lIsAllSame)
488  {
489  for (l = 0; l < pMesh->GetElementMaterialCount(); l++)
490  {
491 
492  FbxGeometryElementMaterial* lMaterialElement = pMesh->GetElementMaterial( l);
493  if( lMaterialElement->GetMappingMode() == FbxGeometryElement::eAllSame)
494  {
495  FbxSurfaceMaterial* lMaterial = pMesh->GetNode()->GetMaterial(lMaterialElement->GetIndexArray().GetAt(0));
496  int lMatId = lMaterialElement->GetIndexArray().GetAt(0);
497  if(lMatId >= 0)
498  {
499  DisplayInt(" All polygons share the same material in mesh ", l);
500  DisplayMaterialTextureConnections(lMaterial, header, lMatId, l);
501  }
502  }
503  }
504 
505  //no material
506  if(l == 0)
507  DisplayString(" no material applied");
508  }
509 
510  //For eByPolygon mapping type, just out the material and texture mapping info once
511  else
512  {
513  for (i = 0; i < lPolygonCount; i++)
514  {
515  DisplayInt(" Polygon ", i);
516 
517  for (l = 0; l < pMesh->GetElementMaterialCount(); l++)
518  {
519 
520  FbxGeometryElementMaterial* lMaterialElement = pMesh->GetElementMaterial( l);
521  FbxSurfaceMaterial* lMaterial = NULL;
522  int lMatId = -1;
523  lMaterial = pMesh->GetNode()->GetMaterial(lMaterialElement->GetIndexArray().GetAt(i));
524  lMatId = lMaterialElement->GetIndexArray().GetAt(i);
525 
526  if(lMatId >= 0)
527  {
528  DisplayMaterialTextureConnections(lMaterial, header, lMatId, l);
529  }
530  }
531  }
532  }

564

566  leMat->GetReferenceMode() == FbxGeometryElement::eIndexToDirect)
567  {
568  lMaterialCount = lMtrlCount;
569  }
570
