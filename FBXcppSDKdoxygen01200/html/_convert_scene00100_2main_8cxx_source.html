13 //
14 // This program converts any file in a format supported by the FBX SDK
15 // into DAE, FBX, 3DS, OBJ and DXF files.
16 //
17 // Steps:
18 // 1. Initialize SDK objects.
19 // 2. Load a file(fbx, obj,...) to a FBX scene.
20 // 3. Create a exporter.
21 // 4. Retrieve the writer ID according to the description of file format.
22 // 5. Initialize exporter with specified file format
23 // 6. Export.
24 // 7. Destroy the exporter
25 // 8. Destroy the FBX SDK manager
26 //
28

52 int main(int argc, char** argv)
53 {
54  FbxString lFilePath("");
55  for( int i = 1, c = argc; i < c; ++i )
56  {
57  if( FbxString(argv[i]) == "-test" ) continue;
58  else if( lFilePath.IsEmpty() ) lFilePath = argv[i];
59  }
60  if( lFilePath.IsEmpty() ) lFilePath = SAMPLE_FILENAME;
61 
62  FbxManager* lSdkManager = NULL;
63  FbxScene* lScene = NULL;
64 
65  // Prepare the FBX SDK.
66  InitializeSdkObjects(lSdkManager, lScene);
67 
68  bool lResult = LoadScene(lSdkManager, lScene, lFilePath.Buffer());
69  if( lResult )
70  {
71  const size_t lFileNameLength = strlen((argc>=3)?argv[2]:lFilePath.Buffer());
72  char* lNewFileName = new char[lFileNameLength+64];
73  FBXSDK_strcpy(lNewFileName,lFileNameLength+64,(argc>=3)?argv[2]:lFilePath.Buffer());
74 
75  const size_t lFileTypeCount = sizeof(lFileTypes)/sizeof(lFileTypes[0])/2;
76 
77  for(size_t i=0; i<lFileTypeCount; ++i)
78  {
79  // Retrieve the writer ID according to the description of file format.
80  int lFormat = lSdkManager->GetIOPluginRegistry()->FindWriterIDByDescription(lFileTypes[i*2+1]);
81 
82  // Construct the output file name.
83  FBXSDK_strcpy(lNewFileName+lFileNameLength-4,60, lFileTypes[i*2]);
84 
85  // Create an exporter.
86  FbxExporter* lExporter = FbxExporter::Create(lSdkManager, "");
87 
88  // Initialize the exporter.
89  lResult = lExporter->Initialize(lNewFileName, lFormat, lSdkManager->GetIOSettings());
90  if( !lResult )
91  {
92  FBXSDK_printf("%s:\tCall to FbxExporter::Initialize() failed.\n", lFileTypes[i*2+1]);
93  FBXSDK_printf("Error returned: %s\n\n", lExporter->GetStatus().GetErrorString());
94  }
95  else
96  {
97  // Export the scene.
98  lResult = lExporter->Export(lScene);
99  if( !lResult )
100  {
101  FBXSDK_printf("Call to FbxExporter::Export() failed.\n");
102  }
103  }
104 
105  // Destroy the exporter.
106  lExporter->Destroy();
107  }
108  delete[] lNewFileName;
109  }
110  else
111  {
112  FBXSDK_printf("Call to LoadScene() failed.\n");
113  }
114 
115  // Delete the FBX SDK manager. All the objects that have been allocated
116  // using the FBX SDK manager and that haven't been explicitly destroyed
117  // are automatically destroyed at the same time.
118  DestroySdkObjects(lSdkManager, lResult);
119  return 0;
120 }
