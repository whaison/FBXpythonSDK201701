87 // Create a cube.
88 FbxNode* CreateCube(FbxScene* pScene, const char* pName)
89 {
90  // indices of the vertices per each polygon
91  static int vtxId[24] = {
92  0,1,2,3, // front face (Z+)
93  1,5,6,2, // right side (X+)
94  5,4,7,6, // back face (Z-)
95  4,0,3,7, // left side (X-)
96  0,4,5,1, // bottom face (Y-)
97  3,2,6,7 // top face (Y+)
98  };
99 
100  // control points
101  static Vector4 lControlPoints[8] = {
102  { -50.0, 0.0, 50.0, 1.0}, { 50.0, 0.0, 50.0, 1.0}, { 50.0,100.0, 50.0, 1.0}, { -50.0,100.0, 50.0, 1.0},
103  { -50.0, 0.0, -50.0, 1.0}, { 50.0, 0.0, -50.0, 1.0}, { 50.0,100.0, -50.0, 1.0}, { -50.0,100.0, -50.0, 1.0}
104  };
105 
106  // normals
107  static Vector4 lNormals[8] = {
108  {-0.577350258827209,-0.577350258827209, 0.577350258827209, 1.0},
109  { 0.577350258827209,-0.577350258827209, 0.577350258827209, 1.0},
110  { 0.577350258827209, 0.577350258827209, 0.577350258827209, 1.0},
111  {-0.577350258827209, 0.577350258827209, 0.577350258827209, 1.0},
112  {-0.577350258827209,-0.577350258827209,-0.577350258827209, 1.0},
113  { 0.577350258827209,-0.577350258827209,-0.577350258827209, 1.0},
114  { 0.577350258827209, 0.577350258827209,-0.577350258827209, 1.0},
115  {-0.577350258827209, 0.577350258827209,-0.577350258827209, 1.0}
116  };
117 
118  // 2 uvs
119  static Vector2 lUVs0[14] = {
120  { 0.0, 0.0},
121  { 1.0, 0.0},
122  { 0.0, 1.0},
123  { 1.0, 1.0},
124  { 0.0, 2.0},
125  { 1.0, 2.0},
126  { 0.0, 3.0},
127  { 1.0, 3.0},
128  { 0.0, 4.0},
129  { 1.0, 4.0},
130  { 2.0, 0.0},
131  { 2.0, 1.0},
132  {-1.0, 0.0},
133  {-1.0, 1.0}
134  };
135 
136  static Vector2 lUVs1[14] = {
137  { 0.0, 1.0},
138  { 1.0, 0.0},
139  { 0.0, 0.0},
140  { 1.0, 1.0}
141  };
142 
143  // indices of the uvs per each polygon
144  static int uvsId[24] = {
145  0,1,3,2,2,3,5,4,4,5,7,6,6,7,9,8,1,10,11,3,12,0,2,13
146  };
147 
148  // colors
149  static Vector4 lColors[8] = {
150  // colors used for the materials
151  {1.0, 1.0, 1.0, 1.0},
152  {1.0, 1.0, 0.0, 1.0},
153  {1.0, 0.0, 1.0, 1.0},
154  {0.0, 1.0, 1.0, 1.0},
155  {0.0, 0.0, 1.0, 1.0},
156  {1.0, 0.0, 0.0, 1.0},
157  {0.0, 1.0, 0.0, 1.0},
158  {0.0, 0.0, 0.0, 1.0},
159  };
160 
161  // create the main structure.
162  FbxMesh* lMesh = FbxMesh::Create(pScene,"");
163 
164  // Create control points.
165  lMesh->InitControlPoints(8);
166  FbxVector4* vertex = lMesh->GetControlPoints();
167  memcpy((void*)vertex, (void*)lControlPoints, 8*sizeof(FbxVector4));
168 
169  // create the materials.
170  /* Each polygon face will be assigned a unique material.
171  */
172  FbxGeometryElementMaterial* lMaterialElement = lMesh->CreateElementMaterial();
173  lMaterialElement->SetMappingMode(FbxGeometryElement::eByPolygon);
174  lMaterialElement->SetReferenceMode(FbxGeometryElement::eIndexToDirect);
175 
176  // Create polygons later after FbxGeometryElementMaterial is created. Assign material indices.
177  int vId = 0;
178  for (int f=0; f<6; f++)
179  {
180  lMesh->BeginPolygon(f);//Material index.
181  for (int v=0; v<4; v++)
182  lMesh->AddPolygon(vtxId[vId++]);
183  lMesh->EndPolygon();
184  }
185 
186  // specify normals per control point.
187  FbxGeometryElementNormal* lNormalElement = lMesh->CreateElementNormal();
188  lNormalElement->SetMappingMode(FbxGeometryElement::eByControlPoint);
189  lNormalElement->SetReferenceMode(FbxGeometryElement::eDirect);
190 
191  for (int n=0; n<8; n++)
192  lNormalElement->GetDirectArray().Add(FbxVector4(lNormals[n][0], lNormals[n][1], lNormals[n][2]));
193 
194 
195  // create color vertices
196  /* We choose to define one color per control point. The other choice would
197  have been to use the eByPolygonVertex mapping mode. In this second case,
198  the reference mode should become eIndexToDirect.
199  */
200  FbxGeometryElementVertexColor* lVertexColorElement = lMesh->CreateElementVertexColor();
201  lVertexColorElement->SetMappingMode(FbxGeometryElement::eByControlPoint);
202  lVertexColorElement->SetReferenceMode(FbxGeometryElement::eDirect);
203 
204  for (int c=0; c<8; c++)
205  lVertexColorElement->GetDirectArray().Add(FbxColor(lColors[c][0]*.8, lColors[c][1]*0.8, lColors[c][2]*.8, lColors[c][3]*.8));
206 
207 
208 
209  // create polygroups.
210  /* We are going to make a first group with the 4 sides.
211  And a second group with the top and bottom sides.
212 
213  NOTE that the only reference mode allowed is eIndex
214  */
215  FbxGeometryElementPolygonGroup* lPolygonGroupElement = lMesh->CreateElementPolygonGroup();
216  lPolygonGroupElement->SetMappingMode(FbxGeometryElement::eByPolygon);
217  lPolygonGroupElement->SetReferenceMode(FbxGeometryElement::eIndex);
218  lPolygonGroupElement->GetIndexArray().Add(0); // front face assigned to group 0
219  lPolygonGroupElement->GetIndexArray().Add(0); // right side assigned to group 0
220  lPolygonGroupElement->GetIndexArray().Add(0); // back face assigned to group 0
221  lPolygonGroupElement->GetIndexArray().Add(0); // left side assigned to group 0
222  lPolygonGroupElement->GetIndexArray().Add(1); // bottom face assigned to group 1
223  lPolygonGroupElement->GetIndexArray().Add(1); // top face assigned to group 1
224 
225 
226  // create the UV textures mapping.
227  FbxTexture* lTexture[3];
228  FbxLayeredTexture::EBlendMode lBlendMode[3];
229 
230  // On layer 0 all the faces have the same texture
231  FbxGeometryElementUV* lUVElement0 = lMesh->CreateElementUV( BACKGROUND_IMAGE_NAME);
232  FBX_ASSERT( lUVElement0 != NULL);
233  lUVElement0->SetMappingMode(FbxGeometryElement::eByPolygonVertex);
234  lUVElement0->SetReferenceMode(FbxGeometryElement::eIndexToDirect);
235 
236  int i;
237  for (i = 0; i<14; i++)
238  lUVElement0->GetDirectArray().Add(FbxVector2(lUVs0[i][0], lUVs0[i][1]));
239 
240  for (i = 0; i<24; i++)
241  lUVElement0->GetIndexArray().Add(uvsId[i]);
242 
243  // Create the node containing the mesh
244  FbxNode* lNode = FbxNode::Create(pScene,pName);
245 
246  lNode->SetNodeAttribute(lMesh);
247  lNode->SetShadingMode(FbxNode::eTextureShading);
248 
249 
250  // Put a different material on each polygon
251  FbxSurfacePhong* lMaterial[6];
252  for (i = 0; i < 6; i++ )
253  {
254  FbxString lMaterialName = "material";
255  lMaterialName += i;
256 
257  lMaterial[i] = FbxSurfacePhong::Create(pScene,lMaterialName.Buffer());
258 
259  // Generate primary and secondary colors.
260  lMaterial[i]->Emissive.Set(FbxDouble3(0.0, 0.0, 0.0));
261  lMaterial[i]->Ambient.Set(FbxDouble3(lColors[i][0], lColors[i][1], lColors[i][2]));
262  lMaterial[i]->Diffuse.Set(FbxDouble3(1.0, 1.0, 1.0));
263  lMaterial[i]->Specular.Set(FbxDouble3(0.0, 0.0, 0.0));
264  lMaterial[i]->TransparencyFactor.Set(0.0);
265  lMaterial[i]->Shininess.Set(0.5);
266  lMaterial[i]->ShadingModel.Set(FbxString("phong"));
267 
268  // add materials to the node
269  lNode->AddMaterial(lMaterial[i]);
270  }
271 
272  // Create textures and texture mappings.
273  lTexture[0] = CreateTexture(pScene, BACKGROUND_IMAGE_NAME, BACKGROUND_IMAGE);
274 
275  lBlendMode[0] = FbxLayeredTexture::eTranslucent;
276 
277  // create second UVset
278  FbxGeometryElementUV* lUVElement1 = lMesh->CreateElementUV( GEO1_IMAGE_NAME);
279  FBX_ASSERT( lUVElement1 != NULL);
280  lUVElement1->SetMappingMode(FbxGeometryElement::eByPolygonVertex);
281  lUVElement1->SetReferenceMode(FbxGeometryElement::eIndexToDirect);
282  for (i = 0; i <4; i++)
283  lUVElement1->GetDirectArray().Add(FbxVector2(lUVs1[i][0], lUVs1[i][1]));
284 
285  for (i = 0; i<24; i++)
286  lUVElement1->GetIndexArray().Add(uvsId[i%4]);
287 
288  lTexture[1] = CreateTexture(pScene, GEO1_IMAGE_NAME, GEO1_IMAGE);
289  lBlendMode[1] = FbxLayeredTexture::eModulate;
290 
291 
292 
293  FbxGeometryElementUV* lUVElement2 = lMesh->CreateElementUV( GEO2_IMAGE_NAME);
294  FBX_ASSERT( lUVElement2 != NULL);
295  // we re-use the UV mapping.
296  *lUVElement2 = *lUVElement0;
297 
298  lTexture[2] = CreateTexture(pScene, GEO2_IMAGE_NAME, GEO2_IMAGE);
299 
300  lBlendMode[2] = FbxLayeredTexture::eModulate;
301 
302  // Because we can only connect one texture to the material propery, we need
303  // to use a layered texture object to connect the multiple textures created above
304  // we know that 5 faces of the cube will use 2 textures (lTexture[0] and lTexture[1])
305  // and only one face uses the three textures. Therefore we need two layered textures.
306 
307  FbxLayeredTexture* lLayeredTexture[2];
308 
309  // the lLayeredTexure[1] is used for the 5 faces with two textures and lLayeredTexture[0] for
310  // the only face that uses the three textures
311  lLayeredTexture[0] = FbxLayeredTexture::Create(pScene, "layeredTexture0");
312  lLayeredTexture[1] = FbxLayeredTexture::Create(pScene, "layeredTexture1");
313 
314  // the first connected texture is the bottom one!
315  for (i = 0; i < 3; i++)
316  {
317  lLayeredTexture[0]->ConnectSrcObject(lTexture[i]);
318  lLayeredTexture[0]->SetTextureBlendMode(i, lBlendMode[i]);
319  if (i < 2)
320  {
321  lLayeredTexture[1]->ConnectSrcObject(lTexture[i]);
322  lLayeredTexture[1]->SetTextureBlendMode(i, lBlendMode[i]);
323  }
324  }
325 
326  // connect the layered textures to the 6 materials allocated before (material[0] is the
327  // material connected to the face 0 of the cube so it is the one that will have lLayeredTexture[0].
328  lMaterial[0]->Diffuse.ConnectSrcObject(lLayeredTexture[0]);
329  for (i = 1; i < 6; i++)
330  lMaterial[i]->Diffuse.ConnectSrcObject(lLayeredTexture[1]);
331 
332  return lNode;
333 }

336 FbxTexture* CreateTexture(FbxScene* pScene, const char* name, const char* filename)
337 {
338  FbxFileTexture* lTexture = FbxFileTexture::Create(pScene,name);

341  lTexture->SetMappingType(FbxTexture::eUV);
342  lTexture->SetMaterialUse(FbxFileTexture::eModelMaterial);
343  lTexture->SetSwapUV(false);
344  lTexture->SetTranslation(0.0, 0.0);
345  lTexture->SetScale(1.0, 1.0);
346  lTexture->SetRotation(0.0, 0.0);
347  return lTexture;
348 }
