63 // to get the filters for the <Open file> dialog (description + file extention)
64 const char *GetReaderOFNFilters()
65 {
66  int nbReaders = gSdkManager->GetIOPluginRegistry()->GetReaderFormatCount();;
67 
68  FbxString s;
69  int i = 0;
70 
71  for(i=0; i < nbReaders; i++)
72  {
73  s += gSdkManager->GetIOPluginRegistry()->GetReaderFormatDescription(i);
74  s += "|*.";
75  s += gSdkManager->GetIOPluginRegistry()->GetReaderFormatExtension(i);
76  s += "|";
77  }
78 
79  // replace | by \0
80  int nbChar = int(strlen(s.Buffer())) + 1;
81  char *filter = new char[ nbChar ];
82  memset(filter, 0, nbChar);
83 
84  FBXSDK_strcpy(filter, nbChar, s.Buffer());
85 
86  for(i=0; i < int(strlen(s.Buffer())); i++)
87  {
88  if(filter[i] == '|')
89  {
90  filter[i] = 0;
91  }
92  }
93 
94  // the caller must delete this allocated memory
95  return filter;
96 }

98 // to read a file using an FBXSDK reader
99 bool LoadScene(
100  FbxManager *pSdkManager,
101  FbxScene *pScene,
102  const char *pFbxFilePath
103  )
104 {
105  bool lStatus;
106 
107  // Create an importer.
108  FbxImporter* lImporter = FbxImporter::Create(pSdkManager,"");
109 
110  // Initialize the importer by providing a filename.
111  bool lImportStatus = lImporter->Initialize(pFbxFilePath, -1, pSdkManager->GetIOSettings() );
112 
113  if( !lImportStatus )
114  {
115  // Destroy the importer
116  lImporter->Destroy();
117  return false;
118  }
119 
120  if (lImporter->IsFBX())
121  {
122  // Set the import states. By default, the import states are always set to
123  // true. The code below shows how to change these states.
124  IOS_REF.SetBoolProp(IMP_FBX_MATERIAL, true);
125  IOS_REF.SetBoolProp(IMP_FBX_TEXTURE, true);
126  IOS_REF.SetBoolProp(IMP_FBX_LINK, true);
127  IOS_REF.SetBoolProp(IMP_FBX_SHAPE, true);
128  IOS_REF.SetBoolProp(IMP_FBX_GOBO, true);
129  IOS_REF.SetBoolProp(IMP_FBX_ANIMATION, true);
130  IOS_REF.SetBoolProp(IMP_FBX_GLOBAL_SETTINGS, true);
131  }
132 
133  // Import the scene
134  lStatus = lImporter->Import(pScene);
135 
136  // Destroy the importer
137  lImporter->Destroy();
138 
139  return lStatus;
140 }

154 // to get a string from the node name and attribute type
155 FbxString GetNodeNameAndAttributeTypeName(const FbxNode* pNode)
156 {
157  FbxString s = pNode->GetName();
158 
159  FbxNodeAttribute::EType lAttributeType;
160 
161  if(pNode->GetNodeAttribute() == NULL)
162  {
163  s += " (No node attribute type)";
164  }
165  else
166  {
167  lAttributeType = (pNode->GetNodeAttribute()->GetAttributeType());
168 
169  switch (lAttributeType)
170  {
171  case FbxNodeAttribute::eMarker: s += " (Marker)"; break;
172  case FbxNodeAttribute::eSkeleton: s += " (Skeleton)"; break;
173  case FbxNodeAttribute::eMesh: s += " (Mesh)"; break;
174  case FbxNodeAttribute::eCamera: s += " (Camera)"; break;
175  case FbxNodeAttribute::eLight: s += " (Light)"; break;
176  case FbxNodeAttribute::eBoundary: s += " (Boundary)"; break;
177  case FbxNodeAttribute::eOpticalMarker: s += " (Optical marker)"; break;
178  case FbxNodeAttribute::eOpticalReference: s += " (Optical reference)"; break;
179  case FbxNodeAttribute::eCameraSwitcher: s += " (Camera switcher)"; break;
180  case FbxNodeAttribute::eNull: s += " (Null)"; break;
181  case FbxNodeAttribute::ePatch: s += " (Patch)"; break;
182  case FbxNodeAttribute::eNurbs: s += " (NURB)"; break;
183  case FbxNodeAttribute::eNurbsSurface: s += " (Nurbs surface)"; break;
184  case FbxNodeAttribute::eNurbsCurve: s += " (NURBS curve)"; break;
185  case FbxNodeAttribute::eTrimNurbsSurface: s += " (Trim nurbs surface)"; break;
186  case FbxNodeAttribute::eUnknown: s += " (Unidentified)"; break;
187  }
188  }
189 
190  return s;
191 }
