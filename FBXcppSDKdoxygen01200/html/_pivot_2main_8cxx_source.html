13 //
14 // Illustrates how to get/set pivots, how to convert pivots
15 // and how to query local and global transform.
16 //
17 //Steps:
18 // 1. Initialize FBX SDK object.
19 // 2. Create default animation stack and animation layer.
20 // 3. Create a pyramid mesh and attach it to a node.
21 // 4. Set pivots.
22 // 5. Add animation to the pyramid node.
23 // 6. Evaluate the local and global transform.
24 // 7. Save the scene before pivot converting.
25 // 8. Convert the animation to reset pivots.
26 // 9. Save the scene after pivot converting.
27 //
29 
30 #include <fbxsdk.h>
31 #include "../Common/Common.h"
32 #include "../Common/AnimationUtility.h"
33 #include "../Common/GeometryUtility.h"
34 
35 const char * SAMPLE_FILENAME_BEFORE_CONVECTION = "pivot_before_convection.fbx";
36 const char * SAMPLE_FILENAME_AFTER_CONVECTION = "pivot_after_convection.fbx";
37 
38 const char * PYRAMID_NAME = "Pyramid";
39 
40 const double KEY_TIME[] = {0.0, 0.5, 1.0};
41 const float KEY_VALUE[] = {0.0, 90.0, 180.0};
42 
43 int main(int /*argc*/, char** /*argv*/)
44 {
45  FbxManager* lSdkManager = NULL;
46  FbxScene* lScene = NULL;
47 
48  // Prepare the FBX SDK.
49  InitializeSdkObjects(lSdkManager, lScene);
50 
51  // Create a animation stack and layer for the scene
52  FbxAnimStack * lAnimStack;
53  FbxAnimLayer * lAnimLayer = CreateDefaultAnimStackAndLayer(lScene, lAnimStack);
54 
55  // Create a pyramid mesh whose bottom width is 4 and height is 4.
56  // Attach this pyramid to a node and as a child of the root node of the scene
57  FbxNode * lPyramidNode = CreatePyramid(lScene, PYRAMID_NAME, 4, 4);
58 
59  // Enable pivot
60  lPyramidNode->SetRotationActive(true);
61  // Set the rotation pivot at the center of the pyramid
62  lPyramidNode->SetRotationPivot(FbxNode::eSourcePivot, FbxVector4(0, 2, 0));
63  // Add a post rotation for the pyramid
64  lPyramidNode->SetPostRotation(FbxNode::eSourcePivot, FbxVector4(0, 0, -90));
65 
66  // Animate the Y channel of the local rotation
67  FbxAnimCurve * lAnimCurve = lPyramidNode->LclRotation.GetCurve(lAnimLayer, "Y", true);
68  if (lAnimCurve)
69  {
70  const int lKeyCount = sizeof(KEY_TIME)/sizeof(double);
71  for (int lKeyIndex = 0; lKeyIndex < lKeyCount; ++lKeyIndex)
72  {
73  FbxTime lTime;
74  FbxAnimCurveKey lKey;
75  lTime.SetSecondDouble(KEY_TIME[lKeyIndex]);
76  lKey.Set(lTime, KEY_VALUE[lKeyIndex]);
77  lAnimCurve->KeyAdd(lTime, lKey);
78  }
79  }
80 
81  // Query the local transform and global transform of the pyramid node at 0.5 second
82  FbxTime lTime;
83  lTime.SetSecondDouble(0.5);
84  FbxAMatrix lLocalTransform = lPyramidNode->EvaluateLocalTransform(lTime);
85  FbxAMatrix lGlobalTransform = lPyramidNode->EvaluateGlobalTransform(lTime);
86 
87  // Save the scene before pivot converting
88  bool lResult = SaveScene(lSdkManager, lScene, SAMPLE_FILENAME_BEFORE_CONVECTION);
89 
90  // Set the target of pivot converting
91  // Reset the rotation pivot and post rotation, and maintain the animation
92  lPyramidNode->SetPivotState(FbxNode::eSourcePivot, FbxNode::ePivotActive);
93  lPyramidNode->SetPivotState(FbxNode::eDestinationPivot, FbxNode::ePivotActive);
94  lPyramidNode->SetPostRotation(FbxNode::eDestinationPivot, FbxVector4(0, 0, 0));
95  lPyramidNode->SetRotationPivot(FbxNode::eDestinationPivot, FbxVector4(0, 0, 0));
96 
97  // Convert the animation between source pivot set and destination pivot set with a frame rate of 30 per second
98  lScene->GetRootNode()->ConvertPivotAnimationRecursive(lAnimStack, FbxNode::eDestinationPivot, 30.0);
99 
100  // Copy the rotation pivot and post rotation from destination set to source set in order to save them in file
101  lPyramidNode->SetRotationPivot(FbxNode::eSourcePivot, FbxVector4(0, 0, 0));
102  lPyramidNode->SetPostRotation(FbxNode::eSourcePivot, FbxVector4(0, 0, 0));
103 
104  // Save the scene after pivot converting
105  lResult = SaveScene(lSdkManager, lScene, SAMPLE_FILENAME_AFTER_CONVECTION);
106 
107  // Destroy all objects created by the FBX SDK.
108  DestroySdkObjects(lSdkManager, lResult);
109  return 0;
110 }
