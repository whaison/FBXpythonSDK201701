97 bool CreateScene(FbxManager *pSdkManager, FbxScene* pScene)
98 {
99  // create scene info
100  FbxDocumentInfo* sceneInfo = FbxDocumentInfo::Create(pSdkManager,"SceneInfo");
101  sceneInfo->mTitle = "Example scene";
102  sceneInfo->mSubject = "Illustrates the creation and animation of a deformed cylinder.";
103  sceneInfo->mAuthor = "ExportScene01.exe sample program.";
104  sceneInfo->mRevision = "rev. 1.0";
105  sceneInfo->mKeywords = "deformed cylinder";
106  sceneInfo->mComment = "no particular comments required.";
107 
108  // we need to add the sceneInfo before calling AddThumbNailToScene because
109  // that function is asking the scene for the sceneInfo.
110  pScene->SetSceneInfo(sceneInfo);
111 
112  AddThumbnailToScene(pScene);
113 
114  FbxNode* lPatch = CreatePatch(pScene, "Patch");
115  FbxNode* lSkeletonRoot = CreateSkeleton(pScene, "Skeleton");
116 
117 
118  // Build the node tree.
119  FbxNode* lRootNode = pScene->GetRootNode();
120  lRootNode->AddChild(lPatch);
121  lRootNode->AddChild(lSkeletonRoot);
122 
123  // Store poses
124  LinkPatchToSkeleton(pScene, lPatch, lSkeletonRoot);
125  StoreBindPose(pScene, lPatch);
126  StoreRestPose(pScene, lSkeletonRoot);
127 
128  // Animation
129  AnimateSkeleton(pScene, lSkeletonRoot);
130 
131  return true;
132 }

134 // Create a cylinder centered on the Z axis.
135 FbxNode* CreatePatch(FbxScene* pScene, const char* pName)
136 {
137  FbxPatch* lPatch = FbxPatch::Create(pScene,pName);
138 
139  // Set patch properties.
140  lPatch->InitControlPoints(4, FbxPatch::eBSpline, 7, FbxPatch::eBSpline);
141  lPatch->SetStep(4, 4);
142  lPatch->SetClosed(true, false);
143 
144  FbxVector4* lVector4 = lPatch->GetControlPoints();
145  int i;
146 
147  for (i = 0; i < 7; i++)
148  {
149  double lRadius = 15.0;
150  double lSegmentLength = 20.0;
151  lVector4[4*i + 0].Set(lRadius, 0.0, (i-3)*lSegmentLength);
152  lVector4[4*i + 1].Set(0.0, -lRadius, (i-3)*lSegmentLength);
153  lVector4[4*i + 2].Set(-lRadius, 0.0, (i-3)*lSegmentLength);
154  lVector4[4*i + 3].Set(0.0, lRadius, (i-3)*lSegmentLength);
155  }
156 
157  FbxNode* lNode = FbxNode::Create(pScene,pName);
158 
159  // Rotate the cylinder along the X axis so the axis
160  // of the cylinder is the same as the bone axis (Y axis)
161  FbxVector4 lR(-90.0, 0.0, 0.0);
162  lNode->LclRotation.Set(lR);
163  lNode->SetNodeAttribute(lPatch);
164 
165  return lNode;
166 }

169 FbxNode* CreateSkeleton(FbxScene* pScene, const char* pName)
170 {
171  // Create skeleton root.
172  FbxString lRootName(pName);
173  lRootName += "Root";
174  FbxSkeleton* lSkeletonRootAttribute = FbxSkeleton::Create(pScene, pName);
175  lSkeletonRootAttribute->SetSkeletonType(FbxSkeleton::eRoot);
176  FbxNode* lSkeletonRoot = FbxNode::Create(pScene,lRootName.Buffer());
177  lSkeletonRoot->SetNodeAttribute(lSkeletonRootAttribute);
178  lSkeletonRoot->LclTranslation.Set(FbxVector4(0.0, -40.0, 0.0));
179 
180  // Create skeleton first limb node.
181  FbxString lLimbNodeName1(pName);
182  lLimbNodeName1 += "LimbNode1";
183  FbxSkeleton* lSkeletonLimbNodeAttribute1 = FbxSkeleton::Create(pScene,lLimbNodeName1);
184  lSkeletonLimbNodeAttribute1->SetSkeletonType(FbxSkeleton::eLimbNode);
185  lSkeletonLimbNodeAttribute1->Size.Set(1.0);
186  FbxNode* lSkeletonLimbNode1 = FbxNode::Create(pScene,lLimbNodeName1.Buffer());
187  lSkeletonLimbNode1->SetNodeAttribute(lSkeletonLimbNodeAttribute1);
188  lSkeletonLimbNode1->LclTranslation.Set(FbxVector4(0.0, 40.0, 0.0));
189 
190  // Create skeleton second limb node.
191  FbxString lLimbNodeName2(pName);
192  lLimbNodeName2 += "LimbNode2";
193  FbxSkeleton* lSkeletonLimbNodeAttribute2 = FbxSkeleton::Create(pScene,lLimbNodeName2);
194  lSkeletonLimbNodeAttribute2->SetSkeletonType(FbxSkeleton::eLimbNode);
195  lSkeletonLimbNodeAttribute2->Size.Set(1.0);
196  FbxNode* lSkeletonLimbNode2 = FbxNode::Create(pScene,lLimbNodeName2.Buffer());
197  lSkeletonLimbNode2->SetNodeAttribute(lSkeletonLimbNodeAttribute2);
198  lSkeletonLimbNode2->LclTranslation.Set(FbxVector4(0.0, 40.0, 0.0));
199 
200  // Build skeleton node hierarchy.
201  lSkeletonRoot->AddChild(lSkeletonLimbNode1);
202  lSkeletonLimbNode1->AddChild(lSkeletonLimbNode2);
203 
204  return lSkeletonRoot;

207 // Set the influence of the skeleton segments over the cylinder.
208 // The link mode is FbxCluster::eTotalOne which means the total
209 // of the weights assigned to a given control point must equal 1.
210 void LinkPatchToSkeleton(FbxScene* pScene, FbxNode* pPatch, FbxNode* pSkeletonRoot)
211 {
212  int i, j;
213  FbxAMatrix lXMatrix;
214 
215  FbxNode* lRoot = pSkeletonRoot;
216  FbxNode* lLimbNode1 = pSkeletonRoot->GetChild(0);
217  FbxNode* lLimbNode2 = lLimbNode1->GetChild(0);
218 
219  // Bottom section of cylinder is clustered to skeleton root.
220  FbxCluster *lClusterToRoot = FbxCluster::Create(pScene,"");
221  lClusterToRoot->SetLink(lRoot);
222  lClusterToRoot->SetLinkMode(FbxCluster::eTotalOne);
223  for(i=0; i<4; ++i)
224  for(j=0; j<4; ++j)
225  lClusterToRoot->AddControlPointIndex(4*i + j, 1.0 - 0.25*i);
226 
227  // Center section of cylinder is clustered to skeleton limb node.
228  FbxCluster* lClusterToLimbNode1 = FbxCluster::Create(pScene, "");
229  lClusterToLimbNode1->SetLink(lLimbNode1);
230  lClusterToLimbNode1->SetLinkMode(FbxCluster::eTotalOne);
231 
232  for (i =1; i<6; ++i)
233  for (j=0; j<4; ++j)
234  lClusterToLimbNode1->AddControlPointIndex(4*i + j, (i == 1 || i == 5 ? 0.25 : 0.50));
235 
236 
237  // Top section of cylinder is clustered to skeleton limb.
238 
239  FbxCluster * lClusterToLimbNode2 = FbxCluster::Create(pScene,"");
240  lClusterToLimbNode2->SetLink(lLimbNode2);
241  lClusterToLimbNode2->SetLinkMode(FbxCluster::eTotalOne);
242 
243  for (i=3; i<7; ++i)
244  for (j=0; j<4; ++j)
245  lClusterToLimbNode2->AddControlPointIndex(4*i + j, 0.25*(i - 2));
246 
247  // Now we have the Patch and the skeleton correctly positioned,
248  // set the Transform and TransformLink matrix accordingly.
249  FbxScene* lScene = pPatch->GetScene();
250  if( lScene ) lXMatrix = pPatch->EvaluateGlobalTransform();
251 
252  lClusterToRoot->SetTransformMatrix(lXMatrix);
253  lClusterToLimbNode1->SetTransformMatrix(lXMatrix);
254  lClusterToLimbNode2->SetTransformMatrix(lXMatrix);
255 
256 
257 
258  if( lScene ) lXMatrix = lRoot->EvaluateGlobalTransform();
259  lClusterToRoot->SetTransformLinkMatrix(lXMatrix);
260 
261 
262  if( lScene ) lXMatrix = lLimbNode1->EvaluateGlobalTransform();
263  lClusterToLimbNode1->SetTransformLinkMatrix(lXMatrix);
264 
265 
266  if( lScene ) lXMatrix = lLimbNode2->EvaluateGlobalTransform();
267  lClusterToLimbNode2->SetTransformLinkMatrix(lXMatrix);
268 
269 
270  // Add the clusters to the patch by creating a skin and adding those clusters to that skin.
271  // After add that skin.
272 
273  FbxGeometry* lPatchAttribute = (FbxGeometry*) pPatch->GetNodeAttribute();
274  FbxSkin* lSkin = FbxSkin::Create(pScene, "");
275  lSkin->AddCluster(lClusterToRoot);
276  lSkin->AddCluster(lClusterToLimbNode1);
277  lSkin->AddCluster(lClusterToLimbNode2);
278  lPatchAttribute->AddDeformer(lSkin);
279 
280 }

420 // Store the Bind Pose
421 void StoreBindPose(FbxScene* pScene, FbxNode* pPatch)
422 {
423  // In the bind pose, we must store all the link's global matrix at the time of the bind.
424  // Plus, we must store all the parent(s) global matrix of a link, even if they are not
425  // themselves deforming any model.
426 
427  // In this example, since there is only one model deformed, we don't need walk through
428  // the scene
429  //
430 
431  // Now list the all the link involve in the patch deformation
432  FbxArray<FbxNode*> lClusteredFbxNodes;
433  int i, j;
434 
435  if (pPatch && pPatch->GetNodeAttribute())
436  {
437  int lSkinCount=0;
438  int lClusterCount=0;
439  switch (pPatch->GetNodeAttribute()->GetAttributeType())
440  {
441  default:
442  break;
443  case FbxNodeAttribute::eMesh:
444  case FbxNodeAttribute::eNurbs:
445  case FbxNodeAttribute::ePatch:
446 
447  lSkinCount = ((FbxGeometry*)pPatch->GetNodeAttribute())->GetDeformerCount(FbxDeformer::eSkin);
448  //Go through all the skins and count them
449  //then go through each skin and get their cluster count
450  for(i=0; i<lSkinCount; ++i)
451  {
452  FbxSkin *lSkin=(FbxSkin*)((FbxGeometry*)pPatch->GetNodeAttribute())->GetDeformer(i, FbxDeformer::eSkin);
453  lClusterCount+=lSkin->GetClusterCount();
454  }
455  break;
456  }
457  //if we found some clusters we must add the node
458  if (lClusterCount)
459  {
460  //Again, go through all the skins get each cluster link and add them
461  for (i=0; i<lSkinCount; ++i)
462  {
463  FbxSkin *lSkin=(FbxSkin*)((FbxGeometry*)pPatch->GetNodeAttribute())->GetDeformer(i, FbxDeformer::eSkin);
464  lClusterCount=lSkin->GetClusterCount();
465  for (j=0; j<lClusterCount; ++j)
466  {
467  FbxNode* lClusterNode = lSkin->GetCluster(j)->GetLink();
468  AddNodeRecursively(lClusteredFbxNodes, lClusterNode);
469  }
470 
471  }
472 
473  // Add the patch to the pose
474  lClusteredFbxNodes.Add(pPatch);
475  }
476  }
477 
478  // Now create a bind pose with the link list
479  if (lClusteredFbxNodes.GetCount())
480  {
481  // A pose must be named. Arbitrarily use the name of the patch node.
482  FbxPose* lPose = FbxPose::Create(pScene,pPatch->GetName());
483 
484  // default pose type is rest pose, so we need to set the type as bind pose
485  lPose->SetIsBindPose(true);
486 
487  for (i=0; i<lClusteredFbxNodes.GetCount(); i++)
488  {
489  FbxNode* lKFbxNode = lClusteredFbxNodes.GetAt(i);
490  FbxMatrix lBindMatrix = lKFbxNode->EvaluateGlobalTransform();
491 
492  lPose->Add(lKFbxNode, lBindMatrix);
493  }
494 
495  // Add the pose to the scene
496  pScene->AddPose(lPose);
497  }
498 }

500 // Store a Rest Pose
501 void StoreRestPose(FbxScene* pScene, FbxNode* pSkeletonRoot)
502 {
503  // This example show an arbitrary rest pose assignment.
504  // This rest pose will set the bone rotation to the same value
505  // as time 1 second in the first stack of animation, but the
506  // position of the bone will be set elsewhere in the scene.
507  FbxString lNodeName;
508  FbxNode* lKFbxNode;
509  FbxMatrix lTransformMatrix;
510  FbxVector4 lT,lR,lS(1.0, 1.0, 1.0);
511 
512  // Create the rest pose
513  FbxPose* lPose = FbxPose::Create(pScene,"A Bind Pose");
514 
515  // Set the skeleton root node to the global position (10, 10, 10)
516  // and global rotation of 45deg along the Z axis.
517  lT.Set(10.0, 10.0, 10.0);
518  lR.Set( 0.0, 0.0, 45.0);
519 
520  lTransformMatrix.SetTRS(lT, lR, lS);
521 
522  // Add the skeleton root node to the pose
523  lKFbxNode = pSkeletonRoot;
524  lPose->Add(lKFbxNode, lTransformMatrix, false /*it's a global matrix*/);
525 
526  // Set the lLimbNode1 node to the local position of (0, 40, 0)
527  // and local rotation of -90deg along the Z axis. This show that
528  // you can mix local and global coordinates in a rest pose.
529  lT.Set(0.0, 40.0, 0.0);
530  lR.Set(0.0, 0.0, -90.0);
531 
532  lTransformMatrix.SetTRS(lT, lR, lS);
533 
534  // Add the skeleton second node to the pose
535  lKFbxNode = lKFbxNode->GetChild(0);
536  lPose->Add(lKFbxNode, lTransformMatrix, true /*it's a local matrix*/);
537 
538  // Set the lLimbNode2 node to the local position of (0, 40, 0)
539  // and local rotation of 45deg along the Z axis.
540  lT.Set(0.0, 40.0, 0.0);
541  lR.Set(0.0, 0.0, 45.0);
542 
543  lTransformMatrix.SetTRS(lT, lR, lS);
544 
545  // Add the skeleton second node to the pose
546  lKFbxNode = lKFbxNode->GetChild(0);
547  lNodeName = lKFbxNode->GetName();
548  lPose->Add(lKFbxNode, lTransformMatrix, true /*it's a local matrix*/);
549 
550  // Now add the pose to the scene
551  pScene->AddPose(lPose);
552 }
