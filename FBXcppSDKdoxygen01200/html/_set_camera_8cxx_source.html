44 // Set the view to the current camera settings.
45 void SetCamera(FbxScene* pScene,
46  FbxTime& pTime,
47  FbxAnimLayer* pAnimLayer,
48  const FbxArray<FbxNode*>& pCameraArray,
49  int pWindowWidth, int pWindowHeight)
50 {
51  // Find the current camera at the given time.
52  FbxCamera* lCamera = GetCurrentCamera(pScene, pTime, pAnimLayer, pCameraArray);
53  if( lCamera == NULL)
54  return;
55  FbxNode* lCameraNode = lCamera ? lCamera->GetNode() : NULL;
56 
57  // Compute the camera position and direction.
58  FbxVector4 lEye(0,0,1);
59  FbxVector4 lCenter(0,0,0);
60  FbxVector4 lUp(0,1,0);
61  FbxVector4 lForward, lRight;
62 
63  if (lCamera)
64  {
65  lEye = lCamera->Position.Get();
66  lUp = lCamera->UpVector.Get();
67  }
68 
69  if (lCameraNode && lCameraNode->GetTarget())
70  {
71  lCenter = GetGlobalPosition(lCameraNode->GetTarget(), pTime).GetT();
72  }
73  else
74  {
75  if (!lCameraNode || IsProducerCamera(pScene, lCamera))
76  {
77  if (lCamera)
78  lCenter = lCamera->InterestPosition.Get();
79  }
80  else
81  {
82  // Get the direction
83  FbxAMatrix lGlobalRotation;
84  FbxVector4 lRotationVector(GetGlobalPosition(lCameraNode, pTime).GetR());
85  lGlobalRotation.SetR(lRotationVector);
86 
87  // Get the length
88  FbxVector4 lInterestPosition(lCamera->InterestPosition.Get());
89  FbxVector4 lCameraGlobalPosition(GetGlobalPosition(lCameraNode, pTime).GetT());
90  double lLength = (FbxVector4(lInterestPosition - lCameraGlobalPosition).Length());
91 
92  // Set the center.
93  // A camera with rotation = {0,0,0} points to the X direction. So create a
94  // vector in the X direction, rotate that vector by the global rotation amount
95  // and then position the center by scaling and translating the resulting vector
96  lRotationVector = FbxVector4(1.0,0,0);
97  lCenter = lGlobalRotation.MultT(lRotationVector);
98  lCenter *= lLength;
99  lCenter += lEye;
100 
101  // Update the default up vector with the camera rotation.
102  lRotationVector = FbxVector4(0,1.0,0);
103  lUp = lGlobalRotation.MultT(lRotationVector);
104  }
105  }
106 
107  // Align the up vector.
108  lForward = lCenter - lEye;
109  lForward.Normalize();
110  lRight = lForward.CrossProduct(lUp);
111  lRight.Normalize();
112  lUp = lRight.CrossProduct(lForward);
113  lUp.Normalize();
114 
115  // Rotate the up vector with the roll value.
116  double lRadians = 0;
117 
118  if (lCamera)
119  lRadians = lCamera->Roll.Get() * FBXSDK_PI_DIV_180;
120  lUp = lUp * cos( lRadians) + lRight * sin(lRadians);
121 
122 
123  double lNearPlane = 0.01;
124  if (lCamera)
125  lNearPlane = lCamera->GetNearPlane();
126  double lFarPlane = 4000.0;
127  if (lCamera)
128  lFarPlane = lCamera->GetFarPlane();
129 
130  //Get global scaling.
131  FbxVector4 lCameraScaling = GetGlobalPosition(lCameraNode, pTime).GetS();
132  static const int FORWARD_SCALE = 2;
133 
134  //scaling near plane and far plane
135  lNearPlane *= lCameraScaling[ FORWARD_SCALE];
136  lFarPlane *= lCameraScaling[ FORWARD_SCALE];
137 
138 
139 
140 
141 
142 
143  // Get the relevant camera settings for a perspective view.
144  if (lCamera && lCamera->ProjectionType.Get() == FbxCamera::ePerspective)
145  {
146  //get the aspect ratio
147  FbxCamera::EAspectRatioMode lCamAspectRatioMode = lCamera->GetAspectRatioMode();
148  double lAspectX = lCamera->AspectWidth.Get();
149  double lAspectY = lCamera->AspectHeight.Get();
150  double lAspectRatio = 1.333333;
151  switch( lCamAspectRatioMode)
152  {
153  case FbxCamera::eWindowSize:
154  lAspectRatio = lAspectX / lAspectY;
155  break;
156  case FbxCamera::eFixedRatio:
157  lAspectRatio = lAspectX;
158 
159  break;
160  case FbxCamera::eFixedResolution:
161  lAspectRatio = lAspectX / lAspectY * lCamera->GetPixelRatio();
162  break;
163  case FbxCamera::eFixedWidth:
164  lAspectRatio = lCamera->GetPixelRatio() / lAspectY;
165  break;
166  case FbxCamera::eFixedHeight:
167  lAspectRatio = lCamera->GetPixelRatio() * lAspectX;
168  break;
169  default:
170  break;
171 
172  }
173 
174  //get the aperture ratio
175  double lFilmHeight = lCamera->GetApertureHeight();
176  double lFilmWidth = lCamera->GetApertureWidth() * lCamera->GetSqueezeRatio();
177  //here we use Height : Width
178  double lApertureRatio = lFilmHeight / lFilmWidth;
179 
180 
181  //change the aspect ratio to Height : Width
182  lAspectRatio = 1 / lAspectRatio;
183  //revise the aspect ratio and aperture ratio
184  FbxCamera::EGateFit lCameraGateFit = lCamera->GateFit.Get();
185  switch( lCameraGateFit )
186  {
187 
188  case FbxCamera::eFitFill:
189  if( lApertureRatio > lAspectRatio) // the same as eHORIZONTAL_FIT
190  {
191  lFilmHeight = lFilmWidth * lAspectRatio;
192  lCamera->SetApertureHeight( lFilmHeight);
193  lApertureRatio = lFilmHeight / lFilmWidth;
194  }
195  else if( lApertureRatio < lAspectRatio) //the same as eVERTICAL_FIT
196  {
197  lFilmWidth = lFilmHeight / lAspectRatio;
198  lCamera->SetApertureWidth( lFilmWidth);
199  lApertureRatio = lFilmHeight / lFilmWidth;
200  }
201  break;
202  case FbxCamera::eFitVertical:
203  lFilmWidth = lFilmHeight / lAspectRatio;
204  lCamera->SetApertureWidth( lFilmWidth);
205  lApertureRatio = lFilmHeight / lFilmWidth;
206  break;
207  case FbxCamera::eFitHorizontal:
208  lFilmHeight = lFilmWidth * lAspectRatio;
209  lCamera->SetApertureHeight( lFilmHeight);
210  lApertureRatio = lFilmHeight / lFilmWidth;
211  break;
212  case FbxCamera::eFitStretch:
213  lAspectRatio = lApertureRatio;
214  break;
215  case FbxCamera::eFitOverscan:
216  if( lFilmWidth > lFilmHeight)
217  {
218  lFilmHeight = lFilmWidth * lAspectRatio;
219  }
220  else
221  {
222  lFilmWidth = lFilmHeight / lAspectRatio;
223  }
224  lApertureRatio = lFilmHeight / lFilmWidth;
225  break;
226  case FbxCamera::eFitNone:
227  default:
228  break;
229  }
230  //change the aspect ratio to Width : Height
231  lAspectRatio = 1 / lAspectRatio;
232 
233  double lFieldOfViewX = 0.0;
234  double lFieldOfViewY = 0.0;
235  if ( lCamera->GetApertureMode() == FbxCamera::eVertical)
236  {
237  lFieldOfViewY = lCamera->FieldOfView.Get();
238  lFieldOfViewX = VFOV2HFOV( lFieldOfViewY, 1 / lApertureRatio);
239  }
240  else if (lCamera->GetApertureMode() == FbxCamera::eHorizontal)
241  {
242  lFieldOfViewX = lCamera->FieldOfView.Get(); //get HFOV
243  lFieldOfViewY = HFOV2VFOV( lFieldOfViewX, lApertureRatio);
244  }
245  else if (lCamera->GetApertureMode() == FbxCamera::eFocalLength)
246  {
247  lFieldOfViewX = lCamera->ComputeFieldOfView(lCamera->FocalLength.Get()); //get HFOV
248  lFieldOfViewY = HFOV2VFOV( lFieldOfViewX, lApertureRatio);
249  }
250  else if (lCamera->GetApertureMode() == FbxCamera::eHorizAndVert) {
251  lFieldOfViewX = lCamera->FieldOfViewX.Get();
252  lFieldOfViewY = lCamera->FieldOfViewY.Get();
253  }
254 
255 
256 
257  double lRealScreenRatio = (double)pWindowWidth / (double)pWindowHeight;
258  int lViewPortPosX = 0,
259  lViewPortPosY = 0,
260  lViewPortSizeX = pWindowWidth,
261  lViewPortSizeY = pWindowHeight;
262  //compute the view port
263  if( lRealScreenRatio > lAspectRatio)
264  {
265  lViewPortSizeY = pWindowHeight;
266  lViewPortSizeX = (int)( lViewPortSizeY * lAspectRatio);
267  lViewPortPosY = 0;
268  lViewPortPosX = (int)((pWindowWidth - lViewPortSizeX) * 0.5);
269  }
270  else
271  {
272  lViewPortSizeX = pWindowWidth;
273  lViewPortSizeY = (int)(lViewPortSizeX / lAspectRatio);
274  lViewPortPosX = 0;
275  lViewPortPosY = (int)((pWindowHeight - lViewPortSizeY) * 0.5);
276  }
277 
278  //revise the Perspective since we have film offset
279  double lFilmOffsetX = lCamera->FilmOffsetX.Get();
280  double lFilmOffsetY = lCamera->FilmOffsetY.Get();
281  lFilmOffsetX = 0 - lFilmOffsetX / lFilmWidth * 2.0;
282  lFilmOffsetY = 0 - lFilmOffsetY / lFilmHeight * 2.0;
283 
284  GlSetCameraPerspective( lFieldOfViewY, lAspectRatio, lNearPlane, lFarPlane, lEye, lCenter, lUp, lFilmOffsetX, lFilmOffsetY);
285 
286 
287 
288  //glMatrixMode(GL_PROJECTION);
289  //double lTestPerpMatrix[ 16];
290  //glGetDoublev( GL_PROJECTION_MATRIX, lTestPerpMatrix);
291 
292  //lTestPerpMatrix[ 8] -= lFilmOffsetX;
293  //lTestPerpMatrix[ 9] -= lFilmOffsetY;
294  //
295  //glLoadMatrixd( lTestPerpMatrix);
296  //glMatrixMode(GL_MODELVIEW);
297 
298 
299  glViewport( lViewPortPosX, lViewPortPosY, lViewPortSizeX, lViewPortSizeY);
300 
301 
302  }
303  // Get the relevant camera settings for an orthogonal view.
304  else
305  {
306  double lPixelRatio = 1.0;
307  if (lCamera)
308  lPixelRatio = lCamera->GetPixelRatio();
309 
310  double lLeftPlane, lRightPlane, lBottomPlane, lTopPlane;
311 
312  if(pWindowWidth < pWindowHeight)
313  {
314  lLeftPlane = -gsOrthoCameraScale * lPixelRatio;
315  lRightPlane = gsOrthoCameraScale * lPixelRatio;
316  lBottomPlane = -gsOrthoCameraScale * pWindowHeight / pWindowWidth;
317  lTopPlane = gsOrthoCameraScale * pWindowHeight / pWindowWidth;
318  }
319  else
320  {
321  pWindowWidth *= (int) lPixelRatio;
322  lLeftPlane = -gsOrthoCameraScale * pWindowWidth / pWindowHeight;
323  lRightPlane = gsOrthoCameraScale * pWindowWidth / pWindowHeight;
324  lBottomPlane = -gsOrthoCameraScale;
325  lTopPlane = gsOrthoCameraScale;
326  }
327 
328  GlSetCameraOrthogonal(lLeftPlane,
329  lRightPlane,
330  lBottomPlane,
331  lTopPlane,
332  lNearPlane,
333  lFarPlane,
334  lEye,
335  lCenter,
336  lUp);
337  }
338 }

342 FbxCamera* GetCurrentCamera(FbxScene* pScene,
343  FbxTime& pTime,
344  FbxAnimLayer* pAnimLayer,
345  const FbxArray<FbxNode*>& pCameraArray)
346 {
347  FbxGlobalSettings& lGlobalSettings = pScene->GetGlobalSettings();
348  FbxGlobalCameraSettings& lGlobalCameraSettings = pScene->GlobalCameraSettings();
349  FbxString lCurrentCameraName = lGlobalSettings.GetDefaultCamera();
350 
351  // check if we need to create the Producer cameras!
352  if (lGlobalCameraSettings.GetCameraProducerPerspective() == NULL &&
353  lGlobalCameraSettings.GetCameraProducerBottom() == NULL &&
354  lGlobalCameraSettings.GetCameraProducerTop() == NULL &&
355  lGlobalCameraSettings.GetCameraProducerFront() == NULL &&
356  lGlobalCameraSettings.GetCameraProducerBack() == NULL &&
357  lGlobalCameraSettings.GetCameraProducerRight() == NULL &&
358  lGlobalCameraSettings.GetCameraProducerLeft() == NULL)
359  {
360  lGlobalCameraSettings.CreateProducerCameras();
361  }
362 
363  if (lCurrentCameraName.Compare(FBXSDK_CAMERA_PERSPECTIVE) == 0)
364  {
365  return lGlobalCameraSettings.GetCameraProducerPerspective();
366  }
367  else if (lCurrentCameraName.Compare(FBXSDK_CAMERA_BOTTOM) == 0)
368  {
369  return lGlobalCameraSettings.GetCameraProducerBottom();
370  }
371  else if (lCurrentCameraName.Compare(FBXSDK_CAMERA_TOP) == 0)
372  {
373  return lGlobalCameraSettings.GetCameraProducerTop();
374  }
375  else if (lCurrentCameraName.Compare(FBXSDK_CAMERA_FRONT) == 0)
376  {
377  return lGlobalCameraSettings.GetCameraProducerFront();
378  }
379  else if (lCurrentCameraName.Compare(FBXSDK_CAMERA_BACK) == 0)
380  {
381  return lGlobalCameraSettings.GetCameraProducerBack();
382  }
383  else if (lCurrentCameraName.Compare(FBXSDK_CAMERA_RIGHT) == 0)
384  {
385  return lGlobalCameraSettings.GetCameraProducerRight();
386  }
387  else if (lCurrentCameraName.Compare(FBXSDK_CAMERA_LEFT) == 0)
388  {
389  return lGlobalCameraSettings.GetCameraProducerLeft();
390  }
391  else if (lCurrentCameraName.Compare(FBXSDK_CAMERA_SWITCHER) == 0)
392  {
393  FbxCameraSwitcher* lCameraSwitcher = pScene->GlobalCameraSettings().GetCameraSwitcher();
394  FbxAnimCurve* lCurve = NULL;
395  if (lCameraSwitcher)
396  {
397  lCurve = lCameraSwitcher->CameraIndex.GetCurve(pAnimLayer);
398  int lCameraIndex = lCurve ? int(lCurve->Evaluate(pTime)) - 1 : 0;
399  if (lCameraIndex >= 0 && lCameraIndex < pCameraArray.GetCount())
400  {
401  FbxNode* lNode = pCameraArray[lCameraIndex];
402 
403  // Get the animated parameters of the camera.
404  GetCameraAnimatedParameters(lNode, pTime, pAnimLayer);
405 
406  return (FbxCamera*) lNode->GetNodeAttribute();
407  }
408  }
409  }
410  else
411  {
412  int i;
413  FbxNode* lNode = NULL;
414 
415  // Find the camera in the camera array.
416  for (i = 0; i < pCameraArray.GetCount(); i++)
417  {
418  if (lCurrentCameraName.Compare(pCameraArray[i]->GetName()) == 0)
419  {
420  lNode = pCameraArray[i];
421  break;
422  }
423  }
424 
425  if (lNode)
426  {
427  // Get the animated parameters of the camera.
428  GetCameraAnimatedParameters(lNode, pTime, pAnimLayer);
429 
430  return (FbxCamera*) lNode->GetNodeAttribute();
431  }
432  }
433 
434  return lGlobalCameraSettings.GetCameraProducerPerspective();
435 }

437 // Get the animated parameters of a camera contained in the scene
438 // and store them in the associated member variables contained in
439 // the camera.
440 void GetCameraAnimatedParameters(FbxNode* pNode, FbxTime& pTime, FbxAnimLayer* pAnimLayer)
441 {
442  FbxCamera* lCamera = (FbxCamera*) pNode->GetNodeAttribute();
443  lCamera->Position.Set(GetGlobalPosition(pNode, pTime).GetT());
444 
445  FbxAnimCurve* fc = lCamera->Roll.GetCurve(pAnimLayer);
446  if (fc)
447  lCamera->Roll.Set(fc->Evaluate(pTime));
448 
449  FbxCamera::EApertureMode lCameraApertureMode = lCamera->GetApertureMode();
450  if (lCameraApertureMode == FbxCamera::eHorizontal ||
451  lCameraApertureMode == FbxCamera::eVertical)
452  {
453  double lFieldOfView = lCamera->FieldOfView.Get();
454  fc = lCamera->FieldOfView.GetCurve(pAnimLayer);
455  if (fc)
456  lFieldOfView = fc->Evaluate(pTime);
457 
458  //update FOV and focal length
459  lCamera->FieldOfView.Set( lFieldOfView);
460  lCamera->FocalLength.Set( lCamera->ComputeFocalLength( lFieldOfView));
461 
462  }
463  else if ( lCameraApertureMode == FbxCamera::eHorizAndVert)
464  {
465  double lOldFieldOfViewX = lCamera->FieldOfViewX.Get();
466  double lOldFieldOfViewY = lCamera->FieldOfViewY.Get();
467 
468  //update FOV
469  double lNewFieldOfViewX = lOldFieldOfViewX;
470  double lNewFieldOfViewY = lOldFieldOfViewY;
471  fc = lCamera->FieldOfViewX.GetCurve(pAnimLayer);
472  if (fc)
473  lNewFieldOfViewX = fc->Evaluate(pTime);
474 
475  fc = lCamera->FieldOfViewY.GetCurve(pAnimLayer);
476  if (fc)
477  lNewFieldOfViewY = fc->Evaluate(pTime);
478 
479  lCamera->FieldOfViewX.Set(lNewFieldOfViewX);
480  lCamera->FieldOfViewY.Set(lNewFieldOfViewY);
481 
482  //update aspect
483  double lUpdatedApertureX = lCamera->GetApertureWidth();
484  double lUpdatedApertureY = lCamera->GetApertureHeight();
485  lUpdatedApertureX *= tan( lNewFieldOfViewX * 0.5 * FBXSDK_PI_DIV_180) / tan( lOldFieldOfViewX * 0.5 * FBXSDK_PI_DIV_180);
486  lUpdatedApertureY *= tan( lNewFieldOfViewY * 0.5 * FBXSDK_PI_DIV_180) / tan( lOldFieldOfViewY * 0.5 * FBXSDK_PI_DIV_180);
487 
488  lCamera->FilmWidth.Set( lUpdatedApertureX);
489  lCamera->FilmHeight.Set( lUpdatedApertureY);
490  lCamera->FilmAspectRatio.Set( lUpdatedApertureX / lUpdatedApertureY);
491 
492 
493  }
494  else if ( lCameraApertureMode == FbxCamera::eFocalLength)
495  {
496  double lFocalLength = lCamera->FocalLength.Get();
497  fc = lCamera->FocalLength.GetCurve(pAnimLayer);
498  if (fc && fc ->Evaluate(pTime))
499  lFocalLength = fc->Evaluate( pTime);
500 
501 
502  //update FOV and focal length
503  lCamera->FocalLength.Set( lFocalLength);
504  lCamera->FieldOfView.Set( lCamera->ComputeFieldOfView( lFocalLength));
505  }
506 }

593 void UpdatePerspCameraAttributes( FbxCamera* pCamera, double pNewApertureW, double pNewApertureH)
594 {
595 
596  if( pCamera == NULL || FbxAnimUtilities::IsAnimated( pCamera))
597  return;
598  // update focal length or field of view.
599  double lApertureWidthOrig = pCamera->GetApertureWidth();
600  double lApertureHeightOrig = pCamera->GetApertureHeight();
601 
602  if (pCamera->GetApertureMode() == FbxCamera::eFocalLength) {
603  // update focal length according to hypothetic new apertures.
604  double lFLOrig = pCamera->FocalLength.Get();
605  double lFOVOrig = pCamera->ComputeFieldOfView(lFLOrig); // recompute to be sure.
606  // transform aperture width and height temporarily
607  pCamera->SetApertureWidth( pNewApertureW );
608  pCamera->SetApertureHeight(pNewApertureH );
609  double lFLNew = pCamera->ComputeFocalLength(lFOVOrig);
610  double lFOVNew = pCamera->ComputeFieldOfView(lFLNew);
611  pCamera->FocalLength.Set(lFLNew);
612  pCamera->FieldOfView.Set(lFOVNew);
613  } else if (pCamera->GetApertureMode() == FbxCamera::eVertical
614  || pCamera->GetApertureMode() == FbxCamera::eHorizontal) {
615  double lFOVOrig = pCamera->FieldOfView.Get();
616  double lFLOrig = pCamera->ComputeFocalLength(lFOVOrig); // recompute to be sure.
617  // transform aperture width and height temporarily
618  pCamera->SetApertureWidth(pNewApertureW );
619  pCamera->SetApertureHeight(pNewApertureH );
620  double lFOVNew = pCamera->ComputeFieldOfView(lFLOrig);
621  double lFLNew = pCamera->ComputeFocalLength(lFOVNew);
622  pCamera->FieldOfView.Set(lFOVNew);
623  pCamera->FocalLength.Set(lFLNew);

643 void CameraZoom(FbxScene* pScene, int pZoomDepth, int pZoomMode)
644 {
645  FbxCamera* lCamera = GetCurrentCamera(pScene);
646  if( lCamera == NULL)
647  return;
648  if( pZoomMode == SceneContext::ZOOM_FOCAL_LENGTH)
649  {
650  if (lCamera->ProjectionType.Get() == FbxCamera::ePerspective)
651  {
652  double lTransform = 0 - pZoomDepth / 400.0;
653 
654  double lApertureW = lCamera->GetApertureWidth();
655  lApertureW = TransformAperture( lApertureW, lTransform);
656 
657  double lApertureH = lCamera->GetApertureHeight();
658  lApertureH = TransformAperture( lApertureH, lTransform);
659 
660  UpdatePerspCameraAttributes( lCamera, lApertureW, lApertureH);
661 
662 
663  }
664  else
665  {
666  if( pZoomDepth > 0)
667  gsOrthoCameraScale *= 0.8;
668  else
669  gsOrthoCameraScale *= 1.25;
670  }
671  }
672  else
673  {
674  FbxNode* lCameraNode = lCamera ? lCamera->GetNode() : NULL;
675 
676  // Compute the camera position and direction.
677  FbxVector4 lEye(0,0,1);
678  FbxVector4 lCenter(0,0,0);
679  FbxVector4 lForward(0,0,0);
680 
681  if (lCamera)
682  {
683  lEye = lCamera->Position.Get();
684  }
685 
686  if (lCameraNode && lCameraNode->GetTarget())
687  {
688  lCenter = lCameraNode->GetTarget()->LclTranslation.Get();
689  lForward = lCenter - lEye;
690  }
691  else
692  {
693  if (!lCameraNode || IsProducerCamera(pScene, lCamera))
694  {
695  if (lCamera)
696  {
697  lCenter = lCamera->InterestPosition.Get();
698  lForward = lCenter - lEye;
699  }
700  }
701  else
702  {
703  // Get the direction
704  FbxAMatrix lGlobalRotation;
705  FbxVector4 lRotationVector( lCameraNode->LclRotation.Get());
706  lGlobalRotation.SetR(lRotationVector);
707 
708  // Set the center.
709  // A camera with rotation = {0,0,0} points to the X direction. So create a
710  // vector in the X direction, rotate that vector by the global rotation amount
711  // and then position the center by scaling and translating the resulting vector
712  lRotationVector = FbxVector4(1.0,0,0);
713  lForward = lGlobalRotation.MultT(lRotationVector);
714  }
715  }
716  lForward.Normalize();
717  lEye += lForward * pZoomDepth;
718  FbxDouble3 lPosition(lEye[0], lEye[1], lEye[2]);
719  lCamera->Position.Set(lPosition);
720 
721  }

724 void CameraOrbit(FbxScene* pScene, FbxVector4 lOrigCamPos, double OrigRoll, int dX, int dY)
725 {
726  // Orbit the camera horizontally dX degrees, vertically dY degrees.
727  FbxCamera* lCamera = GetCurrentCamera(pScene);
728  if (!lCamera) return;
729  FbxGlobalCameraSettings& lGlobalCameraSettings = pScene->GlobalCameraSettings();
730  if (lCamera != lGlobalCameraSettings.GetCameraProducerPerspective()) return;
731  if (lCamera->LockMode.Get()) return;
732  if (dX == 0 && dY == 0) return;
733 
734  FbxVector4 lRotationVector, lNewPosition, lCurPosition;
735  FbxAMatrix lRotation;
736  FbxVector4 lCenter = lCamera->InterestPosition.Get();
737 
738  // current position
739  FbxVector4 lPosition = lCamera->Position.Get();
740  lCurPosition = lPosition-lCenter;
741 
742  // translate
743  lNewPosition = lOrigCamPos-lCenter;
744 
745  int rotX;
746  if (lNewPosition[2] == 0) {
747  rotX = 90;
748  } else {
749  rotX = (int) (atan((double)lNewPosition[0]/(double)lNewPosition[2]) * FBXSDK_180_DIV_PI);
750  }
751  bool bRoll = (((int)OrigRoll % 360) != 0);
752  if ( (lNewPosition[2] < 0 && !bRoll)
753  || (lNewPosition[2] > 0 && bRoll) ) {
754  dY = -dY;
755  }
756  if (bRoll) dX = -dX;
757 
758  // Center on the X axis (push)
759  lRotationVector[1] = -rotX;
760  lRotation.SetR(lRotationVector);
761  lNewPosition = lRotation.MultT(lNewPosition);
762  // Rotation for the vertical movement: around the X axis
763  lRotationVector[1] = 0;
764  lRotationVector[0] = dY;
765  lRotation.SetR(lRotationVector);
766  lNewPosition = lRotation.MultT(lNewPosition);
767  // Back from the X axis (pop)
768  lRotationVector[0] = 0;
769  lRotationVector[1] = rotX;
770  lRotation.SetR(lRotationVector);
771  lNewPosition = lRotation.MultT(lNewPosition);
772  // Rotation for the horizontal movement
773  lRotationVector[1] = -dX;
774  lRotation.SetR(lRotationVector);
775  lNewPosition = lRotation.MultT(lNewPosition);
776 
777  // Detect camera flip
778  if ( lNewPosition[0]*lCurPosition[0] < 0
779  && lNewPosition[2]*lCurPosition[2] < 0) {
780  // flip -> roll 180.
781  double lRoll = lCamera->Roll.Get();
782  lRoll = 180.0-lRoll;
783  lCamera->Roll.Set(lRoll);
784  }
785 
786  // Back from center
787  lNewPosition = lNewPosition + lCenter;
788 
789  lCamera->Position.Set(lNewPosition);
790 
791 }

793 void CameraPan(FbxScene* pScene, FbxVector4 lOrigCamPos, FbxVector4 lOrigCamCenter, double OrigRoll, int dX, int dY) {
794  // Pan the camera horizontally dX degrees, vertically dY degrees.
795  FbxCamera* lCamera = GetCurrentCamera(pScene);
796  if (!lCamera) return;
797  if (!IsProducerCamera(pScene, lCamera)) return;
798  if (lCamera->LockMode.Get()) return;
799  if (dX == 0 && dY == 0) return;
800 
801  FbxGlobalCameraSettings& lGlobalCameraSettings = pScene->GlobalCameraSettings();
802 
803  FbxVector4 lRotationXV, lRotationYV, lTranslationV;
804  FbxAMatrix lRotationX, lRotationY, lRotationXInverse, lRotationYInverse, lTranslation;
805  FbxVector4 lNewPosition = lOrigCamPos;
806  FbxVector4 lNewCenter = lOrigCamCenter;
807 
808  // Translate the camera in dX and dY according to its point of view.
809  if (lCamera == lGlobalCameraSettings.GetCameraProducerTop()) {
810  lTranslationV[0] = -dX;
811  lTranslationV[1] = 0;
812  lTranslationV[2] = dY;
813  } else if (lCamera == lGlobalCameraSettings.GetCameraProducerBottom()) {
814  lTranslationV[0] = dX;
815  lTranslationV[1] = 0;
816  lTranslationV[2] = dY;
817  } else if (lCamera == lGlobalCameraSettings.GetCameraProducerFront()) {
818  lTranslationV[0] = -dX;
819  lTranslationV[1] = -dY;
820  lTranslationV[2] = 0;
821  } else if (lCamera == lGlobalCameraSettings.GetCameraProducerBack()) {
822  lTranslationV[0] = dX;
823  lTranslationV[1] = -dY;
824  lTranslationV[2] = 0;
825  } else if (lCamera == lGlobalCameraSettings.GetCameraProducerRight()) {
826  lTranslationV[0] = 0;
827  lTranslationV[1] = -dY;
828  lTranslationV[2] = dX;
829  } else if (lCamera == lGlobalCameraSettings.GetCameraProducerLeft()) {
830  lTranslationV[0] = 0;
831  lTranslationV[1] = -dY;
832  lTranslationV[2] = -dX;
833  } else {
834  // Perspective view. More computation.
835 
836  // Adjust displacement when there is roll
837  bool bRoll = (((int)OrigRoll % 360) != 0);
838  if (bRoll) {
839  dX = -dX;
840  dY = -dY;
841  }
842 
843  // Compute angles aY and aZ of the camera with Y and Z axis.
844  double aZ, aY;
846  FbxVector4 lDist = lNewPosition - lNewCenter;
847  // Euclidian distance between camera and lookat
848  double dist = (double)(lDist[0]*lDist[0]+lDist[1]*lDist[1]+lDist[2]*lDist[2]);
849  // aZ
850  if (lDist[2] == 0) {
851  aZ = 90.0;
852  } else {
853  aZ = (atan((double)lDist[0]/(double)lDist[2]) * FBXSDK_180_DIV_PI);
854  }
855  if (lNewPosition[2] < lNewCenter[2]) aZ += 180;
856  // aY
857  if (dist > 0.001) {
858  aY = (asin(sqrt((double)(lDist[1]*lDist[1])/ dist)) * FBXSDK_180_DIV_PI);
859  } else {
860  aY = 0;
861  }
862  if (lNewPosition[1] < lNewCenter[1]) aY = -aY;
863 
864 
865  // Basis translation
866  lTranslationV[0] = -dX;
867  lTranslationV[1] = -dY;
868  lTranslationV[2] = 0;
869 
870  // Rotation around Y axis
871  lRotationYV[0] = 0;
872  lRotationYV[1] = -aZ;
873  lRotationYV[2] = 0;
874  lRotationY.SetR(lRotationYV);
875  // Rotation around X axis
876  lRotationXV[0] = aY;
877  lRotationXV[1] = 0;
878  lRotationXV[2] = 0;
879  lRotationX.SetR(lRotationXV);
880 
881  // Modify translation according to aY and aZ.
882  lTranslation.SetT(lTranslationV);
883  lRotationYInverse = lRotationY.Inverse();
884  lRotationXInverse = lRotationX.Inverse();
885  lTranslation = lRotationYInverse * lRotationXInverse * lTranslation * lRotationY * lRotationX;
886  lTranslationV = lTranslation.GetT();
887  }
888 
889  // Translate camera and center according to pan.
890  lNewPosition += lTranslationV;
891  lNewCenter += lTranslationV;
892 
893  lCamera->Position.Set(lNewPosition);
894  lCamera->InterestPosition.Set(lNewCenter);
895 }
