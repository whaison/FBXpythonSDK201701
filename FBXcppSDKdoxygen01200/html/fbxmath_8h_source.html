294 template<class T> inline T FbxSqrt(const T x)
295 {
296  if( x > 1 )
297  {
298  T z, y = x >> 1;
299  do
300  {
301  z = y;
302  y = (y + (x / y)) >> 1;
303  }
304  while(y < z);
305 
306  return z;
307  }
308  else
309  {
310  return x;
311  }
312 }
