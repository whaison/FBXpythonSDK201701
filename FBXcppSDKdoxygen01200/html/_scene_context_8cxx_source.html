74  void PreparePointCacheData(FbxScene* pScene, FbxTime &pCache_Start, FbxTime &pCache_Stop)
75  {
76  // This function show how to cycle through scene elements in a linear way.
77  const int lNodeCount = pScene->GetSrcObjectCount<FbxNode>();
78  FbxStatus lStatus;
79 
80  for (int lIndex=0; lIndex<lNodeCount; lIndex++)
81  {
82  FbxNode* lNode = pScene->GetSrcObject<FbxNode>(lIndex);
83 
84  if (lNode->GetGeometry())
85  {
86  int i, lVertexCacheDeformerCount = lNode->GetGeometry()->GetDeformerCount(FbxDeformer::eVertexCache);
87 
88  // There should be a maximum of 1 Vertex Cache Deformer for the moment
89  lVertexCacheDeformerCount = lVertexCacheDeformerCount > 0 ? 1 : 0;
90 
91  for (i=0; i<lVertexCacheDeformerCount; ++i )
92  {
93  // Get the Point Cache object
94  FbxVertexCacheDeformer* lDeformer = static_cast<FbxVertexCacheDeformer*>(lNode->GetGeometry()->GetDeformer(i, FbxDeformer::eVertexCache));
95  if( !lDeformer ) continue;
96  FbxCache* lCache = lDeformer->GetCache();
97  if( !lCache ) continue;
98 
99  // Process the point cache data only if the constraint is active
100  if (lDeformer->Active.Get())
101  {
102  if (lCache->GetCacheFileFormat() == FbxCache::eMaxPointCacheV2)
103  {
104  // This code show how to convert from PC2 to MC point cache format
105  // turn it on if you need it.
106 #if 0
107  if (!lCache->ConvertFromPC2ToMC(FbxCache::eMCOneFile,
108  FbxTime::GetFrameRate(pScene->GetGlobalTimeSettings().GetTimeMode())))
109  {
110  // Conversion failed, retrieve the error here
111  FbxString lTheErrorIs = lCache->GetStaus().GetErrorString();
112  }
113 #endif
114  }
115  else if (lCache->GetCacheFileFormat() == FbxCache::eMayaCache)
116  {
117  // This code show how to convert from MC to PC2 point cache format
118  // turn it on if you need it.
119  //#if 0
120  if (!lCache->ConvertFromMCToPC2(FbxTime::GetFrameRate(pScene->GetGlobalSettings().GetTimeMode()), 0, &lStatus))
121  {
122  // Conversion failed, retrieve the error here
123  FbxString lTheErrorIs = lStatus.GetErrorString();
124  }
125  //#endif
126  }
127 
128 
129  // Now open the cache file to read from it
130  if (!lCache->OpenFileForRead(&lStatus))
131  {
132  // Cannot open file
133  FbxString lTheErrorIs = lStatus.GetErrorString();
134 
135  // Set the deformer inactive so we don't play it back
136  lDeformer->Active = false;
137  }
138  else
139  {
140  // get the start and stop time of the cache
141  FbxTime lChannel_Start;
142  FbxTime lChannel_Stop;
143  int lChannelIndex = lCache->GetChannelIndex(lDeformer->Channel.Get());
144  if(lCache->GetAnimationRange(lChannelIndex, lChannel_Start, lChannel_Stop))
145  {
146  // get the smallest start time
147  if(lChannel_Start < pCache_Start) pCache_Start = lChannel_Start;
148 
149  // get the biggest stop time
150  if(lChannel_Stop > pCache_Stop) pCache_Stop = lChannel_Stop;
151  }
152  }
153  }
154  }
155  }
156  }
157  }

159  // Load a texture file (TGA only now) into GPU and return the texture object name
160  bool LoadTextureFromFile(const FbxString & pFilePath, unsigned int & pTextureObject)
161  {
162  if (pFilePath.Right(3).Upper() == "TGA")
163  {
164  tga_image lTGAImage;
165 
166  if (tga_read(&lTGAImage, pFilePath.Buffer()) == TGA_NOERR)
167  {
168  // Make sure the image is left to right
169  if (tga_is_right_to_left(&lTGAImage))
170  tga_flip_horiz(&lTGAImage);
171 
172  // Make sure the image is bottom to top
173  if (tga_is_top_to_bottom(&lTGAImage))
174  tga_flip_vert(&lTGAImage);
175 
176  // Make the image BGR 24
177  tga_convert_depth(&lTGAImage, 24);
178 
179  // Transfer the texture date into GPU
180  glGenTextures(1, &pTextureObject);
181  glBindTexture(GL_TEXTURE_2D, pTextureObject);
182  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
183  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
184  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_CLAMP);
185  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_CLAMP);
186  glTexEnvi(GL_TEXTURE_ENV, GL_TEXTURE_ENV_MODE, GL_REPLACE);
187  glTexImage2D(GL_TEXTURE_2D, 0, 3, lTGAImage.width, lTGAImage.height, 0, GL_BGR,
188  GL_UNSIGNED_BYTE, lTGAImage.image_data);
189  glBindTexture(GL_TEXTURE_2D, 0);
190 
191  tga_free_buffers(&lTGAImage);
192 
193  return true;
194  }
195  }
196 
197  return false;
198  }

200  // Bake node attributes and materials under this node recursively.
201  // Currently only mesh, light and material.
202  void LoadCacheRecursive(FbxNode * pNode, FbxAnimLayer * pAnimLayer, bool pSupportVBO)
203  {
204  // Bake material and hook as user data.
205  const int lMaterialCount = pNode->GetMaterialCount();
206  for (int lMaterialIndex = 0; lMaterialIndex < lMaterialCount; ++lMaterialIndex)
207  {
208  FbxSurfaceMaterial * lMaterial = pNode->GetMaterial(lMaterialIndex);
209  if (lMaterial && !lMaterial->GetUserDataPtr())
210  {
211  FbxAutoPtr<MaterialCache> lMaterialCache(new MaterialCache);
212  if (lMaterialCache->Initialize(lMaterial))
213  {
214  lMaterial->SetUserDataPtr(lMaterialCache.Release());
215  }
216  }
217  }
218 
219  FbxNodeAttribute* lNodeAttribute = pNode->GetNodeAttribute();
220  if (lNodeAttribute)
221  {
222  // Bake mesh as VBO(vertex buffer object) into GPU.
223  if (lNodeAttribute->GetAttributeType() == FbxNodeAttribute::eMesh)
224  {
225  FbxMesh * lMesh = pNode->GetMesh();
226  if (pSupportVBO && lMesh && !lMesh->GetUserDataPtr())
227  {
228  FbxAutoPtr<VBOMesh> lMeshCache(new VBOMesh);
229  if (lMeshCache->Initialize(lMesh))
230  {
231  lMesh->SetUserDataPtr(lMeshCache.Release());
232  }
233  }
234  }
235  // Bake light properties.
236  else if (lNodeAttribute->GetAttributeType() == FbxNodeAttribute::eLight)
237  {
238  FbxLight * lLight = pNode->GetLight();
239  if (lLight && !lLight->GetUserDataPtr())
240  {
241  FbxAutoPtr<LightCache> lLightCache(new LightCache);
242  if (lLightCache->Initialize(lLight, pAnimLayer))
243  {
244  lLight->SetUserDataPtr(lLightCache.Release());
245  }
246  }
247  }
248  }
249 
250  const int lChildCount = pNode->GetChildCount();
251  for (int lChildIndex = 0; lChildIndex < lChildCount; ++lChildIndex)
252  {
253  LoadCacheRecursive(pNode->GetChild(lChildIndex), pAnimLayer, pSupportVBO);
254  }
255  }

257  // Unload the cache and release the memory under this node recursively.
258  void UnloadCacheRecursive(FbxNode * pNode)
259  {
260  // Unload the material cache
261  const int lMaterialCount = pNode->GetMaterialCount();
262  for (int lMaterialIndex = 0; lMaterialIndex < lMaterialCount; ++lMaterialIndex)
263  {
264  FbxSurfaceMaterial * lMaterial = pNode->GetMaterial(lMaterialIndex);
265  if (lMaterial && lMaterial->GetUserDataPtr())
266  {
267  MaterialCache * lMaterialCache = static_cast<MaterialCache *>(lMaterial->GetUserDataPtr());
268  lMaterial->SetUserDataPtr(NULL);
269  delete lMaterialCache;
270  }
271  }
272 
273  FbxNodeAttribute* lNodeAttribute = pNode->GetNodeAttribute();
274  if (lNodeAttribute)
275  {
276  // Unload the mesh cache
277  if (lNodeAttribute->GetAttributeType() == FbxNodeAttribute::eMesh)
278  {
279  FbxMesh * lMesh = pNode->GetMesh();
280  if (lMesh && lMesh->GetUserDataPtr())
281  {
282  VBOMesh * lMeshCache = static_cast<VBOMesh *>(lMesh->GetUserDataPtr());
283  lMesh->SetUserDataPtr(NULL);
284  delete lMeshCache;
285  }
286  }
287  // Unload the light cache
288  else if (lNodeAttribute->GetAttributeType() == FbxNodeAttribute::eLight)
289  {
290  FbxLight * lLight = pNode->GetLight();
291  if (lLight && lLight->GetUserDataPtr())
292  {
293  LightCache * lLightCache = static_cast<LightCache *>(lLight->GetUserDataPtr());
294  lLight->SetUserDataPtr(NULL);
295  delete lLightCache;
296  }
297  }
298  }
299 
300  const int lChildCount = pNode->GetChildCount();
301  for (int lChildIndex = 0; lChildIndex < lChildCount; ++lChildIndex)
302  {
303  UnloadCacheRecursive(pNode->GetChild(lChildIndex));
304  }
305  }

307  // Bake node attributes and materials for this scene and load the textures.
308  void LoadCacheRecursive(FbxScene * pScene, FbxAnimLayer * pAnimLayer, const char * pFbxFileName, bool pSupportVBO)
309  {
310  // Load the textures into GPU, only for file texture now
311  const int lTextureCount = pScene->GetTextureCount();
312  for (int lTextureIndex = 0; lTextureIndex < lTextureCount; ++lTextureIndex)
313  {
314  FbxTexture * lTexture = pScene->GetTexture(lTextureIndex);
315  FbxFileTexture * lFileTexture = FbxCast<FbxFileTexture>(lTexture);
316  if (lFileTexture && !lFileTexture->GetUserDataPtr())
317  {
318  // Try to load the texture from absolute path
319  const FbxString lFileName = lFileTexture->GetFileName();
320 
321  // Only TGA textures are supported now.
322  if (lFileName.Right(3).Upper() != "TGA")
323  {
324  FBXSDK_printf("Only TGA textures are supported now: %s\n", lFileName.Buffer());
325  continue;
326  }
327 
328  GLuint lTextureObject = 0;
329  bool lStatus = LoadTextureFromFile(lFileName, lTextureObject);
330 
331  const FbxString lAbsFbxFileName = FbxPathUtils::Resolve(pFbxFileName);
332  const FbxString lAbsFolderName = FbxPathUtils::GetFolderName(lAbsFbxFileName);
333  if (!lStatus)
334  {
335  // Load texture from relative file name (relative to FBX file)
336  const FbxString lResolvedFileName = FbxPathUtils::Bind(lAbsFolderName, lFileTexture->GetRelativeFileName());
337  lStatus = LoadTextureFromFile(lResolvedFileName, lTextureObject);
338  }
339 
340  if (!lStatus)
341  {
342  // Load texture from file name only (relative to FBX file)
343  const FbxString lTextureFileName = FbxPathUtils::GetFileName(lFileName);
344  const FbxString lResolvedFileName = FbxPathUtils::Bind(lAbsFolderName, lTextureFileName);
345  lStatus = LoadTextureFromFile(lResolvedFileName, lTextureObject);
346  }
347 
348  if (!lStatus)
349  {
350  FBXSDK_printf("Failed to load texture file: %s\n", lFileName.Buffer());
351  continue;
352  }
353 
354  if (lStatus)
355  {
356  GLuint * lTextureName = new GLuint(lTextureObject);
357  lFileTexture->SetUserDataPtr(lTextureName);
358  }
359  }
360  }
361 
362  LoadCacheRecursive(pScene->GetRootNode(), pAnimLayer, pSupportVBO);
363  }

365  // Unload the cache and release the memory fro this scene and release the textures in GPU
366  void UnloadCacheRecursive(FbxScene * pScene)
367  {
368  const int lTextureCount = pScene->GetTextureCount();
369  for (int lTextureIndex = 0; lTextureIndex < lTextureCount; ++lTextureIndex)
370  {
371  FbxTexture * lTexture = pScene->GetTexture(lTextureIndex);
372  FbxFileTexture * lFileTexture = FbxCast<FbxFileTexture>(lTexture);
373  if (lFileTexture && lFileTexture->GetUserDataPtr())
374  {
375  GLuint * lTextureName = static_cast<GLuint *>(lFileTexture->GetUserDataPtr());
376  lFileTexture->SetUserDataPtr(NULL);
377  glDeleteTextures(1, lTextureName);
378  delete lTextureName;
379  }
380  }
381 
382  UnloadCacheRecursive(pScene->GetRootNode());
383  }

409 SceneContext::SceneContext(const char * pFileName, int pWindowWidth, int pWindowHeight, bool pSupportVBO)
410 : mFileName(pFileName), mStatus(UNLOADED),
411 mSdkManager(NULL), mScene(NULL), mImporter(NULL), mCurrentAnimLayer(NULL), mSelectedNode(NULL),
412 mPoseIndex(-1), mCameraStatus(CAMERA_NOTHING), mPause(false), mShadingMode(SHADING_MODE_SHADED),
413 mSupportVBO(pSupportVBO), mCameraZoomMode(ZOOM_FOCAL_LENGTH),
414 mWindowWidth(pWindowWidth), mWindowHeight(pWindowHeight), mDrawText(new DrawText)
415 {
416  if (mFileName == NULL)
417  mFileName = SAMPLE_FILENAME;
418 
419  // initialize cache start and stop time
420  mCache_Start = FBXSDK_TIME_INFINITE;
421  mCache_Stop = FBXSDK_TIME_MINUS_INFINITE;
422 
423  // Create the FBX SDK manager which is the object allocator for almost
424  // all the classes in the SDK and create the scene.
425  InitializeSdkObjects(mSdkManager, mScene);
426 
427  if (mSdkManager)
428  {
429  // Create the importer.
430  int lFileFormat = -1;
431  mImporter = FbxImporter::Create(mSdkManager,"");
432  if (!mSdkManager->GetIOPluginRegistry()->DetectReaderFileFormat(mFileName, lFileFormat) )
433  {
434  // Unrecognizable file format. Try to fall back to FbxImporter::eFBX_BINARY
435  lFileFormat = mSdkManager->GetIOPluginRegistry()->FindReaderIDByDescription( "FBX binary (*.fbx)" );;
436  }
437 
438  // Initialize the importer by providing a filename.
439  if(mImporter->Initialize(mFileName, lFileFormat) == true)
440  {
441  // The file is going to be imported at
442  // the end of the first display callback.
443  mWindowMessage = "Importing file ";
444  mWindowMessage += mFileName;
445  mWindowMessage += "\nPlease wait!";
446 
447  // Set scene status flag to ready to load.
448  mStatus = MUST_BE_LOADED;
449  }
450  else
451  {
452  mWindowMessage = "Unable to open file ";
453  mWindowMessage += mFileName;
454  mWindowMessage += "\nError reported: ";
455  mWindowMessage += mImporter->GetStatus().GetErrorString();
456  mWindowMessage += "\nEsc to exit";
457  }
458  }
459  else
460  {
461  mWindowMessage = "Unable to create the FBX SDK manager";
462  mWindowMessage += "\nEsc to exit";
463  }
464 }

484 bool SceneContext::LoadFile()
485 {
486  bool lResult = false;
487  // Make sure that the scene is ready to load.
488  if (mStatus == MUST_BE_LOADED)
489  {
490  if (mImporter->Import(mScene) == true)
491  {
492  // Set the scene status flag to refresh
493  // the scene in the first timer callback.
494  mStatus = MUST_BE_REFRESHED;
495 
496  // Convert Axis System to what is used in this example, if needed
497  FbxAxisSystem SceneAxisSystem = mScene->GetGlobalSettings().GetAxisSystem();
498  FbxAxisSystem OurAxisSystem(FbxAxisSystem::eYAxis, FbxAxisSystem::eParityOdd, FbxAxisSystem::eRightHanded);
499  if( SceneAxisSystem != OurAxisSystem )
500  {
501  OurAxisSystem.ConvertScene(mScene);
502  }
503 
504  // Convert Unit System to what is used in this example, if needed
505  FbxSystemUnit SceneSystemUnit = mScene->GetGlobalSettings().GetSystemUnit();
506  if( SceneSystemUnit.GetScaleFactor() != 1.0 )
507  {
508  //The unit in this example is centimeter.
509  FbxSystemUnit::cm.ConvertScene( mScene);
510  }
511 
512  // Get the list of all the animation stack.
513  mScene->FillAnimStackNameArray(mAnimStackNameArray);
514 
515  // Get the list of all the cameras in the scene.
516  FillCameraArray(mScene, mCameraArray);
517 
518  // Convert mesh, NURBS and patch into triangle mesh
519  FbxGeometryConverter lGeomConverter(mSdkManager);
520  lGeomConverter.Triangulate(mScene, /*replace*/true);
521 
522  // Bake the scene for one frame
523  LoadCacheRecursive(mScene, mCurrentAnimLayer, mFileName, mSupportVBO);
524 
525  // Convert any .PC2 point cache data into the .MC format for
526  // vertex cache deformer playback.
527  PreparePointCacheData(mScene, mCache_Start, mCache_Stop);
528 
529  // Get the list of pose in the scene
530  FillPoseArray(mScene, mPoseArray);
531 
532  // Initialize the window message.
533  mWindowMessage = "File ";
534  mWindowMessage += mFileName;
535  mWindowMessage += "\nClick on the right mouse button to enter menu.";
536  mWindowMessage += "\nEsc to exit.";
537 
538  // Initialize the frame period.
539  mFrameTime.SetTime(0, 0, 0, 1, 0, mScene->GetGlobalSettings().GetTimeMode());
540 
541  // Print the keyboard shortcuts.
542  FBXSDK_printf("Play/Pause Animation: Space Bar.\n");
543  FBXSDK_printf("Camera Rotate: Left Mouse Button.\n");
544  FBXSDK_printf("Camera Pan: Left Mouse Button + Middle Mouse Button.\n");
545  FBXSDK_printf("Camera Zoom: Middle Mouse Button.\n");
546 
547  lResult = true;
548  }
549  else
550  {
551  // Import failed, set the scene status flag accordingly.
552  mStatus = UNLOADED;
553 
554  mWindowMessage = "Unable to import file ";
555  mWindowMessage += mFileName;
556  mWindowMessage += "\nError reported: ";
557  mWindowMessage += mImporter->GetStatus().GetErrorString();
558  }
559 
560  // Destroy the importer to release the file.
561  mImporter->Destroy();
562  mImporter = NULL;
563  }
564 
565  return lResult;
566 }

568 bool SceneContext::SetCurrentAnimStack(int pIndex)
569 {
570  const int lAnimStackCount = mAnimStackNameArray.GetCount();
571  if (!lAnimStackCount || pIndex >= lAnimStackCount)
572  {
573  return false;
574  }
575 
576  // select the base layer from the animation stack
577  FbxAnimStack * lCurrentAnimationStack = mScene->FindMember<FbxAnimStack>(mAnimStackNameArray[pIndex]->Buffer());
578  if (lCurrentAnimationStack == NULL)
579  {
580  // this is a problem. The anim stack should be found in the scene!
581  return false;
582  }
583 
584  // we assume that the first animation layer connected to the animation stack is the base layer
585  // (this is the assumption made in the FBXSDK)
586  mCurrentAnimLayer = lCurrentAnimationStack->GetMember<FbxAnimLayer>();
587  mScene->SetCurrentAnimationStack(lCurrentAnimationStack);
588 
589  FbxTakeInfo* lCurrentTakeInfo = mScene->GetTakeInfo(*(mAnimStackNameArray[pIndex]));
590  if (lCurrentTakeInfo)
591  {
592  mStart = lCurrentTakeInfo->mLocalTimeSpan.GetStart();
593  mStop = lCurrentTakeInfo->mLocalTimeSpan.GetStop();
594  }
595  else
596  {
597  // Take the time line value
598  FbxTimeSpan lTimeLineTimeSpan;
599  mScene->GetGlobalSettings().GetTimelineDefaultTimeSpan(lTimeLineTimeSpan);
600 
601  mStart = lTimeLineTimeSpan.GetStart();
602  mStop = lTimeLineTimeSpan.GetStop();
603  }
604 
605  // check for smallest start with cache start
606  if(mCache_Start < mStart)
607  mStart = mCache_Start;
608 
609  // check for biggest stop with cache stop
610  if(mCache_Stop > mStop)
611  mStop = mCache_Stop;
612 
613  // move to beginning
614  mCurrentTime = mStart;
615 
616  // Set the scene status flag to refresh
617  // the scene in the next timer callback.
618  mStatus = MUST_BE_REFRESHED;
619 
620  return true;
621 }

643 void SceneContext::OnTimerClick() const
644 {
645  // Loop in the animation stack if not paused.
646  if (mStop > mStart && !mPause)
647  {
648  // Set the scene status flag to refresh
649  // the scene in the next timer callback.
650  mStatus = MUST_BE_REFRESHED;
651 
652  mCurrentTime += mFrameTime;
653 
654  if (mCurrentTime > mStop)
655  {
656  mCurrentTime = mStart;
657  }
658  }
659  // Avoid displaying the same frame on
660  // and on if the animation stack has no length.
661  else
662  {
663  // Set the scene status flag to avoid refreshing
664  // the scene in the next timer callback.
665  mStatus = REFRESHED;
666  }
667 }

670 bool SceneContext::OnDisplay()
671 {
672  glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);
