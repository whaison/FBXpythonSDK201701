13 // Illustrates the followings:
14 // 1. How to use FbxAMatrix in transformation calculation.
15 // 2. What EvaluateGlobalTransform() and EvaluateLocalTransform() actually do.
16 // 3. How to get global and local transform of each joint in a joint hierarchy.
17 //
18 // Steps:
19 // 1. Initialize FBX SDK Manager and FBX Scene.
20 // 2. Load the input file to scene.
21 // 3. Calculate global and local transform by EvaluateGlobalTransform() and EvaluateLocalTransform().
22 // 4. Alternative way to calculate global and local transform from scratch by node's properties.
23 // 5. Compare, the above two results should be the same.
24 // 6. Display the joint's global and local transformation.
25 // 7. Destroy the FBX SDK Manager and FBX Scene.
26 //
27 // Notice: This sample does not take rotation order into consideration. It only shows the transformation
28 // calculation with XYZ rotation order.

189 /*
190 Terminology:
191 Suffix "M" means this is a matrix, suffix "V" means it is a vector.
192 T is translation.
193 R is rotation.
194 S is scaling.
195 SH is shear.
196 GlobalRM(x) means the Global Rotation Matrix of node "x".
197 GlobalRM(P(x)) means the Global Rotation Matrix of the parent node of node "x".
198 All other transforms are described in the similar way.
199 
200 The algorithm description:
201 To calculate global transform of a node x according to different InheritType,
202 we need to calculate GlobalTM(x) and [GlobalRM(x) * (GlobalSHM(x) * GlobalSM(x))] separately.
203 GlobalM(x) = GlobalTM(x) * [GlobalRM(x) * (GlobalSHM(x) * GlobalSM(x))];
204 
205 InhereitType = RrSs:
206 GlobalRM(x) * (GlobalSHM(x) * GlobalSM(x)) = GlobalRM(P(x)) * LocalRM(x) * [GlobalSHM(P(x)) * GlobalSM(P(x))] * LocalSM(x);
207 
208 InhereitType = RSrs:
209 GlobalRM(x) * (GlobalSHM(x) * GlobalSM(x)) = GlobalRM(P(x)) * [GlobalSHM(P(x)) * GlobalSM(P(x))] * LocalRM(x) * LocalSM(x);
210 
211 InhereitType = Rrs:
212 GlobalRM(x) * (GlobalSHM(x) * GlobalSM(x)) = GlobalRM(P(x)) * LocalRM(x) * LocalSM(x);
213 
214 LocalM(x)= TM(x) * RoffsetM(x) * RpivotM(x) * RpreM(x) * RM(x) * RpostM(x) * RpivotM(x)^-1 * SoffsetM(x) *SpivotM(x) * SM(x) * SpivotM(x)^-1
215 LocalTWithAllPivotAndOffsetInformationV(x) = Local(x).GetT();
216 GlobalTV(x) = GlobalM(P(x)) * LocalTWithAllPivotAndOffsetInformationV(x);
217 
218 Notice: FBX SDK does not support shear yet, so all local transform won't have shear.
219 However, global transform might bring in shear by combine the global transform of node in higher hierarchy.
220 For example, if you scale the parent by a non-uniform scale and then rotate the child node, then a shear will
221 be generated on the child node's global transform.
222 In this case, we always compensates shear and store it in the scale matrix too according to following formula:
223 Shear*Scaling = RotationMatrix.Inverse * TranslationMatrix.Inverse * WholeTranformMatrix
224 */
225 FbxAMatrix CalculateGlobalTransform(FbxNode* pNode)
226 {
227  FbxAMatrix lTranlationM, lScalingM, lScalingPivotM, lScalingOffsetM, lRotationOffsetM, lRotationPivotM, \
228  lPreRotationM, lRotationM, lPostRotationM, lTransform;
229 
230  FbxAMatrix lParentGX, lGlobalT, lGlobalRS;
231 
232  if(!pNode)
233  {
234  lTransform.SetIdentity();
235  return lTransform;
236  }
237 
238  // Construct translation matrix
239  FbxVector4 lTranslation = pNode->LclTranslation.Get();
240  lTranlationM.SetT(lTranslation);
241 
242  // Construct rotation matrices
243  FbxVector4 lRotation = pNode->LclRotation.Get();
244  FbxVector4 lPreRotation = pNode->PreRotation.Get();
245  FbxVector4 lPostRotation = pNode->PostRotation.Get();
246  lRotationM.SetR(lRotation);
247  lPreRotationM.SetR(lPreRotation);
248  lPostRotationM.SetR(lPostRotation);
249 
250  // Construct scaling matrix
251  FbxVector4 lScaling = pNode->LclScaling.Get();
252  lScalingM.SetS(lScaling);
253 
254  // Construct offset and pivot matrices
255  FbxVector4 lScalingOffset = pNode->ScalingOffset.Get();
256  FbxVector4 lScalingPivot = pNode->ScalingPivot.Get();
257  FbxVector4 lRotationOffset = pNode->RotationOffset.Get();
258  FbxVector4 lRotationPivot = pNode->RotationPivot.Get();
259  lScalingOffsetM.SetT(lScalingOffset);
260  lScalingPivotM.SetT(lScalingPivot);
261  lRotationOffsetM.SetT(lRotationOffset);
262  lRotationPivotM.SetT(lRotationPivot);
263 
264  // Calculate the global transform matrix of the parent node
265  FbxNode* lParentNode = pNode->GetParent();
266  if(lParentNode)
267  {
268  lParentGX = CalculateGlobalTransform(lParentNode);
269  }
270  else
271  {
272  lParentGX.SetIdentity();
273  }
274 
275  //Construct Global Rotation
276  FbxAMatrix lLRM, lParentGRM;
277  FbxVector4 lParentGR = lParentGX.GetR();
278  lParentGRM.SetR(lParentGR);
279  lLRM = lPreRotationM * lRotationM * lPostRotationM;
280 
281  //Construct Global Shear*Scaling
282  //FBX SDK does not support shear, to patch this, we use:
283  //Shear*Scaling = RotationMatrix.Inverse * TranslationMatrix.Inverse * WholeTranformMatrix
284  FbxAMatrix lLSM, lParentGSM, lParentGRSM, lParentTM;
285  FbxVector4 lParentGT = lParentGX.GetT();
286  lParentTM.SetT(lParentGT);
287  lParentGRSM = lParentTM.Inverse() * lParentGX;
288  lParentGSM = lParentGRM.Inverse() * lParentGRSM;
289  lLSM = lScalingM;
290 
291  //Do not consider translation now
292  FbxTransform::EInheritType lInheritType = pNode->InheritType.Get();
293  if(lInheritType == FbxTransform::eInheritRrSs)
294  {
295  lGlobalRS = lParentGRM * lLRM * lParentGSM * lLSM;
296  }
297  else if(lInheritType == FbxTransform::eInheritRSrs)
298  {
299  lGlobalRS = lParentGRM * lParentGSM * lLRM * lLSM;
300  }
301  else if(lInheritType == FbxTransform::eInheritRrs)
302  {
303  FbxAMatrix lParentLSM;
304  FbxVector4 lParentLS = lParentNode->LclScaling.Get();
305  lParentLSM.SetS(lParentLS);
306 
307  FbxAMatrix lParentGSM_noLocal = lParentGSM * lParentLSM.Inverse();
308  lGlobalRS = lParentGRM * lLRM * lParentGSM_noLocal * lLSM;
309  }
310  else
311  {
312  FBXSDK_printf("error, unknown inherit type! \n");
313  }
314 
315  // Construct translation matrix
316  // Calculate the local transform matrix
317  lTransform = lTranlationM * lRotationOffsetM * lRotationPivotM * lPreRotationM * lRotationM * lPostRotationM * lRotationPivotM.Inverse()\
318  * lScalingOffsetM * lScalingPivotM * lScalingM * lScalingPivotM.Inverse();
319  FbxVector4 lLocalTWithAllPivotAndOffsetInfo = lTransform.GetT();
320  // Calculate global translation vector according to:
321  // GlobalTranslation = ParentGlobalTransform * LocalTranslationWithPivotAndOffsetInfo
322  FbxVector4 lGlobalTranslation = lParentGX.MultT(lLocalTWithAllPivotAndOffsetInfo);
323  lGlobalT.SetT(lGlobalTranslation);
324 
325  //Construct the whole global transform
326  lTransform = lGlobalT * lGlobalRS;
327 
328  return lTransform;
329 }
