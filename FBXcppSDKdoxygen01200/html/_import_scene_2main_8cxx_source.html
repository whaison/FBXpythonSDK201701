50 int main(int argc, char** argv)
51 {
52  FbxManager* lSdkManager = NULL;
53  FbxScene* lScene = NULL;
54  bool lResult;
55 
56  // Prepare the FBX SDK.
57  InitializeSdkObjects(lSdkManager, lScene);
58  // Load the scene.
59 
60  // The example can take a FBX file as an argument.
61  FbxString lFilePath("");
62  for( int i = 1, c = argc; i < c; ++i )
63  {
64  if( FbxString(argv[i]) == "-test" ) gVerbose = false;
65  else if( lFilePath.IsEmpty() ) lFilePath = argv[i];
66  }
67 
68  if( lFilePath.IsEmpty() )
69  {
70  lResult = false;
71  FBXSDK_printf("\n\nUsage: ImportScene <FBX file name>\n\n");
72  }
73  else
74  {
75  FBXSDK_printf("\n\nFile: %s\n\n", lFilePath.Buffer());
76  lResult = LoadScene(lSdkManager, lScene, lFilePath.Buffer());
77  }
78 
79  if(lResult == false)
80  {
81  FBXSDK_printf("\n\nAn error occurred while loading the scene...");
82  }
83  else
84  {
85  // Display the scene.
86  DisplayMetaData(lScene);
87 
88  FBXSDK_printf("\n\n---------------------\nGlobal Light Settings\n---------------------\n\n");
89 
90  if( gVerbose ) DisplayGlobalLightSettings(&lScene->GetGlobalSettings());
91 
92  FBXSDK_printf("\n\n----------------------\nGlobal Camera Settings\n----------------------\n\n");
93 
94  if( gVerbose ) DisplayGlobalCameraSettings(&lScene->GetGlobalSettings());
95 
96  FBXSDK_printf("\n\n--------------------\nGlobal Time Settings\n--------------------\n\n");
97 
98  if( gVerbose ) DisplayGlobalTimeSettings(&lScene->GetGlobalSettings());
99 
100  FBXSDK_printf("\n\n---------\nHierarchy\n---------\n\n");
101 
102  if( gVerbose ) DisplayHierarchy(lScene);
103 
104  FBXSDK_printf("\n\n------------\nNode Content\n------------\n\n");
105 
106  if( gVerbose ) DisplayContent(lScene);
107 
108  FBXSDK_printf("\n\n----\nPose\n----\n\n");
109 
110  if( gVerbose ) DisplayPose(lScene);
111 
112  FBXSDK_printf("\n\n---------\nAnimation\n---------\n\n");
113 
114  if( gVerbose ) DisplayAnimation(lScene);
115 
116  //now display generic information
117 
118  FBXSDK_printf("\n\n---------\nGeneric Information\n---------\n\n");
119  if( gVerbose ) DisplayGenericInfo(lScene);
120  }
121 
122  // Destroy all objects created by the FBX SDK.
123  DestroySdkObjects(lSdkManager, lResult);
124 
125  return 0;
126 }

142 void DisplayContent(FbxNode* pNode)
143 {
144  FbxNodeAttribute::EType lAttributeType;
145  int i;
146 
147  if(pNode->GetNodeAttribute() == NULL)
148  {
149  FBXSDK_printf("NULL Node Attribute\n\n");
150  }
151  else
152  {
153  lAttributeType = (pNode->GetNodeAttribute()->GetAttributeType());
154 
155  switch (lAttributeType)
156  {
157  default:
158  break;
159  case FbxNodeAttribute::eMarker:
160  DisplayMarker(pNode);
161  break;
162 
163  case FbxNodeAttribute::eSkeleton:
164  DisplaySkeleton(pNode);
165  break;
166 
167  case FbxNodeAttribute::eMesh:
168  DisplayMesh(pNode);
169  break;
170 
171  case FbxNodeAttribute::eNurbs:
172  DisplayNurb(pNode);
173  break;
174 
175  case FbxNodeAttribute::ePatch:
176  DisplayPatch(pNode);
177  break;
178 
179  case FbxNodeAttribute::eCamera:
180  DisplayCamera(pNode);
181  break;
182 
183  case FbxNodeAttribute::eLight:
184  DisplayLight(pNode);
185  break;
186 
187  case FbxNodeAttribute::eLODGroup:
188  DisplayLodGroup(pNode);
189  break;
190  }
191  }
192 
193  DisplayUserProperties(pNode);
194  DisplayTarget(pNode);
195  DisplayPivotsAndLimits(pNode);
196  DisplayTransformPropagation(pNode);
197  DisplayGeometricTransform(pNode);
198 
199  for(i = 0; i < pNode->GetChildCount(); i++)
200  {
201  DisplayContent(pNode->GetChild(i));
202  }
203 }

214 void DisplayTransformPropagation(FbxNode* pNode)
215 {
216  FBXSDK_printf(" Transformation Propagation\n");
217 
218  //
219  // Rotation Space
220  //
221  EFbxRotationOrder lRotationOrder;
222  pNode->GetRotationOrder(FbxNode::eSourcePivot, lRotationOrder);
223 
224  FBXSDK_printf(" Rotation Space: ");
225 
226  switch (lRotationOrder)
227  {
228  case eEulerXYZ:
229  FBXSDK_printf("Euler XYZ\n");
230  break;
231  case eEulerXZY:
232  FBXSDK_printf("Euler XZY\n");
233  break;
234  case eEulerYZX:
235  FBXSDK_printf("Euler YZX\n");
236  break;
237  case eEulerYXZ:
238  FBXSDK_printf("Euler YXZ\n");
239  break;
240  case eEulerZXY:
241  FBXSDK_printf("Euler ZXY\n");
242  break;
243  case eEulerZYX:
244  FBXSDK_printf("Euler ZYX\n");
245  break;
246  case eSphericXYZ:
247  FBXSDK_printf("Spheric XYZ\n");
248  break;
249  }
250 
251  //
252  // Use the Rotation space only for the limits
253  // (keep using eEulerXYZ for the rest)
254  //
255  FBXSDK_printf(" Use the Rotation Space for Limit specification only: %s\n",
256  pNode->GetUseRotationSpaceForLimitOnly(FbxNode::eSourcePivot) ? "Yes" : "No");
257 
258 
259  //
260  // Inherit Type
261  //
262  FbxTransform::EInheritType lInheritType;
263  pNode->GetTransformationInheritType(lInheritType);
264 
265  FBXSDK_printf(" Transformation Inheritance: ");
266 
267  switch (lInheritType)
268  {
269  case FbxTransform::eInheritRrSs:
270  FBXSDK_printf("RrSs\n");
271  break;
272  case FbxTransform::eInheritRSrs:
273  FBXSDK_printf("RSrs\n");
274  break;
275  case FbxTransform::eInheritRrs:
276  FBXSDK_printf("Rrs\n");
277  break;
278  }
279 }

307 void DisplayMetaData(FbxScene* pScene)
308 {

310  if (sceneInfo)
311  {
312  FBXSDK_printf("\n\n--------------------\nMeta-Data\n--------------------\n\n");

315  FBXSDK_printf(" Author: %s\n", sceneInfo->mAuthor.Buffer());
316  FBXSDK_printf(" Keywords: %s\n", sceneInfo->mKeywords.Buffer());
317  FBXSDK_printf(" Revision: %s\n", sceneInfo->mRevision.Buffer());
318  FBXSDK_printf(" Comment: %s\n", sceneInfo->mComment.Buffer());
319 
320  FbxThumbnail* thumbnail = sceneInfo->GetSceneThumbnail();
321  if (thumbnail)
322  {
323  FBXSDK_printf(" Thumbnail:\n");
324 
325  switch (thumbnail->GetDataFormat())
326  {
327  case FbxThumbnail::eRGB_24:
328  FBXSDK_printf(" Format: RGB\n");
329  break;
330  case FbxThumbnail::eRGBA_32:
331  FBXSDK_printf(" Format: RGBA\n");
332  break;
333  }
334 
335  switch (thumbnail->GetSize())
336  {
337  default:
338  break;
339  case FbxThumbnail::eNotSet:
340  FBXSDK_printf(" Size: no dimensions specified (%ld bytes)\n", thumbnail->GetSizeInBytes());
341  break;
342  case FbxThumbnail::e64x64:
343  FBXSDK_printf(" Size: 64 x 64 pixels (%ld bytes)\n", thumbnail->GetSizeInBytes());
344  break;
345  case FbxThumbnail::e128x128:
346  FBXSDK_printf(" Size: 128 x 128 pixels (%ld bytes)\n", thumbnail->GetSizeInBytes());
347  }
348  }
349  }
