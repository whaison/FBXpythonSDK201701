67 void InitializeLights(const FbxScene* pScene, const FbxTime & pTime, FbxPose* pPose)
68 {
69  // Set ambient light. Turn on light0 and set its attributes to default (white directional light in Z axis).
70  // If the scene contains at least one light, the attributes of light0 will be overridden.
71  LightCache::IntializeEnvironment(pScene->GetGlobalSettings().GetAmbientColor());
72 
73  // Setting the lights before drawing the whole scene
74  const int lLightCount = pScene->GetSrcObjectCount<FbxLight>();
75  for (int lLightIndex = 0; lLightIndex < lLightCount; ++lLightIndex)
76  {
77  FbxLight * lLight = pScene->GetSrcObject<FbxLight>(lLightIndex);
78  FbxNode * lNode = lLight->GetNode();
79  if (lNode)
80 {
81  FbxAMatrix lGlobalPosition = GetGlobalPosition(lNode, pTime, pPose);
82  FbxAMatrix lGeometryOffset = GetGeometry(lNode);
83  FbxAMatrix lGlobalOffPosition = lGlobalPosition * lGeometryOffset;
84  DrawLight(lNode, pTime, lGlobalOffPosition);
85  }
86  }
87 }

89 // Draw recursively each node of the scene. To avoid recomputing
90 // uselessly the global positions, the global position of each
91 // node is passed to it's children while browsing the node tree.
92 // If the node is part of the given pose for the current scene,
93 // it will be drawn at the position specified in the pose, Otherwise
94 // it will be drawn at the given time.
95 void DrawNodeRecursive(FbxNode* pNode, FbxTime& pTime, FbxAnimLayer* pAnimLayer,
96  FbxAMatrix& pParentGlobalPosition, FbxPose* pPose,
97  ShadingMode pShadingMode)
98 {
99  FbxAMatrix lGlobalPosition = GetGlobalPosition(pNode, pTime, pPose, &pParentGlobalPosition);
100 
101  if (pNode->GetNodeAttribute())
102  {
103  // Geometry offset.
104  // it is not inherited by the children.
105  FbxAMatrix lGeometryOffset = GetGeometry(pNode);
106  FbxAMatrix lGlobalOffPosition = lGlobalPosition * lGeometryOffset;
107 
108  DrawNode(pNode, pTime, pAnimLayer, pParentGlobalPosition, lGlobalOffPosition, pPose, pShadingMode);
109  }
110 
111  const int lChildCount = pNode->GetChildCount();
112  for (int lChildIndex = 0; lChildIndex < lChildCount; ++lChildIndex)
113  {
114  DrawNodeRecursive(pNode->GetChild(lChildIndex), pTime, pAnimLayer, lGlobalPosition, pPose, pShadingMode);
115  }
116 }

118 // Draw the node following the content of it's node attribute.
119 void DrawNode(FbxNode* pNode,
120  FbxTime& pTime,
121  FbxAnimLayer* pAnimLayer,
122  FbxAMatrix& pParentGlobalPosition,
123  FbxAMatrix& pGlobalPosition,
124  FbxPose* pPose, ShadingMode pShadingMode)
125 {
126  FbxNodeAttribute* lNodeAttribute = pNode->GetNodeAttribute();
127 
128  if (lNodeAttribute)
129  {
130  // All lights has been processed before the whole scene because they influence every geometry.
131  if (lNodeAttribute->GetAttributeType() == FbxNodeAttribute::eMarker)
132  {
133  DrawMarker(pGlobalPosition);
134  }
135  else if (lNodeAttribute->GetAttributeType() == FbxNodeAttribute::eSkeleton)
136  {
137  DrawSkeleton(pNode, pParentGlobalPosition, pGlobalPosition);
138  }
139  // NURBS and patch have been converted into triangluation meshes.
140  else if (lNodeAttribute->GetAttributeType() == FbxNodeAttribute::eMesh)
141  {
142  DrawMesh(pNode, pTime, pAnimLayer, pGlobalPosition, pPose, pShadingMode);
143  }
144  else if (lNodeAttribute->GetAttributeType() == FbxNodeAttribute::eCamera)
145  {
146  DrawCamera(pNode, pTime, pAnimLayer, pGlobalPosition);
147  }
148  else if (lNodeAttribute->GetAttributeType() == FbxNodeAttribute::eNull)
149  {
150  DrawNull(pGlobalPosition);
151  }
152  }
153  else
154  {
155  // Draw a Null for nodes without attribute.
156  DrawNull(pGlobalPosition);
157  }
158 }

185 // Draw the vertices of a mesh.
186 void DrawMesh(FbxNode* pNode, FbxTime& pTime, FbxAnimLayer* pAnimLayer,
187  FbxAMatrix& pGlobalPosition, FbxPose* pPose, ShadingMode pShadingMode)
188 {
189  FbxMesh* lMesh = pNode->GetMesh();
190  const int lVertexCount = lMesh->GetControlPointsCount();
191 
192  // No vertex to draw.
193  if (lVertexCount == 0)
194  {
195  return;
196  }
197 
198  const VBOMesh * lMeshCache = static_cast<const VBOMesh *>(lMesh->GetUserDataPtr());
199 
200  // If it has some defomer connection, update the vertices position
201  const bool lHasVertexCache = lMesh->GetDeformerCount(FbxDeformer::eVertexCache) &&
202  (static_cast<FbxVertexCacheDeformer*>(lMesh->GetDeformer(0, FbxDeformer::eVertexCache)))->Active.Get();
203  const bool lHasShape = lMesh->GetShapeCount() > 0;
204  const bool lHasSkin = lMesh->GetDeformerCount(FbxDeformer::eSkin) > 0;
205  const bool lHasDeformation = lHasVertexCache || lHasShape || lHasSkin;
206 
207  FbxVector4* lVertexArray = NULL;
208  if (!lMeshCache || lHasDeformation)
209  {
210  lVertexArray = new FbxVector4[lVertexCount];
211  memcpy(lVertexArray, lMesh->GetControlPoints(), lVertexCount * sizeof(FbxVector4));
212  }
213 
214  if (lHasDeformation)
215  {
216  // Active vertex cache deformer will overwrite any other deformer
217  if (lHasVertexCache)
218  {
219  ReadVertexCacheData(lMesh, pTime, lVertexArray);
220  }
221  else
222  {
223  if (lHasShape)
224  {
225  // Deform the vertex array with the shapes.
226  ComputeShapeDeformation(lMesh, pTime, pAnimLayer, lVertexArray);
227  }
228 
229  //we need to get the number of clusters
230  const int lSkinCount = lMesh->GetDeformerCount(FbxDeformer::eSkin);
231  int lClusterCount = 0;
232  for (int lSkinIndex = 0; lSkinIndex < lSkinCount; ++lSkinIndex)
233  {
234  lClusterCount += ((FbxSkin *)(lMesh->GetDeformer(lSkinIndex, FbxDeformer::eSkin)))->GetClusterCount();
235  }
236  if (lClusterCount)
237  {
238  // Deform the vertex array with the skin deformer.
239  ComputeSkinDeformation(pGlobalPosition, lMesh, pTime, lVertexArray, pPose);
240  }
241  }
242 
243  if (lMeshCache)
244  lMeshCache->UpdateVertexPosition(lMesh, lVertexArray);
245  }
246 
247  glPushMatrix();
248  glMultMatrixd((const double*)pGlobalPosition);
249 
250  if (lMeshCache)
251  {
252  lMeshCache->BeginDraw(pShadingMode);
253  const int lSubMeshCount = lMeshCache->GetSubMeshCount();
254  for (int lIndex = 0; lIndex < lSubMeshCount; ++lIndex)
255  {
256  if (pShadingMode == SHADING_MODE_SHADED)
257  {
258  const FbxSurfaceMaterial * lMaterial = pNode->GetMaterial(lIndex);
259  if (lMaterial)
260  {
261  const MaterialCache * lMaterialCache = static_cast<const MaterialCache *>(lMaterial->GetUserDataPtr());
262  if (lMaterialCache)
263  {
264  lMaterialCache->SetCurrentMaterial();
265  }
266  }
267  else
268  {
269  // Draw green for faces without material
270  MaterialCache::SetDefaultMaterial();
271  }
272  }
273 
274  lMeshCache->Draw(lIndex, pShadingMode);
275  }
276  lMeshCache->EndDraw();
277  }
278  else
279  {
280  // OpenGL driver is too lower and use Immediate Mode
281  glColor4f(0.5f, 0.5f, 0.5f, 1.0f);
282  const int lPolygonCount = lMesh->GetPolygonCount();
283  for (int lPolygonIndex = 0; lPolygonIndex < lPolygonCount; lPolygonIndex++)
284  {
285  const int lVerticeCount = lMesh->GetPolygonSize(lPolygonIndex);
286  glBegin(GL_LINE_LOOP);
287  for (int lVerticeIndex = 0; lVerticeIndex < lVerticeCount; lVerticeIndex++)
288  {
289  glVertex3dv((GLdouble *)lVertexArray[lMesh->GetPolygonVertex(lPolygonIndex, lVerticeIndex)]);
290  }
291  glEnd();
292  }
293  }
294 
295  glPopMatrix();
296 
297  delete [] lVertexArray;
298 }

301 // Deform the vertex array with the shapes contained in the mesh.
302 void ComputeShapeDeformation(FbxMesh* pMesh, FbxTime& pTime, FbxAnimLayer * pAnimLayer, FbxVector4* pVertexArray)
303 {
304  int lVertexCount = pMesh->GetControlPointsCount();
305 
306  FbxVector4* lSrcVertexArray = pVertexArray;
307  FbxVector4* lDstVertexArray = new FbxVector4[lVertexCount];
308  memcpy(lDstVertexArray, pVertexArray, lVertexCount * sizeof(FbxVector4));
309 
310  int lBlendShapeDeformerCount = pMesh->GetDeformerCount(FbxDeformer::eBlendShape);
311  for(int lBlendShapeIndex = 0; lBlendShapeIndex<lBlendShapeDeformerCount; ++lBlendShapeIndex)
312  {
313  FbxBlendShape* lBlendShape = (FbxBlendShape*)pMesh->GetDeformer(lBlendShapeIndex, FbxDeformer::eBlendShape);
314 
315  int lBlendShapeChannelCount = lBlendShape->GetBlendShapeChannelCount();
316  for(int lChannelIndex = 0; lChannelIndex<lBlendShapeChannelCount; ++lChannelIndex)
317  {
318  FbxBlendShapeChannel* lChannel = lBlendShape->GetBlendShapeChannel(lChannelIndex);
319  if(lChannel)
320  {
321  // Get the percentage of influence on this channel.
322  FbxAnimCurve* lFCurve = pMesh->GetShapeChannel(lBlendShapeIndex, lChannelIndex, pAnimLayer);
323  if (!lFCurve) continue;
324  double lWeight = lFCurve->Evaluate(pTime);
325 
326  /*
327  If there is only one targetShape on this channel, the influence is easy to calculate:
328  influence = (targetShape - baseGeometry) * weight * 0.01
329  dstGeometry = baseGeometry + influence
330 
331  But if there are more than one targetShapes on this channel, this is an in-between
332  blendshape, also called progressive morph. The calculation of influence is different.
333 
334  For example, given two in-between targets, the full weight percentage of first target
335  is 50, and the full weight percentage of the second target is 100.
336  When the weight percentage reach 50, the base geometry is already be fully morphed
337  to the first target shape. When the weight go over 50, it begin to morph from the
338  first target shape to the second target shape.
339 
340  To calculate influence when the weight percentage is 25:
341  1. 25 falls in the scope of 0 and 50, the morphing is from base geometry to the first target.
342  2. And since 25 is already half way between 0 and 50, so the real weight percentage change to
343  the first target is 50.
344  influence = (firstTargetShape - baseGeometry) * (25-0)/(50-0) * 100
345  dstGeometry = baseGeometry + influence
346 
347  To calculate influence when the weight percentage is 75:
348  1. 75 falls in the scope of 50 and 100, the morphing is from the first target to the second.
349  2. And since 75 is already half way between 50 and 100, so the real weight percentage change
350  to the second target is 50.
351  influence = (secondTargetShape - firstTargetShape) * (75-50)/(100-50) * 100
352  dstGeometry = firstTargetShape + influence
353  */
354 
355  // Find the two shape indices for influence calculation according to the weight.
356  // Consider index of base geometry as -1.
357 
358  int lShapeCount = lChannel->GetTargetShapeCount();
359  double* lFullWeights = lChannel->GetTargetShapeFullWeights();
360 
361  // Find out which scope the lWeight falls in.
362  int lStartIndex = -1;
363  int lEndIndex = -1;
364  for(int lShapeIndex = 0; lShapeIndex<lShapeCount; ++lShapeIndex)
365  {
366  if(lWeight > 0 && lWeight <= lFullWeights[0])
367  {
368  lEndIndex = 0;
369  break;
370  }
371  if(lWeight > lFullWeights[lShapeIndex] && lWeight < lFullWeights[lShapeIndex+1])
372  {
373  lStartIndex = lShapeIndex;
374  lEndIndex = lShapeIndex + 1;
375  break;
376  }
377  }
378 
379  FbxShape* lStartShape = NULL;
380  FbxShape* lEndShape = NULL;
381  if(lStartIndex > -1)
382  {
383  lStartShape = lChannel->GetTargetShape(lStartIndex);
384  }
385  if(lEndIndex > -1)
386  {
387  lEndShape = lChannel->GetTargetShape(lEndIndex);
388  }
389 
390  //The weight percentage falls between base geometry and the first target shape.
391  if(lStartIndex == -1 && lEndShape)
392  {
393  double lEndWeight = lFullWeights[0];
394  // Calculate the real weight.
395  lWeight = (lWeight/lEndWeight) * 100;
396  // Initialize the lDstVertexArray with vertex of base geometry.
397  memcpy(lDstVertexArray, lSrcVertexArray, lVertexCount * sizeof(FbxVector4));
398  for (int j = 0; j < lVertexCount; j++)
399  {
400  // Add the influence of the shape vertex to the mesh vertex.
401  FbxVector4 lInfluence = (lEndShape->GetControlPoints()[j] - lSrcVertexArray[j]) * lWeight * 0.01;
402  lDstVertexArray[j] += lInfluence;
403  }
404  }
405  //The weight percentage falls between two target shapes.
406  else if(lStartShape && lEndShape)
407  {
408  double lStartWeight = lFullWeights[lStartIndex];
409  double lEndWeight = lFullWeights[lEndIndex];
410  // Calculate the real weight.
411  lWeight = ((lWeight-lStartWeight)/(lEndWeight-lStartWeight)) * 100;
412  // Initialize the lDstVertexArray with vertex of the previous target shape geometry.
413  memcpy(lDstVertexArray, lStartShape->GetControlPoints(), lVertexCount * sizeof(FbxVector4));
414  for (int j = 0; j < lVertexCount; j++)
415  {
416  // Add the influence of the shape vertex to the previous shape vertex.
417  FbxVector4 lInfluence = (lEndShape->GetControlPoints()[j] - lStartShape->GetControlPoints()[j]) * lWeight * 0.01;
418  lDstVertexArray[j] += lInfluence;
419  }
420  }
421  }//If lChannel is valid
422  }//For each blend shape channel
423  }//For each blend shape deformer
424 
425  memcpy(pVertexArray, lDstVertexArray, lVertexCount * sizeof(FbxVector4));
426 
427  delete [] lDstVertexArray;
428 }

430 //Compute the transform matrix that the cluster will transform the vertex.
431 void ComputeClusterDeformation(FbxAMatrix& pGlobalPosition,
432  FbxMesh* pMesh,
433  FbxCluster* pCluster,
434  FbxAMatrix& pVertexTransformMatrix,
435  FbxTime pTime,
436  FbxPose* pPose)
437 {
438  FbxCluster::ELinkMode lClusterMode = pCluster->GetLinkMode();
439 
440  FbxAMatrix lReferenceGlobalInitPosition;
441  FbxAMatrix lReferenceGlobalCurrentPosition;
442  FbxAMatrix lAssociateGlobalInitPosition;
443  FbxAMatrix lAssociateGlobalCurrentPosition;
444  FbxAMatrix lClusterGlobalInitPosition;
445  FbxAMatrix lClusterGlobalCurrentPosition;
446 
447  FbxAMatrix lReferenceGeometry;
448  FbxAMatrix lAssociateGeometry;
449  FbxAMatrix lClusterGeometry;
450 
451  FbxAMatrix lClusterRelativeInitPosition;
452  FbxAMatrix lClusterRelativeCurrentPositionInverse;
453 
454  if (lClusterMode == FbxCluster::eAdditive && pCluster->GetAssociateModel())
455  {
456  pCluster->GetTransformAssociateModelMatrix(lAssociateGlobalInitPosition);
457  // Geometric transform of the model
458  lAssociateGeometry = GetGeometry(pCluster->GetAssociateModel());
459  lAssociateGlobalInitPosition *= lAssociateGeometry;
460  lAssociateGlobalCurrentPosition = GetGlobalPosition(pCluster->GetAssociateModel(), pTime, pPose);
461 
462  pCluster->GetTransformMatrix(lReferenceGlobalInitPosition);
463  // Multiply lReferenceGlobalInitPosition by Geometric Transformation
464  lReferenceGeometry = GetGeometry(pMesh->GetNode());
465  lReferenceGlobalInitPosition *= lReferenceGeometry;
466  lReferenceGlobalCurrentPosition = pGlobalPosition;
467 
468  // Get the link initial global position and the link current global position.
469  pCluster->GetTransformLinkMatrix(lClusterGlobalInitPosition);
470  // Multiply lClusterGlobalInitPosition by Geometric Transformation
471  lClusterGeometry = GetGeometry(pCluster->GetLink());
472  lClusterGlobalInitPosition *= lClusterGeometry;
473  lClusterGlobalCurrentPosition = GetGlobalPosition(pCluster->GetLink(), pTime, pPose);
474 
475  // Compute the shift of the link relative to the reference.
476  //ModelM-1 * AssoM * AssoGX-1 * LinkGX * LinkM-1*ModelM
477  pVertexTransformMatrix = lReferenceGlobalInitPosition.Inverse() * lAssociateGlobalInitPosition * lAssociateGlobalCurrentPosition.Inverse() *
478  lClusterGlobalCurrentPosition * lClusterGlobalInitPosition.Inverse() * lReferenceGlobalInitPosition;
479  }
480  else
481  {
482  pCluster->GetTransformMatrix(lReferenceGlobalInitPosition);
483  lReferenceGlobalCurrentPosition = pGlobalPosition;
484  // Multiply lReferenceGlobalInitPosition by Geometric Transformation
485  lReferenceGeometry = GetGeometry(pMesh->GetNode());
486  lReferenceGlobalInitPosition *= lReferenceGeometry;
487 
488  // Get the link initial global position and the link current global position.
489  pCluster->GetTransformLinkMatrix(lClusterGlobalInitPosition);
490  lClusterGlobalCurrentPosition = GetGlobalPosition(pCluster->GetLink(), pTime, pPose);
491 
492  // Compute the initial position of the link relative to the reference.
493  lClusterRelativeInitPosition = lClusterGlobalInitPosition.Inverse() * lReferenceGlobalInitPosition;
494 
495  // Compute the current position of the link relative to the reference.
496  lClusterRelativeCurrentPositionInverse = lReferenceGlobalCurrentPosition.Inverse() * lClusterGlobalCurrentPosition;
497 
498  // Compute the shift of the link relative to the reference.
499  pVertexTransformMatrix = lClusterRelativeCurrentPositionInverse * lClusterRelativeInitPosition;
500  }
501 }

503 // Deform the vertex array in classic linear way.
504 void ComputeLinearDeformation(FbxAMatrix& pGlobalPosition,
505  FbxMesh* pMesh,
506  FbxTime& pTime,
507  FbxVector4* pVertexArray,
508  FbxPose* pPose)
509 {
510  // All the links must have the same link mode.
511  FbxCluster::ELinkMode lClusterMode = ((FbxSkin*)pMesh->GetDeformer(0, FbxDeformer::eSkin))->GetCluster(0)->GetLinkMode();
512 
513  int lVertexCount = pMesh->GetControlPointsCount();
514  FbxAMatrix* lClusterDeformation = new FbxAMatrix[lVertexCount];
515  memset(lClusterDeformation, 0, lVertexCount * sizeof(FbxAMatrix));
516 
517  double* lClusterWeight = new double[lVertexCount];
518  memset(lClusterWeight, 0, lVertexCount * sizeof(double));
519 
520  if (lClusterMode == FbxCluster::eAdditive)
521  {
522  for (int i = 0; i < lVertexCount; ++i)
523  {
524  lClusterDeformation[i].SetIdentity();
525  }
526  }
527 
528  // For all skins and all clusters, accumulate their deformation and weight
529  // on each vertices and store them in lClusterDeformation and lClusterWeight.
530  int lSkinCount = pMesh->GetDeformerCount(FbxDeformer::eSkin);
531  for ( int lSkinIndex=0; lSkinIndex<lSkinCount; ++lSkinIndex)
532  {
533  FbxSkin * lSkinDeformer = (FbxSkin *)pMesh->GetDeformer(lSkinIndex, FbxDeformer::eSkin);
534 
535  int lClusterCount = lSkinDeformer->GetClusterCount();
536  for ( int lClusterIndex=0; lClusterIndex<lClusterCount; ++lClusterIndex)
537  {
538  FbxCluster* lCluster = lSkinDeformer->GetCluster(lClusterIndex);
539  if (!lCluster->GetLink())
540  continue;
541 
542  FbxAMatrix lVertexTransformMatrix;
543  ComputeClusterDeformation(pGlobalPosition, pMesh, lCluster, lVertexTransformMatrix, pTime, pPose);
544 
545  int lVertexIndexCount = lCluster->GetControlPointIndicesCount();
546  for (int k = 0; k < lVertexIndexCount; ++k)
547  {
548  int lIndex = lCluster->GetControlPointIndices()[k];
549 
550  // Sometimes, the mesh can have less points than at the time of the skinning
551  // because a smooth operator was active when skinning but has been deactivated during export.
552  if (lIndex >= lVertexCount)
553  continue;
554 
555  double lWeight = lCluster->GetControlPointWeights()[k];
556 
557  if (lWeight == 0.0)
558  {
559  continue;
560  }
561 
562  // Compute the influence of the link on the vertex.
563  FbxAMatrix lInfluence = lVertexTransformMatrix;
564  MatrixScale(lInfluence, lWeight);
565 
566  if (lClusterMode == FbxCluster::eAdditive)
567  {
568  // Multiply with the product of the deformations on the vertex.
569  MatrixAddToDiagonal(lInfluence, 1.0 - lWeight);
570  lClusterDeformation[lIndex] = lInfluence * lClusterDeformation[lIndex];
571 
572  // Set the link to 1.0 just to know this vertex is influenced by a link.
573  lClusterWeight[lIndex] = 1.0;
574  }
575  else // lLinkMode == FbxCluster::eNormalize || lLinkMode == FbxCluster::eTotalOne
576  {
577  // Add to the sum of the deformations on the vertex.
578  MatrixAdd(lClusterDeformation[lIndex], lInfluence);
579 
580  // Add to the sum of weights to either normalize or complete the vertex.
581  lClusterWeight[lIndex] += lWeight;
582  }
583  }//For each vertex
584  }//lClusterCount
585  }
586 
587  //Actually deform each vertices here by information stored in lClusterDeformation and lClusterWeight
588  for (int i = 0; i < lVertexCount; i++)
589  {
590  FbxVector4 lSrcVertex = pVertexArray[i];
591  FbxVector4& lDstVertex = pVertexArray[i];
592  double lWeight = lClusterWeight[i];
593 
594  // Deform the vertex if there was at least a link with an influence on the vertex,
595  if (lWeight != 0.0)
596  {
597  lDstVertex = lClusterDeformation[i].MultT(lSrcVertex);
598  if (lClusterMode == FbxCluster::eNormalize)
599  {
600  // In the normalized link mode, a vertex is always totally influenced by the links.
601  lDstVertex /= lWeight;
602  }
603  else if (lClusterMode == FbxCluster::eTotalOne)
604  {
605  // In the total 1 link mode, a vertex can be partially influenced by the links.
606  lSrcVertex *= (1.0 - lWeight);
607  lDstVertex += lSrcVertex;
608  }
609  }
610  }
611 
612  delete [] lClusterDeformation;
613  delete [] lClusterWeight;

616 // Deform the vertex array in Dual Quaternion Skinning way.
617 void ComputeDualQuaternionDeformation(FbxAMatrix& pGlobalPosition,
618  FbxMesh* pMesh,
619  FbxTime& pTime,
620  FbxVector4* pVertexArray,
621  FbxPose* pPose)
622 {
623  // All the links must have the same link mode.
624  FbxCluster::ELinkMode lClusterMode = ((FbxSkin*)pMesh->GetDeformer(0, FbxDeformer::eSkin))->GetCluster(0)->GetLinkMode();
625 
626  int lVertexCount = pMesh->GetControlPointsCount();
627  int lSkinCount = pMesh->GetDeformerCount(FbxDeformer::eSkin);
628 
629  FbxDualQuaternion* lDQClusterDeformation = new FbxDualQuaternion[lVertexCount];
630  memset(lDQClusterDeformation, 0, lVertexCount * sizeof(FbxDualQuaternion));
631 
632  double* lClusterWeight = new double[lVertexCount];
633  memset(lClusterWeight, 0, lVertexCount * sizeof(double));
634 
635  // For all skins and all clusters, accumulate their deformation and weight
636  // on each vertices and store them in lClusterDeformation and lClusterWeight.
637  for ( int lSkinIndex=0; lSkinIndex<lSkinCount; ++lSkinIndex)
638  {
639  FbxSkin * lSkinDeformer = (FbxSkin *)pMesh->GetDeformer(lSkinIndex, FbxDeformer::eSkin);
640  int lClusterCount = lSkinDeformer->GetClusterCount();
641  for ( int lClusterIndex=0; lClusterIndex<lClusterCount; ++lClusterIndex)
642  {
643  FbxCluster* lCluster = lSkinDeformer->GetCluster(lClusterIndex);
644  if (!lCluster->GetLink())
645  continue;
646 
647  FbxAMatrix lVertexTransformMatrix;
648  ComputeClusterDeformation(pGlobalPosition, pMesh, lCluster, lVertexTransformMatrix, pTime, pPose);
649 
650  FbxQuaternion lQ = lVertexTransformMatrix.GetQ();
651  FbxVector4 lT = lVertexTransformMatrix.GetT();
652  FbxDualQuaternion lDualQuaternion(lQ, lT);
653 
654  int lVertexIndexCount = lCluster->GetControlPointIndicesCount();
655  for (int k = 0; k < lVertexIndexCount; ++k)
656  {
657  int lIndex = lCluster->GetControlPointIndices()[k];
658 
659  // Sometimes, the mesh can have less points than at the time of the skinning
660  // because a smooth operator was active when skinning but has been deactivated during export.
661  if (lIndex >= lVertexCount)
662  continue;
663 
664  double lWeight = lCluster->GetControlPointWeights()[k];
665 
666  if (lWeight == 0.0)
667  continue;
668 
669  // Compute the influence of the link on the vertex.
670  FbxDualQuaternion lInfluence = lDualQuaternion * lWeight;
671  if (lClusterMode == FbxCluster::eAdditive)
672  {
673  // Simply influenced by the dual quaternion.
674  lDQClusterDeformation[lIndex] = lInfluence;
675 
676  // Set the link to 1.0 just to know this vertex is influenced by a link.
677  lClusterWeight[lIndex] = 1.0;
678  }
679  else // lLinkMode == FbxCluster::eNormalize || lLinkMode == FbxCluster::eTotalOne
680  {
681  if(lClusterIndex == 0)
682  {
683  lDQClusterDeformation[lIndex] = lInfluence;
684  }
685  else
686  {
687  // Add to the sum of the deformations on the vertex.
688  // Make sure the deformation is accumulated in the same rotation direction.
689  // Use dot product to judge the sign.
690  double lSign = lDQClusterDeformation[lIndex].GetFirstQuaternion().DotProduct(lDualQuaternion.GetFirstQuaternion());
691  if( lSign >= 0.0 )
692  {
693  lDQClusterDeformation[lIndex] += lInfluence;
694  }
695  else
696  {
697  lDQClusterDeformation[lIndex] -= lInfluence;
698  }
699  }
700  // Add to the sum of weights to either normalize or complete the vertex.
701  lClusterWeight[lIndex] += lWeight;
702  }
703  }//For each vertex
704  }//lClusterCount
705  }
706 
707  //Actually deform each vertices here by information stored in lClusterDeformation and lClusterWeight
708  for (int i = 0; i < lVertexCount; i++)
709  {
710  FbxVector4 lSrcVertex = pVertexArray[i];
711  FbxVector4& lDstVertex = pVertexArray[i];
712  double lWeightSum = lClusterWeight[i];
713 
714  // Deform the vertex if there was at least a link with an influence on the vertex,
715  if (lWeightSum != 0.0)
716  {
717  lDQClusterDeformation[i].Normalize();
718  lDstVertex = lDQClusterDeformation[i].Deform(lDstVertex);
719 
720  if (lClusterMode == FbxCluster::eNormalize)
721  {
722  // In the normalized link mode, a vertex is always totally influenced by the links.
723  lDstVertex /= lWeightSum;
724  }
725  else if (lClusterMode == FbxCluster::eTotalOne)
726  {
727  // In the total 1 link mode, a vertex can be partially influenced by the links.
728  lSrcVertex *= (1.0 - lWeightSum);
729  lDstVertex += lSrcVertex;
730  }
731  }
732  }
733 
734  delete [] lDQClusterDeformation;
735  delete [] lClusterWeight;
736 }

738 // Deform the vertex array according to the links contained in the mesh and the skinning type.
739 void ComputeSkinDeformation(FbxAMatrix& pGlobalPosition,
740  FbxMesh* pMesh,
741  FbxTime& pTime,
742  FbxVector4* pVertexArray,
743  FbxPose* pPose)
744 {
745  FbxSkin * lSkinDeformer = (FbxSkin *)pMesh->GetDeformer(0, FbxDeformer::eSkin);
746  FbxSkin::EType lSkinningType = lSkinDeformer->GetSkinningType();
747 
748  if(lSkinningType == FbxSkin::eLinear || lSkinningType == FbxSkin::eRigid)
749  {
750  ComputeLinearDeformation(pGlobalPosition, pMesh, pTime, pVertexArray, pPose);
751  }
752  else if(lSkinningType == FbxSkin::eDualQuaternion)
753  {
754  ComputeDualQuaternionDeformation(pGlobalPosition, pMesh, pTime, pVertexArray, pPose);
755  }
756  else if(lSkinningType == FbxSkin::eBlend)
757  {
758  int lVertexCount = pMesh->GetControlPointsCount();
759 
760  FbxVector4* lVertexArrayLinear = new FbxVector4[lVertexCount];
761  memcpy(lVertexArrayLinear, pMesh->GetControlPoints(), lVertexCount * sizeof(FbxVector4));
762 
763  FbxVector4* lVertexArrayDQ = new FbxVector4[lVertexCount];
764  memcpy(lVertexArrayDQ, pMesh->GetControlPoints(), lVertexCount * sizeof(FbxVector4));
765 
766  ComputeLinearDeformation(pGlobalPosition, pMesh, pTime, lVertexArrayLinear, pPose);
767  ComputeDualQuaternionDeformation(pGlobalPosition, pMesh, pTime, lVertexArrayDQ, pPose);
768 
769  // To blend the skinning according to the blend weights
770  // Final vertex = DQSVertex * blend weight + LinearVertex * (1- blend weight)
771  // DQSVertex: vertex that is deformed by dual quaternion skinning method;
772  // LinearVertex: vertex that is deformed by classic linear skinning method;
773  int lBlendWeightsCount = lSkinDeformer->GetControlPointIndicesCount();
774  for(int lBWIndex = 0; lBWIndex<lBlendWeightsCount; ++lBWIndex)
775  {
776  double lBlendWeight = lSkinDeformer->GetControlPointBlendWeights()[lBWIndex];
777  pVertexArray[lBWIndex] = lVertexArrayDQ[lBWIndex] * lBlendWeight + lVertexArrayLinear[lBWIndex] * (1 - lBlendWeight);
778  }
779  }
780 }

785  FbxVector4* pVertexArray)
