13 // Switch binding from one cylinder to the other one.
14 // Please check out the animation to see difference.
15 // Illustrates how to:
16 // 1. Get skin deformer and cluster;
17 // 2. Detach skin;
18 // 3. Bind and create corresponding bindpose;
19 //
20 // Steps:
21 // 1. Initialize FBX SDK Manager and FBX Scene
22 // 2. Load the input file to scene
23 // 3. Access the two cylinders
24 // 4. Get the skin deformer and all clusters from the first cylinder
25 // 5. Remove clusters, skin deformer and bind pose from the first cylinder
26 // 6. Move joints to proper position for cylinder2
27 // 7. Update clusters, skin deformer and create a bind pose for the second cylinder
28 // 8. Save the scene to output file
29 // 9. Destroy the FBX SDK Manager and FBX Scene

41 {
42  FbxManager* lSdkManager = NULL;
43  FbxScene* lScene = NULL;
44 
45  // Prepare the FBX SDK.
46  InitializeSdkObjects(lSdkManager, lScene);

64 void SwitchBinding(FbxScene * pScene)
65 {
66  //Get the two cylinders.
67  FbxMesh* lCylinder01 = pScene->GetSrcObject<FbxMesh>(0);
68  FbxMesh* lCylinder02 = pScene->GetSrcObject<FbxMesh>(1);
69 
70  //Get the skin deformer, which will be switched from the first cylinder to the the second one.
71  FbxSkin* lSkin = (FbxSkin *) lCylinder01->GetDeformer(0, FbxDeformer::eSkin);
72 
73  // Get all clusters through the skin deformer, later they will be re-used for the second cylinder.
74  FbxArray<FbxCluster*> lClusterArray;
75  for (int lClusterIndex = 0; lClusterIndex < lSkin->GetClusterCount(); ++lClusterIndex)
76  {
77  FbxCluster* lCluster=lSkin->GetCluster(lClusterIndex);
78  lClusterArray.Add(lCluster);
79  }
80 
81  // Detach the first cylinder.
82  // Remove clusters from the skin deformer.
83  for (int lClusterIndex = 0; lClusterIndex < lClusterArray.GetCount(); ++lClusterIndex)
84  lSkin->RemoveCluster(lClusterArray[lClusterIndex]);
85  // Remove the skin deformer from the first cylinder.
86  // Currently, the only deformer connected to lCylinder01 is the skin deformer.
87  lCylinder01->RemoveDeformer(0);
88  // Remove the corresponding bindpose.
89  pScene->RemovePose(0) ;
90 
91  // Bind the second cylinder and create the corresponding bind pose.
92  // Move the joints into the second cylinder. Only need to move the root joint.
93  lClusterArray[0]->GetLink()->LclTranslation.Set(FbxDouble3 ( -11.0322688253132,5.3883395780739,0) );
94 
95  // Prepare an array to collect nodes for bind pose, which is combined of the geometry and joints.
96  FbxArray<FbxNode*> lPoseNodeArray;
97 
98  // Add the second cylinder to node array of bind pose.
99  lPoseNodeArray.Add(lCylinder02->GetNode());
100 
101  // Set proper transformation for each cluster and add them to the skin deformer.
102  // ThansformMatrix is the global transformation of the mesh when the binding happens.
103  // TransformLinkMatrix is the global transformation of the joint(Link) when the binding happens.
104  FbxAMatrix lTransformMatrix, lTransformLinkMatrix;
105  lTransformMatrix = lCylinder02->GetNode()->EvaluateGlobalTransform();
106  for(int lClusterIndex = 0; lClusterIndex < lClusterArray.GetCount(); lClusterIndex++)
107  {
108  // All joints have the same TransformMatrix.
109  lClusterArray[lClusterIndex]->SetTransformMatrix(lTransformMatrix);
110 
111  // Compute global transformation of each joint and set it as TransformLinkMatrix.
112  lTransformLinkMatrix = lClusterArray[lClusterIndex]->GetLink()->EvaluateGlobalTransform();
113  lClusterArray[lClusterIndex]->SetTransformLinkMatrix(lTransformLinkMatrix);
114 
115  // Add cluster to the skin deformer.
116  lSkin->AddCluster(lClusterArray[lClusterIndex]);
117 
118  // Add each joint(Link) to node array of bindpose.
119  lPoseNodeArray.Add(lClusterArray[lClusterIndex]->GetLink());
120  }
121 
122  // Add the skin deformer to the second cylinder.
123  lCylinder02->AddDeformer(lSkin);
124 
125  // Create a pose by the node array and set it as bindpose.
126  FbxPose* lPose = FbxPose::Create(pScene,lCylinder02->GetNode()->GetName());
127  lPose->SetIsBindPose(true);
128  for(int lPoseNodeIndex = 0; lPoseNodeIndex<lPoseNodeArray.GetCount(); ++lPoseNodeIndex)
129  {
130  FbxMatrix lBindMatrix = lPoseNodeArray[lPoseNodeIndex]->EvaluateGlobalTransform();
131  lPose->Add(lPoseNodeArray[lPoseNodeIndex], lBindMatrix);
132  }
133 
134  // Add the bindpose to the scene.
135  pScene->AddPose(lPose);
136 }
