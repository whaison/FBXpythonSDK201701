40 void DrawText::Display(const char * pText)
41 {
42  if (!mGlyph || mTextureName == 0)
43  return;
44 
45  // Push OpenGL attributes.
46  glPushAttrib(GL_ENABLE_BIT);
47  glPushAttrib(GL_COLOR_BUFFER_BIT);
48  glPushAttrib(GL_TEXTURE_BIT);
49 
50  glBindTexture(GL_TEXTURE_2D, mTextureName);
51  glEnable(GL_TEXTURE_2D);
52 
53  // Blend with background color
54  glEnable(GL_BLEND);
55  glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
56 
57  // Visible for double side
58  glDisable(GL_CULL_FACE);
59 
60  // Blend with foreground color
61  glTexEnvi(GL_TEXTURE_ENV, GL_TEXTURE_ENV_MODE, GL_MODULATE);
62 
63  glPushMatrix();
64  const char * pCharacter = pText;
65  while (*pCharacter != '\0')
66  {
67  if (*pCharacter == '\n')
68  {
69  glPopMatrix();
70  // Move to the next line, left align
71  glTranslatef(0, -mPointSize, 0);
72  glPushMatrix();
73  }
74  else
75  {
76  Glyph * lGlyph = mGlyph + (*pCharacter - TEXTURE_MIN_GLYPH);
77  if (lGlyph)
78  {
79  // Draw a rectangle with texture, alpha on.
80  glBegin(GL_POLYGON);
81  glTexCoord2f(lGlyph->texture_left, lGlyph->texture_bottom);
82  glVertex2f(lGlyph->vertex_left * mPointSize, lGlyph->vertex_bottom * mPointSize);
83  glTexCoord2f(lGlyph->texture_right, lGlyph->texture_bottom);
84  glVertex2f(lGlyph->vertex_right * mPointSize, lGlyph->vertex_bottom * mPointSize);
85  glTexCoord2f(lGlyph->texture_right, lGlyph->texture_top);
86  glVertex2f(lGlyph->vertex_right * mPointSize, lGlyph->vertex_top * mPointSize);
87  glTexCoord2f(lGlyph->texture_left, lGlyph->texture_top);
88  glVertex2f(lGlyph->vertex_left * mPointSize, lGlyph->vertex_top * mPointSize);
89  glEnd();
90 
91  const float advance = lGlyph->advance * mPointSize + mGap;
92  glTranslatef(advance, 0, 0);
93  }
94  else
95  {
96  FBXSDK_printf("Invalid character: %c.\n", *pCharacter);
97  }
98 
99  }
100  ++pCharacter;
101  }
102  glPopMatrix();
103 
104  // Pop OpenGL attributes.
105  glPopAttrib();
106  glPopAttrib();
107  glPopAttrib();
108 }

110 void DrawText::Initialize()
111 {
112  FILE * lFile = NULL;
113  FBXSDK_fopen(lFile, TEXTURE_FILENAME, "rb");
114  if (lFile == NULL)
115  {
116  FBXSDK_printf("Warning: Failed to open %s and no text will be displayed.\n", TEXTURE_FILENAME);
117  return;
118  }
119 
120  // Load the texture coordinates and vertex coordinates.
121  mGlyph = new Glyph[TEXTURE_GLYPH_COUNT];
122  fread(mGlyph, sizeof(Glyph), TEXTURE_GLYPH_COUNT, lFile);
123 
124  // Load the texture data, two bytes every pixel.
125  // One byte for luminance, always be 0xFF; The other for alpha.
126  const int lTextureSize = TEXTURE_WIDTH * TEXTURE_HEIGHT * 2;
127  unsigned char * lTextureData = new unsigned char[lTextureSize];
128  if (fread(lTextureData, 1, lTextureSize, lFile) == lTextureSize)
129  {
130  // Save the texture data in GPU
131  glGenTextures(1, &mTextureName);
132  glBindTexture(GL_TEXTURE_2D, mTextureName);
133  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
134  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
135  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_CLAMP);
136  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_CLAMP);
137  glTexImage2D(GL_TEXTURE_2D, 0, GL_LUMINANCE_ALPHA, TEXTURE_WIDTH, TEXTURE_HEIGHT, 0,
138  GL_LUMINANCE_ALPHA, GL_UNSIGNED_BYTE, lTextureData);
139  glBindTexture(GL_TEXTURE_2D, 0);
140  }
141  delete [] lTextureData;
142 
143  fclose(lFile);
144 }
