434 void DisplayCurveKeys(FbxAnimCurve* pCurve)
435 {
436  static const char* interpolation[] = { "?", "constant", "linear", "cubic"};
437  static const char* constantMode[] = { "?", "Standard", "Next" };
438  static const char* cubicMode[] = { "?", "Auto", "Auto break", "Tcb", "User", "Break", "User break" };
439  static const char* tangentWVMode[] = { "?", "None", "Right", "Next left" };
440 
441  FbxTime lKeyTime;
442  float lKeyValue;
443  char lTimeString[256];
444  FbxString lOutputString;
445  int lCount;
446 
447  int lKeyCount = pCurve->KeyGetCount();
448 
449  for(lCount = 0; lCount < lKeyCount; lCount++)
450  {
451  lKeyValue = static_cast<float>(pCurve->KeyGetValue(lCount));
452  lKeyTime = pCurve->KeyGetTime(lCount);
453 
454  lOutputString = " Key Time: ";
455  lOutputString += lKeyTime.GetTimeString(lTimeString, FbxUShort(256));
456  lOutputString += ".... Key Value: ";
457  lOutputString += lKeyValue;
458  lOutputString += " [ ";
459  lOutputString += interpolation[ InterpolationFlagToIndex(pCurve->KeyGetInterpolation(lCount)) ];
460  if ((pCurve->KeyGetInterpolation(lCount)&FbxAnimCurveDef::eInterpolationConstant) == FbxAnimCurveDef::eInterpolationConstant)
461  {
462  lOutputString += " | ";
463  lOutputString += constantMode[ ConstantmodeFlagToIndex(pCurve->KeyGetConstantMode(lCount)) ];
464  }
465  else if ((pCurve->KeyGetInterpolation(lCount)&FbxAnimCurveDef::eInterpolationCubic) == FbxAnimCurveDef::eInterpolationCubic)
466  {
467  lOutputString += " | ";
468  lOutputString += cubicMode[ TangentmodeFlagToIndex(pCurve->KeyGetTangentMode(lCount)) ];
469  lOutputString += " | ";
470  lOutputString += tangentWVMode[ TangentweightFlagToIndex(pCurve->KeyGet(lCount).GetTangentWeightMode()) ];
471  lOutputString += " | ";
472  lOutputString += tangentWVMode[ TangentVelocityFlagToIndex(pCurve->KeyGet(lCount).GetTangentVelocityMode()) ];
473  }
474  lOutputString += " ]";
475  lOutputString += "\n";
476  FBXSDK_printf (lOutputString);
477  }
478 }
