436 // used to expand all treeview items recursively
437 void Expand_All_Recurse(
438  HWND htv,
439  HTREEITEM htvi
440  )
441 {
442  if(htvi == NULL) return;
443 
444  TreeView_Expand(htv, htvi, TVE_EXPAND);
445 
446  while(htvi)
447  {
448  // expand all children
449  htvi = TreeView_GetChild(htv, htvi);
450  Expand_All_Recurse(htv, htvi);
451 
452  // expand all siblings
453  while(htvi)
454  {
455  htvi = TreeView_GetNextSibling(htv, htvi);
456  Expand_All_Recurse(htv, htvi);
457  }
458  }
459 }
