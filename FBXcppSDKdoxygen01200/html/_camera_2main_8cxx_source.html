545 FbxCamera* CreateMyCamera(FbxScene* pScene)
546 {
547  if(!pScene)
548  return NULL;
549 
550  //create a fbx node for camera
551  FbxNode* lMyCameraNode = FbxNode::Create(pScene,"myCameraNode");
552  //connect camera node to root node
553  FbxNode* lRootNode = pScene->GetRootNode();
554  lRootNode->ConnectSrcObject( lMyCameraNode);
555  //create a cameraStereo, it's a node attribute of camera node.
556  FbxCamera* lMyCamera = FbxCamera::Create(pScene,"myCamera");
557  //set Camera as a node attribute of the FBX node.
558  lMyCameraNode->SetNodeAttribute (lMyCamera);
559  //we recommend to use existed format
560  //set camera format
561  lMyCamera->SetFormat( FbxCamera::eNTSC);
562  //set camera aperture format
563  lMyCamera->SetApertureFormat( FbxCamera::e16mmTheatrical);
564  //set camera aperture mode
565  lMyCamera->SetApertureMode(FbxCamera::eVertical);
566  //set camera FOV
567  double lFOV = 45;
568  lMyCamera->FieldOfView.Set( lFOV);
569  //set camera FOV
570  double lFocalLength = lMyCamera->ComputeFocalLength( lFOV);
571  lMyCamera->FocalLength.Set( lFocalLength);
572 
573 
574 
575  return lMyCamera;

579 //This function computes the pixel ratio
580 double ComputePixelRatio( double pWidth, double pHeight, double pScreenRatio)
581 {
582  if( pWidth < 0.0 || pHeight < 0.0)
583  return 0.0;
584 
585  pWidth = GET_MAX( pWidth, 1.0 );
586  pHeight = GET_MAX( pHeight, 1.0 );
587 
588  double lResolutionRatio = (double) pWidth / pHeight;
589 
590  return pScreenRatio / lResolutionRatio;
591 }
