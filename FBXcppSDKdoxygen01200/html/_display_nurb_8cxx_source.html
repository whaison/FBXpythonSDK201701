26 void DisplayNurb(FbxNode* pNode)
27 {
28  FbxNurbs* lNurbs = (FbxNurbs*) pNode->GetNodeAttribute ();
29  int i;
30 
31  DisplayString("Nurb Name: ", (char *) pNode->GetName());
32  DisplayMetaDataConnections(lNurbs);
33 
34  const char* lSurfaceModes[] = { "Raw", "Low No Normals", "Low", "High No Normals", "High" };
35 
36  DisplayString(" Surface Mode: ", lSurfaceModes[lNurbs->GetSurfaceMode()]);
37 
38  int lControlPointsCount = lNurbs->GetControlPointsCount();
39  FbxVector4* lControlPoints = lNurbs->GetControlPoints();
40 
41  for (i = 0; i < lControlPointsCount; i++)
42  {
43  DisplayInt(" Control Point ", i);
44  Display3DVector(" Coordinates: ", lControlPoints[i]);
45  DisplayDouble(" Weight: ", lControlPoints[i][3]);
46  }
47 
48  const char* lNurbTypes[] = { "Periodic", "Closed", "Open" };
49 
50  DisplayString(" Nurb U Type: ", lNurbTypes[lNurbs->GetNurbsUType()]);
51  DisplayInt(" U Count: ", lNurbs->GetUCount());
52  DisplayString(" Nurb V Type: ", lNurbTypes[lNurbs->GetNurbsVType()]);
53  DisplayInt(" V Count: ", lNurbs->GetVCount());
54  DisplayInt(" U Order: ", lNurbs->GetUOrder());
55  DisplayInt(" V Order: ", lNurbs->GetVOrder());
56  DisplayInt(" U Step: ", lNurbs->GetUStep());
57  DisplayInt(" V Step: ", lNurbs->GetVStep());
58 
59  FbxString lString;
60  int lUKnotCount = lNurbs->GetUKnotCount();
61  int lVKnotCount = lNurbs->GetVKnotCount();
62  int lUMultiplicityCount = lNurbs->GetUCount();
63  int lVMultiplicityCount = lNurbs->GetVCount();
64  double* lUKnotVector = lNurbs->GetUKnotVector();
65  double* lVKnotVector = lNurbs->GetVKnotVector();
66  int* lUMultiplicityVector = lNurbs->GetUMultiplicityVector();
67  int* lVMultiplicityVector = lNurbs->GetVMultiplicityVector();
68 
69  lString = " U Knot Vector: ";
70 
71  for (i = 0; i < lUKnotCount; i++)
72  {
73  lString += (float) lUKnotVector[i];
74 
75  if (i < lUKnotCount - 1)
76  {
77  lString += ", ";
78  }
79  }
80 
81  lString += "\n";
82  FBXSDK_printf(lString);
83 
84  lString = " V Knot Vector: ";
85 
86  for (i = 0; i < lVKnotCount; i++)
87  {
88  lString += (float) lVKnotVector[i];
89 
90  if (i < lVKnotCount - 1)
91  {
92  lString += ", ";
93  }
94  }
95 
96  lString += "\n";
97  FBXSDK_printf(lString);
98 
99  lString = " U Multiplicity Vector: ";
100 
101  for (i = 0; i < lUMultiplicityCount; i++)
102  {
103  lString += lUMultiplicityVector[i];
104 
105  if (i < lUMultiplicityCount - 1)
106  {
107  lString += ", ";
108  }
109  }
110 
111  lString += "\n";
112  FBXSDK_printf(lString);
113 
114  lString = " V Multiplicity Vector: ";
115 
116  for (i = 0; i < lVMultiplicityCount; i++)
117  {
118  lString += lVMultiplicityVector[i];
119 
120  if (i < lVMultiplicityCount - 1)
121  {
122  lString += ", ";
123  }
124  }
125 
126  lString += "\n";
127  FBXSDK_printf(lString);
