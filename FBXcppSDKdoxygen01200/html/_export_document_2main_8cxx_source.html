68 // Export document, the format is ascii by default
69 bool SaveDocument(FbxManager* pManager, FbxDocument* pDocument, const char* pFilename, int pFileFormat=-1, bool pEmbedMedia=false)
70 {
71  int lMajor, lMinor, lRevision;
72  bool lStatus = true;
73 
74  // Create an exporter.
75  FbxExporter* lExporter = FbxExporter::Create(pManager, "");
76 
77  if( pFileFormat < 0 || pFileFormat >= pManager->GetIOPluginRegistry()->GetWriterFormatCount() )
78  {
79  // Write in fall back format if pEmbedMedia is true
80  pFileFormat = pManager->GetIOPluginRegistry()->GetNativeWriterFormat();
81 
82  if (!pEmbedMedia)
83  {
84  //Try to export in ASCII if possible
85  int lFormatIndex, lFormatCount = pManager->GetIOPluginRegistry()->GetWriterFormatCount();
86 
87  for (lFormatIndex=0; lFormatIndex<lFormatCount; lFormatIndex++)
88  {
89  if (pManager->GetIOPluginRegistry()->WriterIsFBX(lFormatIndex))
90  {
91  FbxString lDesc =pManager->GetIOPluginRegistry()->GetWriterFormatDescription(lFormatIndex);
92  const char *lASCII = "ascii";
93  if (lDesc.Find(lASCII)>=0)
94  {
95  pFileFormat = lFormatIndex;
96  break;
97  }
98  }
99  }
100  }
101  }
102 
103  // Set the export states. By default, the export states are always set to
104  // true except for the option eEXPORT_TEXTURE_AS_EMBEDDED. The code below
105  // shows how to change these states.
106  IOS_REF.SetBoolProp(EXP_FBX_MATERIAL, true);
107  IOS_REF.SetBoolProp(EXP_FBX_TEXTURE, true);
108  IOS_REF.SetBoolProp(EXP_FBX_EMBEDDED, pEmbedMedia);
109  IOS_REF.SetBoolProp(EXP_FBX_ANIMATION, true);
110  IOS_REF.SetBoolProp(EXP_FBX_GLOBAL_SETTINGS, true);
111 
112  // Initialize the exporter by providing a filename.
113  if(lExporter->Initialize(pFilename, pFileFormat, pManager->GetIOSettings()) == false)
114  {
115  FBXSDK_printf("Call to FbxExporter::Initialize() failed.\n");
116  FBXSDK_printf("Error returned: %s\n\n", lExporter->GetStatus().GetErrorString());
117  return false;
118  }
119 
120  FbxManager::GetFileFormatVersion(lMajor, lMinor, lRevision);
121  FBXSDK_printf("FBX version number for this version of the FBX SDK is %d.%d.%d\n\n", lMajor, lMinor, lRevision);
122 
123  // Export the scene.
124  lStatus = lExporter->Export(pDocument);
125 
126  // Destroy the exporter.
127  lExporter->Destroy();
128  return lStatus;
129 }

131 int main(int argc, char** argv)
132 {
133  FbxManager* lSdkManager = NULL;
134  FbxDocument* lDocument = NULL;
135  bool lResult = false;
136 
137  // Prepare the FBX SDK.
138  InitializeSdkObjects(lSdkManager);
139 
140  // create the main document
141  lDocument = FbxDocument::Create(lSdkManager, "RootDoc");
142 
143  // The example can take an output file name as an argument.
144  const char* lSampleFileName = NULL;
145  for( int i = 1; i < argc; ++i )
146  {
147  if( FBXSDK_stricmp(argv[i], "-test") == 0 ) continue;
148  else if( !lSampleFileName ) lSampleFileName = argv[i];
149  }
150  if( !lSampleFileName ) lSampleFileName = SAMPLE_FILENAME;
151 
152  // Create the scene.
153  lResult = CreateDocument(lSdkManager, lDocument);
154  if( lResult )
155  {
156  //Save the document
157  lResult = SaveDocument(lSdkManager, lDocument, lSampleFileName);

161 
162  // Destroy all objects created by the FBX SDK.
163  DestroySdkObjects(lSdkManager, lResult);
164 
165  return 0;
166 }

168 bool CreateDocument(FbxManager* pManager, FbxDocument* pDocument)
169 {
170  int lCount;
171 
172  // create document info
173  FbxDocumentInfo* lDocInfo = FbxDocumentInfo::Create(pManager,"DocInfo");
174  lDocInfo->mTitle = "Example document";
175  lDocInfo->mSubject = "Illustrates the creation of FbxDocument with geometries, materials and lights.";
176  lDocInfo->mAuthor = "ExportDocument.exe sample program.";
177  lDocInfo->mRevision = "rev. 1.0";
178  lDocInfo->mKeywords = "Fbx document";
179  lDocInfo->mComment = "no particular comments required.";
180 
181  // add the documentInfo
182  pDocument->SetDocumentInfo(lDocInfo);
183 
184  // NOTE: Objects created directly in the SDK Manager are not visible
185  // to the disk save routines unless they are manually connected to the
186  // documents (see below). Ideally, one would directly use the FbxScene/FbxDocument
187  // during the creation of objects so they are automatically connected and become visible
188  // to the disk save routines.
189  FbxNode* lPlane = CreatePlane(pManager, "Plane");
190 
191  // add the geometry to the main document.
192  pDocument->AddRootMember(lPlane);
193  lCount = pDocument->GetRootMemberCount(); // lCount = 1: only the lPlane
194  lCount = pDocument->GetMemberCount(); // lCount = 3: the FbxNode - lPlane; FbxMesh belongs to lPlane; Material that connect to lPlane
195 
196  // Create sub document to contain materials.
197  FbxDocument* lMatDocument = FbxDocument::Create(pManager,"Material");
198 
199  CreateMatDocument(pManager, lMatDocument);
200  // Connect the light sub document to main document
201  pDocument->AddMember(lMatDocument);
202 
203  // Create sub document to contain lights
204  FbxDocument* lLightDocument = FbxDocument::Create(pManager,"Light");
205  CreateLightDocument(pManager, lLightDocument);
206  // Connect the light sub document to main document
207  pDocument->AddMember(lLightDocument);
208 
209  lCount = pDocument->GetMemberCount(); // lCount = 5 : 3 add two sub document
210 
211  // document can contain animation. Please refer to other sample about how to set animation
212  pDocument->CreateAnimStack("PlanAnim");
213 
214  lCount = pDocument->GetRootMemberCount(); // lCount = 1: only the lPlane
215  lCount = pDocument->GetMemberCount(); // lCount = 7: 5 add AnimStack and AnimLayer
216  lCount = pDocument->GetMemberCount<FbxDocument>(); // lCount = 2
217 
218  return true;
219 }

260 // Create a plane mesh.
261 FbxNode* CreatePlane(FbxManager* pManager, const char* pName)
262 {
263  int i;
264  FbxMesh* lMesh = FbxMesh::Create(pManager,pName);
265 
266  FbxVector4 lControlPoint0(-50, 0, 50);
267  FbxVector4 lControlPoint1(50, 0, 50);
268  FbxVector4 lControlPoint2(50, 100, 50);
269  FbxVector4 lControlPoint3(-50, 100, 50);
270 
271  FbxVector4 lNormalZPos(0, 0, 1);
272 
273  // Create control points.
274  lMesh->InitControlPoints(4);
275  FbxVector4* lControlPoints = lMesh->GetControlPoints();
276 
277  lControlPoints[0] = lControlPoint0;
278  lControlPoints[1] = lControlPoint1;
279  lControlPoints[2] = lControlPoint2;
280  lControlPoints[3] = lControlPoint3;
281 
282 
283  // We want to have one normal for each vertex (or control point),
284  // so we set the mapping mode to eByControlPoint.
285  FbxGeometryElementNormal* lElementNormal= lMesh->CreateElementNormal();
286 
287  lElementNormal->SetMappingMode(FbxGeometryElement::eByControlPoint);
288 
289  // Set the normal values for every control point.
290  lElementNormal->SetReferenceMode(FbxGeometryElement::eDirect);
291 
292  lElementNormal->GetDirectArray().Add(lNormalZPos);
293  lElementNormal->GetDirectArray().Add(lNormalZPos);
294  lElementNormal->GetDirectArray().Add(lNormalZPos);
295  lElementNormal->GetDirectArray().Add(lNormalZPos);
296 
297 
298  // Array of polygon vertices.
299  int lPolygonVertices[] = { 0, 1, 2, 3 };
300 
301  // Create UV for Diffuse channel.
302  FbxGeometryElementUV* lUVDiffuseElement = lMesh->CreateElementUV( "DiffuseUV");
303  FBX_ASSERT( lUVDiffuseElement != NULL);
304  lUVDiffuseElement->SetMappingMode(FbxGeometryElement::eByPolygonVertex);
305  lUVDiffuseElement->SetReferenceMode(FbxGeometryElement::eIndexToDirect);
306 
307  FbxVector2 lVectors0(0, 0);
308  FbxVector2 lVectors1(1, 0);
309  FbxVector2 lVectors2(1, 1);
310  FbxVector2 lVectors3(0, 1);
311 
312  lUVDiffuseElement->GetDirectArray().Add(lVectors0);
313  lUVDiffuseElement->GetDirectArray().Add(lVectors1);
314  lUVDiffuseElement->GetDirectArray().Add(lVectors2);
315  lUVDiffuseElement->GetDirectArray().Add(lVectors3);
316 
317  //Now we have set the UVs as eIndexToDirect reference and in eByPolygonVertex mapping mode
318  //we must update the size of the index array.
319  lUVDiffuseElement->GetIndexArray().SetCount(4);
320 
321  // Create polygons. Assign texture and texture UV indices.
322  // all faces of the cube have the same texture
323  lMesh->BeginPolygon(-1, -1, -1, false);
324 
325  for(i = 0; i < 4; i++)
326  {
327  // Control point index
328  lMesh->AddPolygon(lPolygonVertices[i]);
329 
330  // update the index array of the UVs that map the texture to the face
331  lUVDiffuseElement->GetIndexArray().SetAt(i, i);
332  }
333 
334  lMesh->EndPolygon ();
335 
336  // create a FbxNode
337  FbxNode* lNode = FbxNode::Create(pManager,pName);
338 
339  // set the node attribute
340  lNode->SetNodeAttribute(lMesh);
341 
342  // set the shading mode to view texture
343  lNode->SetShadingMode(FbxNode::eTextureShading);
344 
345  // rotate the plane
346  lNode->LclRotation.Set(FbxVector4(90, 0, 0));
347 
348 
349  // Set material mapping.
350  FbxGeometryElementMaterial* lMaterialElement = lMesh->CreateElementMaterial();
351  lMaterialElement->SetMappingMode(FbxGeometryElement::eByPolygon);
352  lMaterialElement->SetReferenceMode(FbxGeometryElement::eIndexToDirect);
353  if( !lMesh->GetElementMaterial( 0))
354  return NULL;
355 
356  // add material to the node.
357  // the material can't in different document with the geometry node or in sub-document
358  // we create a simple material here which belong to main document
359  FbxString lMaterialName = "material_for_plane";
360  FbxString lShadingName = "Phong";
361  FbxSurfacePhong* lMaterial = FbxSurfacePhong::Create(pManager, lMaterialName.Buffer());
362 
363  lMaterial->Diffuse.Set(FbxDouble3(1.0, 1.0, 0));
364  lMaterial->DiffuseFactor.Set(1.);
365 
366  lNode->AddMaterial(lMaterial);
367 
368  // We are in eByPolygon, so there's only need for index (a plane has 1 polygon).
369  lMaterialElement->GetIndexArray().SetCount(lMesh->GetPolygonCount());
370 
371  // Set the Index to the material
372  for(i=0; i<lMesh->GetPolygonCount(); ++i)
373  lMaterialElement->GetIndexArray().SetAt(i,0);
374 
375  // return the FbxNode
376  return lNode;
377 }

405 // FBX scene must connect materials FbxNode, otherwise materials will not be exported.
406 // FBX document don't need connect materials to FbxNode, it can export standalone materials.
407 FbxSurfacePhong* CreateMaterial(FbxManager* pManager)
408 {
409  FbxString lMaterialName = "material";
410  FbxString lShadingName = "Phong";
411  FbxDouble3 lBlack(0.0, 0.0, 0.0);
412  FbxDouble3 lRed(1.0, 0.0, 0.0);
413  FbxDouble3 lDiffuseColor(0.75, 0.75, 0.0);
414  FbxSurfacePhong* lMaterial = FbxSurfacePhong::Create(pManager, lMaterialName.Buffer());
415 
416  // Generate primary and secondary colors.
417  lMaterial->Emissive .Set(lBlack);
418  lMaterial->Ambient .Set(lRed);
419  lMaterial->AmbientFactor .Set(1.);
420  // Add texture for diffuse channel
421  lMaterial->Diffuse .ConnectSrcObject(CreateTexture(pManager));
422  lMaterial->DiffuseFactor .Set(1.);
423  lMaterial->TransparencyFactor .Set(0.4);
424  lMaterial->ShadingModel .Set(lShadingName);
425  lMaterial->Shininess .Set(0.5);
426  lMaterial->Specular .Set(lBlack);
427  lMaterial->SpecularFactor .Set(0.3);
428 
429  return lMaterial;
430 }
