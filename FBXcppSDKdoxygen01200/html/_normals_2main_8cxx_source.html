13 //
14 // The example illustrates how to:
15 // 1) get normals of mesh.
16 // 2) get smoothing info of mesh.
17 // 3) compute smoothing info from normals.
18 // 4) convert hard/soft edges info to smoothing group info.
19 //
20 //Background knowledge:
21 //There are two kinds of smoothing info:
22 //1. Smoothing groups info which is saved by polygon. It usually come from 3ds Max, because 3ds Max can set smoothing groups for polygon.
23 //2. Hard/soft edges info which is saved by edge. It usually come from Maya, because Maya can set hard/soft edges.
24 //
25 //steps:
26 // 1. initialize FBX sdk object.
27 // 2. load fbx scene form the specified file.
28 // 3. Get root node of the scene.
29 // 4. Recursively traverse each node in the scene.
30 // 5. Get normals of mesh, according to different mapping mode and reference mode.
31 // 6. Recursively traverse each node in the scene.
32 // 7. Computing smoothing info from normals or convert smoothing info
33 // 8. Get smoothing info of mesh, according to different mapping mode and reference mode.
34 // 9. Destroy all objects.
35 //

53 int main(int argc, char** argv)
54 {
55  FbxManager* lSdkManager = NULL;
56  FbxScene* lScene = NULL;
57  bool lResult;
58 
59  // Prepare the FBX SDK.
60  InitializeSdkObjects(lSdkManager, lScene);
61 
62  // Load the scene.
63  // The example can take a FBX file as an argument.
64  FbxString lFilePath("");
65  for( int i = 1, c = argc; i < c; ++i )
66  {
67  if( FbxString(argv[i]) == "-test" ) gVerbose = false;
68  else if( lFilePath.IsEmpty() ) lFilePath = argv[i];
69  }
70  if( lFilePath.IsEmpty() ) lFilePath = SAMPLE_FILENAME;
71 
72  FBXSDK_printf("\n\nFile: %s\n\n", lFilePath.Buffer());
73  lResult = LoadScene(lSdkManager, lScene, lFilePath.Buffer());
74 
75  if(lResult == false)
76  {
77  FBXSDK_printf("\n\nAn error occurred while loading the scene...");
78  }
79  else
80  {
81  if(!lScene)
82  {
83  FBX_ASSERT_NOW("null scene");
84  }
85 
86  //get root node of the fbx scene
87  FbxNode* lRootNode = lScene->GetRootNode();
88 
89  //get normals info, if there're mesh in the scene
90  GetNormals(lRootNode);
91 
92  //set me true to compute smoothing info from normals
93  bool lComputeFromNormals = false;
94  //set me true to convert hard/soft edges info to smoothing groups info
95  bool lConvertToSmoothingGroup = false;
96  //get smoothing info, if there're mesh in the scene
97  GetSmoothing(lSdkManager, lRootNode, lComputeFromNormals, lConvertToSmoothingGroup);
98  }
99 
100  //Destroy all objects created by the FBX SDK.
101  DestroySdkObjects(lSdkManager, lResult);
102  return 0;
103 }

108 void GetSmoothing(FbxManager* pSdkManager, FbxNode* pNode, bool pCompute, bool pConvertToSmoothingGroup)
109 {
110  if(!pNode || !pSdkManager)
111  return;
112 
113  //get mesh
114  FbxMesh* lMesh = pNode->GetMesh();
115  if(lMesh)
116  {
117  //print mesh node name
118  FBXSDK_printf("current mesh node: %s\n", pNode->GetName());
119 
120  //if there's no smoothing info in fbx file, but you still want to get smoothing info.
121  //please compute smoothing info from normals.
122  //Another case to recompute smoothing info from normals is:
123  //If users edit normals manually in 3ds Max or Maya and export the scene to FBX with smoothing info,
124  //The smoothing info may NOT match with normals.
125  //the mesh called "fbx_customNormals" in Normals.fbx is the case. All edges are hard, but normals actually represent the "soft" looking.
126  //Generally, the normals in fbx file holds the smoothing result you'd like to get.
127  //If you want to get correct smoothing info(smoothing group or hard/soft edges) which match with normals,
128  //please drop the original smoothing info of fbx file, and recompute smoothing info from normals.
129  //if you want to get soft/hard edge info, please call FbxGeometryConverter::ComputeEdgeSmoothingFromNormals().
130  //if you want to get smoothing group info, please get soft/hard edge info first by ComputeEdgeSmoothingFromNormals()
131  //And then call FbxGeometryConverter::ComputePolygonSmoothingFromEdgeSmoothing().
132  if(pCompute)
133  {
134  FbxGeometryConverter lGeometryConverter(pSdkManager);
135  lGeometryConverter.ComputeEdgeSmoothingFromNormals(lMesh);
136  //convert soft/hard edge info to smoothing group info
137  if(pConvertToSmoothingGroup)
138  lGeometryConverter.ComputePolygonSmoothingFromEdgeSmoothing(lMesh);
139  }
140 
141  //if there is smoothing groups info in your fbx file, but you want to get hard/soft edges info
142  //please use following code:
143  //FbxGeometryConverter lGeometryConverter(lSdkManager);
144  //lGeometryConverter.ComputeEdgeSmoothingFromPolygonSmoothing(lMesh);
145 
146  //get smoothing info
147  FbxGeometryElementSmoothing* lSmoothingElement = lMesh->GetElementSmoothing();
148  if(lSmoothingElement)
149  {
150  //mapping mode is by edge. The mesh usually come from Maya, because Maya can set hard/soft edges.
151  //we can get smoothing info(which edges are soft, which edges are hard) by retrieving each edge.
152  if( lSmoothingElement->GetMappingMode() == FbxGeometryElement::eByEdge )
153  {
154  //Let's get smoothing of each edge, since the mapping mode of smoothing element is by edge
155  for(int lEdgeIndex = 0; lEdgeIndex < lMesh->GetMeshEdgeCount(); lEdgeIndex++)
156  {
157  int lSmoothingIndex = 0;
158  //reference mode is direct, the smoothing index is same as edge index.
159  //get smoothing by the index of edge
160  if( lSmoothingElement->GetReferenceMode() == FbxGeometryElement::eDirect )
161  lSmoothingIndex = lEdgeIndex;
162 
163  //reference mode is index-to-direct, get smoothing by the index-to-direct
164  if(lSmoothingElement->GetReferenceMode() == FbxGeometryElement::eIndexToDirect)
165  lSmoothingIndex = lSmoothingElement->GetIndexArray().GetAt(lEdgeIndex);
166 
167  //Got smoothing of each vertex.
168  int lSmoothingFlag = lSmoothingElement->GetDirectArray().GetAt(lSmoothingIndex);
169  if( gVerbose ) FBXSDK_printf("hard/soft value for edge[%d]: %d \n", lEdgeIndex, lSmoothingFlag);
170  //add your custom code here, to output smoothing or get them into a list, such as KArrayTemplate<int>
171  //. . .
172  }//end for lEdgeIndex
173  }//end eByEdge
174  //mapping mode is by polygon. The mesh usually come from 3ds Max, because 3ds Max can set smoothing groups for polygon.
175  //we can get smoothing info(smoothing group ID for each polygon) by retrieving each polygon.
176  else if(lSmoothingElement->GetMappingMode() == FbxGeometryElement::eByPolygon)
177  {
178  //Let's get smoothing of each polygon, since the mapping mode of smoothing element is by polygon.
179  for(int lPolygonIndex = 0; lPolygonIndex < lMesh->GetPolygonCount(); lPolygonIndex++)
180  {
181  int lSmoothingIndex = 0;
182  //reference mode is direct, the smoothing index is same as polygon index.
183  if( lSmoothingElement->GetReferenceMode() == FbxGeometryElement::eDirect )
184  lSmoothingIndex = lPolygonIndex;
185 
186  //reference mode is index-to-direct, get smoothing by the index-to-direct
187  if(lSmoothingElement->GetReferenceMode() == FbxGeometryElement::eIndexToDirect)
188  lSmoothingIndex = lSmoothingElement->GetIndexArray().GetAt(lPolygonIndex);
189 
190  //Got smoothing of each polygon.
191  int lSmoothingFlag = lSmoothingElement->GetDirectArray().GetAt(lSmoothingIndex);
192  if( gVerbose ) FBXSDK_printf("smoothing group ID for polygon[%d]: %d \n", lPolygonIndex, lSmoothingFlag);
193  //add your custom code here, to output normals or get them into a list, such as KArrayTemplate<int>
194  //. . .
195 
196  }//end for lPolygonIndex //PolygonCount
197 
198  }//end eByPolygonVertex
199  }//end if lSmoothingElement
200  }//end if lMesh
201 
202  //recursively traverse each node in the scene
203  int i, lCount = pNode->GetChildCount();
204  for (i = 0; i < lCount; i++)
205  {
206  GetSmoothing(pSdkManager, pNode->GetChild(i), pCompute, pConvertToSmoothingGroup);
207  }
208 }

210 //get mesh normals info
211 void GetNormals(FbxNode* pNode)
212 {
213  if(!pNode)
214  return;
215 
216  //get mesh
217  FbxMesh* lMesh = pNode->GetMesh();
218  if(lMesh)
219  {
220  //print mesh node name
221  FBXSDK_printf("current mesh node: %s\n", pNode->GetName());
222 
223  //get the normal element
224  FbxGeometryElementNormal* lNormalElement = lMesh->GetElementNormal();
225  if(lNormalElement)
226  {
227  //mapping mode is by control points. The mesh should be smooth and soft.
228  //we can get normals by retrieving each control point
229  if( lNormalElement->GetMappingMode() == FbxGeometryElement::eByControlPoint )
230  {
231  //Let's get normals of each vertex, since the mapping mode of normal element is by control point
232  for(int lVertexIndex = 0; lVertexIndex < lMesh->GetControlPointsCount(); lVertexIndex++)
233  {
234  int lNormalIndex = 0;
235  //reference mode is direct, the normal index is same as vertex index.
236  //get normals by the index of control vertex
237  if( lNormalElement->GetReferenceMode() == FbxGeometryElement::eDirect )
238  lNormalIndex = lVertexIndex;
239 
240  //reference mode is index-to-direct, get normals by the index-to-direct
241  if(lNormalElement->GetReferenceMode() == FbxGeometryElement::eIndexToDirect)
242  lNormalIndex = lNormalElement->GetIndexArray().GetAt(lVertexIndex);
243 
244  //Got normals of each vertex.
245  FbxVector4 lNormal = lNormalElement->GetDirectArray().GetAt(lNormalIndex);
246  if( gVerbose ) FBXSDK_printf("normals for vertex[%d]: %f %f %f %f \n", lVertexIndex, lNormal[0], lNormal[1], lNormal[2], lNormal[3]);
247  //add your custom code here, to output normals or get them into a list, such as KArrayTemplate<FbxVector4>
248  //. . .
249  }//end for lVertexIndex
250  }//end eByControlPoint
251  //mapping mode is by polygon-vertex.
252  //we can get normals by retrieving polygon-vertex.
253  else if(lNormalElement->GetMappingMode() == FbxGeometryElement::eByPolygonVertex)
254  {
255  int lIndexByPolygonVertex = 0;
256  //Let's get normals of each polygon, since the mapping mode of normal element is by polygon-vertex.
257  for(int lPolygonIndex = 0; lPolygonIndex < lMesh->GetPolygonCount(); lPolygonIndex++)
258  {
259  //get polygon size, you know how many vertices in current polygon.
260  int lPolygonSize = lMesh->GetPolygonSize(lPolygonIndex);
261  //retrieve each vertex of current polygon.
262  for(int i = 0; i < lPolygonSize; i++)
263  {
264  int lNormalIndex = 0;
265  //reference mode is direct, the normal index is same as lIndexByPolygonVertex.
266  if( lNormalElement->GetReferenceMode() == FbxGeometryElement::eDirect )
267  lNormalIndex = lIndexByPolygonVertex;
268 
269  //reference mode is index-to-direct, get normals by the index-to-direct
270  if(lNormalElement->GetReferenceMode() == FbxGeometryElement::eIndexToDirect)
271  lNormalIndex = lNormalElement->GetIndexArray().GetAt(lIndexByPolygonVertex);
272 
273  //Got normals of each polygon-vertex.
274  FbxVector4 lNormal = lNormalElement->GetDirectArray().GetAt(lNormalIndex);
275  if( gVerbose ) FBXSDK_printf("normals for polygon[%d]vertex[%d]: %f %f %f %f \n",
276  lPolygonIndex, i, lNormal[0], lNormal[1], lNormal[2], lNormal[3]);
277  //add your custom code here, to output normals or get them into a list, such as KArrayTemplate<FbxVector4>
278  //. . .
279 
280  lIndexByPolygonVertex++;
281  }//end for i //lPolygonSize
282  }//end for lPolygonIndex //PolygonCount
283 
284  }//end eByPolygonVertex
285  }//end if lNormalElement
286 
287  }//end if lMesh
288 
289  //recursively traverse each node in the scene
290  int i, lCount = pNode->GetChildCount();
291  for (i = 0; i < lCount; i++)
292  {
293  GetNormals(pNode->GetChild(i));
294  }
295 }
