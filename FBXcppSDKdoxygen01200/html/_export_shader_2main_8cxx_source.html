40 int main(int argc, char** argv)
41 {
42  FbxManager* lSdkManager = NULL;
43  FbxScene* lScene = NULL;
44  bool lResult;
45 
46  // Prepare the FBX SDK.
47  InitializeSdkObjects(lSdkManager, lScene);
48 
49  // Create the scene.
50  lResult = CreateScene(lSdkManager, lScene);
51 
52  if(lResult == false)
53  {
54  FBXSDK_printf("\n\nAn error occurred while creating the scene...\n");
55  DestroySdkObjects(lSdkManager, lResult);
56  return 0;
57  }
58 
59  // Save the scene.
60 
61  // The example can take an output file name as an argument.
62  const char* lSampleFileName = NULL;
63  for( int i = 1; i < argc; ++i )
64  {
65  if( FBXSDK_stricmp(argv[i], "-test") == 0 ) continue;
66  else if( !lSampleFileName ) lSampleFileName = argv[i];
67  }
68  if( !lSampleFileName ) lSampleFileName = SAMPLE_FILENAME;
69 
70  lResult = SaveScene(lSdkManager, lScene, lSampleFileName, -1, true);
71  if(lResult == false)
72  {
73  FBXSDK_printf("\n\nAn error occurred while saving the scene...\n");
74  DestroySdkObjects(lSdkManager, lResult);
75  return 0;
76  }
77 
78  // Destroy all objects created by the FBX SDK.
79  DestroySdkObjects(lSdkManager, lResult);
80 
81  return 0;
82 }

85 bool CreateScene(FbxManager *pSdkManager, FbxScene* pScene)
86 {
87  // create scene info
88  FbxDocumentInfo* sceneInfo = FbxDocumentInfo::Create(pSdkManager,"SceneInfo");
89  sceneInfo->mTitle = "Example scene";
90  sceneInfo->mSubject = "Illustrates the creation of shader.";
91  sceneInfo->mAuthor = "ExportShader sample program.";
92  sceneInfo->mRevision = "rev. 1.0";
93  sceneInfo->mKeywords = "shader";
94  sceneInfo->mComment = "no particular comments required.";
95 
96  pScene->SetSceneInfo(sceneInfo);
97 
98  // Cube with CgFX shader
99  FbxDouble3 lCgfxTranslation(0, 0, 0);
100  FbxNode* lCgfxNode = CreateCube(pScene, "CgfxCube", lCgfxTranslation);
101 
102  FbxSurfaceMaterial* lCgfxMat = CreateCgfxShader(pScene);
103  lCgfxNode->AddMaterial(lCgfxMat);
104 
105  // Cube with HLSL shader
106  FbxDouble3 lHlslTranslation(20, 0, 0);
107  FbxNode* lHlslNode = CreateCube(pScene, "HlslCube", lHlslTranslation);
108 
109  FbxSurfaceMaterial* lHlslMat = CreateHlslShader(pScene);
110  lHlslNode->AddMaterial(lHlslMat);
111 
112  // Build the node tree.
113  FbxNode* lRootNode = pScene->GetRootNode();
114  lRootNode->AddChild(lCgfxNode);
115  lRootNode->AddChild(lHlslNode);
116 
117  return true;
118 }

120 FbxSurfaceMaterial* CreateCgfxShader(FbxScene* pScene)
121 {
122  FbxSurfaceMaterial* lCgfxMat = FbxSurfaceMaterial::Create(pScene, "CgfxShader" );
123 
124  FbxImplementation* lImpl = FbxImplementation::Create(pScene, FbxString("Cgfx_Implementation"));
125 
126  lCgfxMat->AddImplementation( lImpl );
127  lCgfxMat->SetDefaultImplementation( lImpl );
128  lImpl->RenderAPI = FBXSDK_RENDERING_API_OPENGL;
129  lImpl->RenderAPIVersion = "1.5";
130 
131  lImpl->Language = FBXSDK_SHADING_LANGUAGE_CGFX;
132  lImpl->LanguageVersion = "1.5";
133 
134  FbxBindingTable* lTable = lTable = lImpl->AddNewTable("root", "shader");
135  lImpl->RootBindingName = "root";
136 
137  // shader file
138  lTable->DescAbsoluteURL = CGFX_SHADERFILE;
139  // technique name
140  lTable->DescTAG = "user";
141 
142  FbxProperty lRootProp = FbxProperty::Create( lCgfxMat, FbxCompoundDT, "ShaderExample");
143  // Property "SurfColor"
144  FbxProperty lProp = FbxProperty::Create( lRootProp, FbxDouble3DT, "SurfColor" );
145  FbxDouble3 lSurfVal(0, 1, 0);
146  lProp.Set(lSurfVal);
147  CreateTableEntry(lTable, lProp); // for every property, create a table entry in binding table
148 
149  // Property MasterAlpha
150  lProp = FbxProperty::Create(lRootProp, FbxFloatDT, "MasterAlpha");
151  lProp.Set(0.35);
152  CreateTableEntry(lTable, lProp);
153 
154  // Property EnvSample, the property type is sample, so connect a texture to it
155  lProp = FbxProperty::Create(lRootProp, FbxDouble3DT, "EnvSampler", "EnvSampler");
156  FbxDouble3 lSampleVal(0, 0, 0);
157  lProp.Set(lSampleVal);
158  CreateTableEntry(lTable, lProp);
159 
160  // create texture and connect to property
161  FbxFileTexture* lTexture = FbxFileTexture::Create(pScene, "EnvSamplerTex");
162  lTexture->SetFileName(TEXTURE_FILE);
163  lTexture->SetTextureUse(FbxTexture::eStandard);
164  lTexture->SetMappingType(FbxTexture::eUV);
165 
166  lTexture->ConnectDstProperty(lProp);
167  return lCgfxMat;
168 }

170 FbxSurfaceMaterial* CreateHlslShader(FbxScene* pScene)
171 {
172  FbxSurfaceMaterial* lHlslMat = FbxSurfaceMaterial::Create(pScene, "HlslShader" );
173 
174  FbxImplementation* lImpl = FbxImplementation::Create(pScene, FbxString("Hlsl_Implementation"));
175 
176  lHlslMat->AddImplementation( lImpl );
177  lHlslMat->SetDefaultImplementation( lImpl );
178  lImpl->RenderAPI = FBXSDK_RENDERING_API_DIRECTX;
179  lImpl->RenderAPIVersion = "9.0";
180 
181  lImpl->Language = FBXSDK_SHADING_LANGUAGE_HLSL;
182  lImpl->LanguageVersion = "1.0";
183 
184  FbxBindingTable* lTable = lTable = lImpl->AddNewTable("root", "shader");
185  lImpl->RootBindingName = "root";
186 
187  // shader file
188  lTable->DescAbsoluteURL = HLSL_SHADERFILE;
189  // technique name
190  lTable->DescTAG = "dx9";
191 
192 
193  // Property Script
194  FbxProperty lProp = FbxProperty::Create(lHlslMat, FbxFloatDT, "Script", "Script");
195  lProp.ModifyFlag(FbxPropertyFlags::eUserDefined, true);
196  lProp.Set(0.8);
197  CreateTableEntry(lTable, lProp);
198 
199  // Property reflectStrength
200  lProp = FbxProperty::Create(lHlslMat, FbxFloatDT, "reflectStrength", "reflectStrength");
201  lProp.ModifyFlag(FbxPropertyFlags::eUserDefined, true);
202  lProp.Set(0.35);
203  CreateTableEntry(lTable, lProp);
204 
205  // Property refractStrength
206  lProp = FbxProperty::Create(lHlslMat, FbxFloatDT, "refractStrength", "refractStrength");
207  lProp.ModifyFlag(FbxPropertyFlags::eUserDefined, true);
208  lProp.Set(0.75);
209  CreateTableEntry(lTable, lProp);
210 
211  // Property etas
212  lProp = FbxProperty::Create(lHlslMat, FbxDouble3DT, "etas", "etas");
213  lProp.ModifyFlag(FbxPropertyFlags::eUserDefined, true);
214  FbxDouble3 lEtasVal(0.8, 0.6, 0.4);
215  lProp.Set(lEtasVal);
216  CreateTableEntry(lTable, lProp);
217 
218  // Property cubeMap, the property type is sample, not connect a texture to it, just skip it
219  lProp = FbxProperty::Create(lHlslMat, FbxDouble3DT, "fresnelTex", "fresnelTex");
220  lProp.ModifyFlag(FbxPropertyFlags::eUserDefined, true);
221  FbxDouble3 lTexVal(0, 0, 0);
222  lProp.Set(lTexVal);
223  CreateTableEntry(lTable, lProp);
224 
225  // Property cubeMap, the property type is sample, so connect a texture to it
226  lProp = FbxProperty::Create(lHlslMat, FbxDouble3DT, "cubeMap", "cubeMap");
227  lProp.ModifyFlag(FbxPropertyFlags::eUserDefined, true);
228  FbxDouble3 lMapVal(0, 1, 0);
229  lProp.Set(lMapVal);
230  CreateTableEntry(lTable, lProp);
231 
232  FbxFileTexture* lTexture = FbxFileTexture::Create(pScene, "cubeMapTex");
233  lTexture->SetFileName(TEXTURE_FILE);
234  lTexture->SetTextureUse(FbxTexture::eStandard);
235  lTexture->SetMappingType(FbxTexture::eUV);
236 
237  lTexture->ConnectDstProperty(lProp);
238 
239  return lHlslMat;
240 }

242 void CreateTableEntry(FbxBindingTable* pTable, FbxProperty& pProp)
243 {
244  // create entry
245  FbxBindingTableEntry& lEntry = pTable->AddNewEntry();
246  // set src to the fbx property, in this sample, fbx properties have the same name with shader parameters
247  FbxPropertyEntryView lSrc( &lEntry, true, true );
248  // Because CgFX uses compound property, so do not use pProp.GetName()
249  lSrc.SetProperty( pProp.GetHierarchicalName());
250 
251  // set dst to the shader parameter
252  FbxSemanticEntryView lDst( &lEntry, false, true );
253  lDst.SetSemantic( pProp.GetName() );
254 }
