14 FbxNode * CreatePyramid(FbxScene * pScene, const char * pName, double pBottomWidth, double pHeight)
15 {
16  FbxMesh * lPyramid = FbxMesh::Create(pScene, pName);
17 
18  // Calculate the vertices of the pyramid
19  const double lBottomWidthHalf = pBottomWidth / 2;
20  const FbxVector4 PyramidControlPointArray[] =
21  {
22  FbxVector4(0, pHeight, 0),
23  FbxVector4(lBottomWidthHalf, 0, lBottomWidthHalf),
24  FbxVector4(lBottomWidthHalf, 0, -lBottomWidthHalf),
25  FbxVector4(-lBottomWidthHalf, 0, -lBottomWidthHalf),
26  FbxVector4(-lBottomWidthHalf, 0, lBottomWidthHalf)
27  };
28 
29  // Initialize and set the control points of the mesh
30  const int lControlPointCount = sizeof(PyramidControlPointArray) / sizeof(FbxVector4);
31  lPyramid->InitControlPoints(lControlPointCount);
32  for (int lIndex = 0; lIndex < lControlPointCount; ++lIndex)
33  {
34  lPyramid->SetControlPointAt(PyramidControlPointArray[lIndex], lIndex);
35  }
36 
37  // Set the control point indices of the bottom side of the pyramid
38  lPyramid->BeginPolygon();
39  lPyramid->AddPolygon(1);
40  lPyramid->AddPolygon(4);
41  lPyramid->AddPolygon(3);
42  lPyramid->AddPolygon(2);
43  lPyramid->EndPolygon();
44 
45  // Set the control point indices of the front side of the pyramid
46  lPyramid->BeginPolygon();
47  lPyramid->AddPolygon(0);
48  lPyramid->AddPolygon(1);
49  lPyramid->AddPolygon(2);
50  lPyramid->EndPolygon();
51 
52  // Set the control point indices of the left side of the pyramid
53  lPyramid->BeginPolygon();
54  lPyramid->AddPolygon(0);
55  lPyramid->AddPolygon(2);
56  lPyramid->AddPolygon(3);
57  lPyramid->EndPolygon();
58 
59  // Set the control point indices of the back side of the pyramid
60  lPyramid->BeginPolygon();
61  lPyramid->AddPolygon(0);
62  lPyramid->AddPolygon(3);
63  lPyramid->AddPolygon(4);
64  lPyramid->EndPolygon();
65 
66  // Set the control point indices of the right side of the pyramid
67  lPyramid->BeginPolygon();
68  lPyramid->AddPolygon(0);
69  lPyramid->AddPolygon(4);
70  lPyramid->AddPolygon(1);
71  lPyramid->EndPolygon();
72 
73  // Attach the mesh to a node
74  FbxNode * lPyramidNode = FbxNode::Create(pScene, pName);
75  lPyramidNode->SetNodeAttribute(lPyramid);
76 
77  // Set this node as a child of the root node
78  pScene->GetRootNode()->AddChild(lPyramidNode);
79 
80  return lPyramidNode;
81 }

88 FbxNode* CreateCube(FbxScene* pScene, const char* pName, FbxDouble3& pLclTranslation)
89 {
90  // indices of the vertices per each polygon
91  static int vtxId[24] = {
92  0,1,2,3, // front face (Z+)
93  1,5,6,2, // right side (X+)
94  5,4,7,6, // back face (Z-)
95  4,0,3,7, // left side (X-)
96  0,4,5,1, // bottom face (Y-)
97  3,2,6,7 // top face (Y+)
98  };
99 
100  // control points
101  static Vector4 lControlPoints[8] = {
102  { -5.0, 0.0, 5.0, 1.0}, { 5.0, 0.0, 5.0, 1.0}, { 5.0,10.0, 5.0, 1.0}, { -5.0,10.0, 5.0, 1.0},
103  { -5.0, 0.0, -5.0, 1.0}, { 5.0, 0.0, -5.0, 1.0}, { 5.0,10.0, -5.0, 1.0}, { -5.0,10.0, -5.0, 1.0}
104  };
105 
106  // normals
107  static Vector4 lNormals[8] = {
108  {-0.577350258827209,-0.577350258827209, 0.577350258827209, 1.0},
109  { 0.577350258827209,-0.577350258827209, 0.577350258827209, 1.0},
110  { 0.577350258827209, 0.577350258827209, 0.577350258827209, 1.0},
111  {-0.577350258827209, 0.577350258827209, 0.577350258827209, 1.0},
112  {-0.577350258827209,-0.577350258827209,-0.577350258827209, 1.0},
113  { 0.577350258827209,-0.577350258827209,-0.577350258827209, 1.0},
114  { 0.577350258827209, 0.577350258827209,-0.577350258827209, 1.0},
115  {-0.577350258827209, 0.577350258827209,-0.577350258827209, 1.0}
116  };
117 
118  // uvs
119  static Vector2 lUVs[14] = {
120  { 0.0, 1.0},
121  { 1.0, 0.0},
122  { 0.0, 0.0},
123  { 1.0, 1.0}
124  };
125 
126  // indices of the uvs per each polygon
127  static int uvsId[24] = {
128  0,1,3,2,2,3,5,4,4,5,7,6,6,7,9,8,1,10,11,3,12,0,2,13
129  };
130 
131  // create the main structure.
132  FbxMesh* lMesh = FbxMesh::Create(pScene,"");
133 
134  // Create control points.
135  lMesh->InitControlPoints(8);
136  FbxVector4* vertex = lMesh->GetControlPoints();
137  memcpy((void*)vertex, (void*)lControlPoints, 8*sizeof(FbxVector4));
138 
139  // create the materials.
140  /* Each polygon face will be assigned a unique material.
141  */
142  FbxGeometryElementMaterial* lMaterialElement = lMesh->CreateElementMaterial();
143  lMaterialElement->SetMappingMode(FbxGeometryElement::eAllSame);
144  lMaterialElement->SetReferenceMode(FbxGeometryElement::eIndexToDirect);
145 
146  lMaterialElement->GetIndexArray().Add(0);
147 
148  // Create polygons later after FbxGeometryElementMaterial is created. Assign material indices.
149  int vId = 0;
150  for (int f=0; f<6; f++)
151  {
152  lMesh->BeginPolygon();
153  for (int v=0; v<4; v++)
154  lMesh->AddPolygon(vtxId[vId++]);
155  lMesh->EndPolygon();
156  }
157 
158  // specify normals per control point.
159  FbxGeometryElementNormal* lNormalElement = lMesh->CreateElementNormal();
160  lNormalElement->SetMappingMode(FbxGeometryElement::eByControlPoint);
161  lNormalElement->SetReferenceMode(FbxGeometryElement::eDirect);
162 
163  for (int n=0; n<8; n++)
164  lNormalElement->GetDirectArray().Add(FbxVector4(lNormals[n][0], lNormals[n][1], lNormals[n][2]));
165 
166 
167  // Create the node containing the mesh
168  FbxNode* lNode = FbxNode::Create(pScene,pName);
169  lNode->LclTranslation.Set(pLclTranslation);
170 
171  lNode->SetNodeAttribute(lMesh);
172  lNode->SetShadingMode(FbxNode::eTextureShading);
173 
174  // create UVset
175  FbxGeometryElementUV* lUVElement1 = lMesh->CreateElementUV("UVSet1");
176  FBX_ASSERT( lUVElement1 != NULL);
177  lUVElement1->SetMappingMode(FbxGeometryElement::eByPolygonVertex);
178  lUVElement1->SetReferenceMode(FbxGeometryElement::eIndexToDirect);
179  for (int i = 0; i <4; i++)
180  lUVElement1->GetDirectArray().Add(FbxVector2(lUVs[i][0], lUVs[i][1]));
181 
182  for (int i = 0; i<24; i++)
183  lUVElement1->GetIndexArray().Add(uvsId[i%4]);
184 
185  return lNode;
186 }
