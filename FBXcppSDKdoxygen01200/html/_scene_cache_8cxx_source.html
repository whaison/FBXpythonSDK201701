37  // Get specific property value and connected texture if any.
38  // Value = Property value * Factor property value (if no factor property, multiply by 1).
39  FbxDouble3 GetMaterialProperty(const FbxSurfaceMaterial * pMaterial,
40  const char * pPropertyName,
41  const char * pFactorPropertyName,
42  GLuint & pTextureName)
43  {
44  FbxDouble3 lResult(0, 0, 0);
45  const FbxProperty lProperty = pMaterial->FindProperty(pPropertyName);
46  const FbxProperty lFactorProperty = pMaterial->FindProperty(pFactorPropertyName);
47  if (lProperty.IsValid() && lFactorProperty.IsValid())
48  {
49  lResult = lProperty.Get<FbxDouble3>();
50  double lFactor = lFactorProperty.Get<FbxDouble>();
51  if (lFactor != 1)
52  {
53  lResult[0] *= lFactor;
54  lResult[1] *= lFactor;
55  lResult[2] *= lFactor;
56  }
57  }
58 
59  if (lProperty.IsValid())
60  {
61  const int lTextureCount = lProperty.GetSrcObjectCount<FbxFileTexture>();
62  if (lTextureCount)
63  {
64  const FbxFileTexture* lTexture = lProperty.GetSrcObject<FbxFileTexture>();
65  if (lTexture && lTexture->GetUserDataPtr())
66  {
67  pTextureName = *(static_cast<GLuint *>(lTexture->GetUserDataPtr()));
68  }
69  }
70  }
71 
72  return lResult;
73  }

101 bool VBOMesh::Initialize(const FbxMesh *pMesh)
102 {
103  if (!pMesh->GetNode())
104  return false;
105 
106  const int lPolygonCount = pMesh->GetPolygonCount();
107 
108  // Count the polygon count of each material
109  FbxLayerElementArrayTemplate<int>* lMaterialIndice = NULL;
110  FbxGeometryElement::EMappingMode lMaterialMappingMode = FbxGeometryElement::eNone;
111  if (pMesh->GetElementMaterial())
112  {
113  lMaterialIndice = &pMesh->GetElementMaterial()->GetIndexArray();
114  lMaterialMappingMode = pMesh->GetElementMaterial()->GetMappingMode();
115  if (lMaterialIndice && lMaterialMappingMode == FbxGeometryElement::eByPolygon)
116  {
117  FBX_ASSERT(lMaterialIndice->GetCount() == lPolygonCount);
118  if (lMaterialIndice->GetCount() == lPolygonCount)
119  {
120  // Count the faces of each material
121  for (int lPolygonIndex = 0; lPolygonIndex < lPolygonCount; ++lPolygonIndex)
122  {
123  const int lMaterialIndex = lMaterialIndice->GetAt(lPolygonIndex);
124  if (mSubMeshes.GetCount() < lMaterialIndex + 1)
125  {
126  mSubMeshes.Resize(lMaterialIndex + 1);
127  }
128  if (mSubMeshes[lMaterialIndex] == NULL)
129  {
130  mSubMeshes[lMaterialIndex] = new SubMesh;
131  }
132  mSubMeshes[lMaterialIndex]->TriangleCount += 1;
133  }
134 
135  // Make sure we have no "holes" (NULL) in the mSubMeshes table. This can happen
136  // if, in the loop above, we resized the mSubMeshes by more than one slot.
137  for (int i = 0; i < mSubMeshes.GetCount(); i++)
138  {
139  if (mSubMeshes[i] == NULL)
140  mSubMeshes[i] = new SubMesh;
141  }
142 
143  // Record the offset (how many vertex)
144  const int lMaterialCount = mSubMeshes.GetCount();
145  int lOffset = 0;
146  for (int lIndex = 0; lIndex < lMaterialCount; ++lIndex)
147  {
148  mSubMeshes[lIndex]->IndexOffset = lOffset;
149  lOffset += mSubMeshes[lIndex]->TriangleCount * 3;
150  // This will be used as counter in the following procedures, reset to zero
151  mSubMeshes[lIndex]->TriangleCount = 0;
152  }
153  FBX_ASSERT(lOffset == lPolygonCount * 3);
154  }
155  }
156  }
157 
158  // All faces will use the same material.
159  if (mSubMeshes.GetCount() == 0)
160  {
161  mSubMeshes.Resize(1);
162  mSubMeshes[0] = new SubMesh();
163  }
164 
165  // Congregate all the data of a mesh to be cached in VBOs.
166  // If normal or UV is by polygon vertex, record all vertex attributes by polygon vertex.
167  mHasNormal = pMesh->GetElementNormalCount() > 0;
168  mHasUV = pMesh->GetElementUVCount() > 0;
169  FbxGeometryElement::EMappingMode lNormalMappingMode = FbxGeometryElement::eNone;
170  FbxGeometryElement::EMappingMode lUVMappingMode = FbxGeometryElement::eNone;
171  if (mHasNormal)
172  {
173  lNormalMappingMode = pMesh->GetElementNormal(0)->GetMappingMode();
174  if (lNormalMappingMode == FbxGeometryElement::eNone)
175  {
176  mHasNormal = false;
177  }
178  if (mHasNormal && lNormalMappingMode != FbxGeometryElement::eByControlPoint)
179  {
180  mAllByControlPoint = false;
181  }
182  }
183  if (mHasUV)
184  {
185  lUVMappingMode = pMesh->GetElementUV(0)->GetMappingMode();
186  if (lUVMappingMode == FbxGeometryElement::eNone)
187  {
188  mHasUV = false;
189  }
190  if (mHasUV && lUVMappingMode != FbxGeometryElement::eByControlPoint)
191  {
192  mAllByControlPoint = false;
193  }
194  }
195 
196  // Allocate the array memory, by control point or by polygon vertex.
197  int lPolygonVertexCount = pMesh->GetControlPointsCount();
198  if (!mAllByControlPoint)
199  {
200  lPolygonVertexCount = lPolygonCount * TRIANGLE_VERTEX_COUNT;
201  }
202  float * lVertices = new float[lPolygonVertexCount * VERTEX_STRIDE];
203  unsigned int * lIndices = new unsigned int[lPolygonCount * TRIANGLE_VERTEX_COUNT];
204  float * lNormals = NULL;
205  if (mHasNormal)
206  {
207  lNormals = new float[lPolygonVertexCount * NORMAL_STRIDE];
208  }
209  float * lUVs = NULL;
210  FbxStringList lUVNames;
211  pMesh->GetUVSetNames(lUVNames);
212  const char * lUVName = NULL;
213  if (mHasUV && lUVNames.GetCount())
214  {
215  lUVs = new float[lPolygonVertexCount * UV_STRIDE];
216  lUVName = lUVNames[0];
217  }
218 
219  // Populate the array with vertex attribute, if by control point.
220  const FbxVector4 * lControlPoints = pMesh->GetControlPoints();
221  FbxVector4 lCurrentVertex;
222  FbxVector4 lCurrentNormal;
223  FbxVector2 lCurrentUV;
224  if (mAllByControlPoint)
225  {
226  const FbxGeometryElementNormal * lNormalElement = NULL;
227  const FbxGeometryElementUV * lUVElement = NULL;
228  if (mHasNormal)
229  {
230  lNormalElement = pMesh->GetElementNormal(0);
231  }
232  if (mHasUV)
233  {
234  lUVElement = pMesh->GetElementUV(0);
235  }
236  for (int lIndex = 0; lIndex < lPolygonVertexCount; ++lIndex)
237  {
238  // Save the vertex position.
239  lCurrentVertex = lControlPoints[lIndex];
240  lVertices[lIndex * VERTEX_STRIDE] = static_cast<float>(lCurrentVertex[0]);
241  lVertices[lIndex * VERTEX_STRIDE + 1] = static_cast<float>(lCurrentVertex[1]);
242  lVertices[lIndex * VERTEX_STRIDE + 2] = static_cast<float>(lCurrentVertex[2]);
243  lVertices[lIndex * VERTEX_STRIDE + 3] = 1;
244 
245  // Save the normal.
246  if (mHasNormal)
247  {
248  int lNormalIndex = lIndex;
249  if (lNormalElement->GetReferenceMode() == FbxLayerElement::eIndexToDirect)
250  {
251  lNormalIndex = lNormalElement->GetIndexArray().GetAt(lIndex);
252  }
253  lCurrentNormal = lNormalElement->GetDirectArray().GetAt(lNormalIndex);
254  lNormals[lIndex * NORMAL_STRIDE] = static_cast<float>(lCurrentNormal[0]);
255  lNormals[lIndex * NORMAL_STRIDE + 1] = static_cast<float>(lCurrentNormal[1]);
256  lNormals[lIndex * NORMAL_STRIDE + 2] = static_cast<float>(lCurrentNormal[2]);
257  }
258 
259  // Save the UV.
260  if (mHasUV)
261  {
262  int lUVIndex = lIndex;
263  if (lUVElement->GetReferenceMode() == FbxLayerElement::eIndexToDirect)
264  {
265  lUVIndex = lUVElement->GetIndexArray().GetAt(lIndex);
266  }
267  lCurrentUV = lUVElement->GetDirectArray().GetAt(lUVIndex);
268  lUVs[lIndex * UV_STRIDE] = static_cast<float>(lCurrentUV[0]);
269  lUVs[lIndex * UV_STRIDE + 1] = static_cast<float>(lCurrentUV[1]);
270  }
271  }
272 
273  }
274 
275  int lVertexCount = 0;
276  for (int lPolygonIndex = 0; lPolygonIndex < lPolygonCount; ++lPolygonIndex)
277  {
278  // The material for current face.
279  int lMaterialIndex = 0;
280  if (lMaterialIndice && lMaterialMappingMode == FbxGeometryElement::eByPolygon)
281  {
282  lMaterialIndex = lMaterialIndice->GetAt(lPolygonIndex);
283  }
284 
285  // Where should I save the vertex attribute index, according to the material
286  const int lIndexOffset = mSubMeshes[lMaterialIndex]->IndexOffset +
287  mSubMeshes[lMaterialIndex]->TriangleCount * 3;
288  for (int lVerticeIndex = 0; lVerticeIndex < TRIANGLE_VERTEX_COUNT; ++lVerticeIndex)
289  {
290  const int lControlPointIndex = pMesh->GetPolygonVertex(lPolygonIndex, lVerticeIndex);
291 
292  if (mAllByControlPoint)
293  {
294  lIndices[lIndexOffset + lVerticeIndex] = static_cast<unsigned int>(lControlPointIndex);
295  }
296  // Populate the array with vertex attribute, if by polygon vertex.
297  else
298  {
299  lIndices[lIndexOffset + lVerticeIndex] = static_cast<unsigned int>(lVertexCount);
300 
301  lCurrentVertex = lControlPoints[lControlPointIndex];
302  lVertices[lVertexCount * VERTEX_STRIDE] = static_cast<float>(lCurrentVertex[0]);
303  lVertices[lVertexCount * VERTEX_STRIDE + 1] = static_cast<float>(lCurrentVertex[1]);
304  lVertices[lVertexCount * VERTEX_STRIDE + 2] = static_cast<float>(lCurrentVertex[2]);
305  lVertices[lVertexCount * VERTEX_STRIDE + 3] = 1;
306 
307  if (mHasNormal)
308  {
309  pMesh->GetPolygonVertexNormal(lPolygonIndex, lVerticeIndex, lCurrentNormal);
310  lNormals[lVertexCount * NORMAL_STRIDE] = static_cast<float>(lCurrentNormal[0]);
311  lNormals[lVertexCount * NORMAL_STRIDE + 1] = static_cast<float>(lCurrentNormal[1]);
312  lNormals[lVertexCount * NORMAL_STRIDE + 2] = static_cast<float>(lCurrentNormal[2]);
313  }
314 
315  if (mHasUV)
316  {
317  bool lUnmappedUV;
318  pMesh->GetPolygonVertexUV(lPolygonIndex, lVerticeIndex, lUVName, lCurrentUV, lUnmappedUV);
319  lUVs[lVertexCount * UV_STRIDE] = static_cast<float>(lCurrentUV[0]);
320  lUVs[lVertexCount * UV_STRIDE + 1] = static_cast<float>(lCurrentUV[1]);
321  }
322  }
323  ++lVertexCount;
324  }
325  mSubMeshes[lMaterialIndex]->TriangleCount += 1;
326  }
327 
328  // Create VBOs
329  glGenBuffers(VBO_COUNT, mVBONames);
330 
331  // Save vertex attributes into GPU
332  glBindBuffer(GL_ARRAY_BUFFER, mVBONames[VERTEX_VBO]);
333  glBufferData(GL_ARRAY_BUFFER, lPolygonVertexCount * VERTEX_STRIDE * sizeof(float), lVertices, GL_STATIC_DRAW);
334  delete [] lVertices;
335 
336  if (mHasNormal)
337  {
338  glBindBuffer(GL_ARRAY_BUFFER, mVBONames[NORMAL_VBO]);
339  glBufferData(GL_ARRAY_BUFFER, lPolygonVertexCount * NORMAL_STRIDE * sizeof(float), lNormals, GL_STATIC_DRAW);
340  delete [] lNormals;
341  }
342 
343  if (mHasUV)
344  {
345  glBindBuffer(GL_ARRAY_BUFFER, mVBONames[UV_VBO]);
346  glBufferData(GL_ARRAY_BUFFER, lPolygonVertexCount * UV_STRIDE * sizeof(float), lUVs, GL_STATIC_DRAW);
347  delete [] lUVs;
348  }
349 
350  glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, mVBONames[INDEX_VBO]);
351  glBufferData(GL_ELEMENT_ARRAY_BUFFER, lPolygonCount * TRIANGLE_VERTEX_COUNT * sizeof(unsigned int), lIndices, GL_STATIC_DRAW);
352  delete [] lIndices;
353 
354  return true;
355 }

357 void VBOMesh::UpdateVertexPosition(const FbxMesh * pMesh, const FbxVector4 * pVertices) const
358 {
359  // Convert to the same sequence with data in GPU.
360  float * lVertices = NULL;
361  int lVertexCount = 0;
362  if (mAllByControlPoint)
363  {
364  lVertexCount = pMesh->GetControlPointsCount();
365  lVertices = new float[lVertexCount * VERTEX_STRIDE];
366  for (int lIndex = 0; lIndex < lVertexCount; ++lIndex)
367  {
368  lVertices[lIndex * VERTEX_STRIDE] = static_cast<float>(pVertices[lIndex][0]);
369  lVertices[lIndex * VERTEX_STRIDE + 1] = static_cast<float>(pVertices[lIndex][1]);
370  lVertices[lIndex * VERTEX_STRIDE + 2] = static_cast<float>(pVertices[lIndex][2]);
371  lVertices[lIndex * VERTEX_STRIDE + 3] = 1;
372  }
373  }
374  else
375  {
376  const int lPolygonCount = pMesh->GetPolygonCount();
377  lVertexCount = lPolygonCount * TRIANGLE_VERTEX_COUNT;
378  lVertices = new float[lVertexCount * VERTEX_STRIDE];
379 
380  lVertexCount = 0;
381  for (int lPolygonIndex = 0; lPolygonIndex < lPolygonCount; ++lPolygonIndex)
382  {
383  for (int lVerticeIndex = 0; lVerticeIndex < TRIANGLE_VERTEX_COUNT; ++lVerticeIndex)
384  {
385  const int lControlPointIndex = pMesh->GetPolygonVertex(lPolygonIndex, lVerticeIndex);
386  lVertices[lVertexCount * VERTEX_STRIDE] = static_cast<float>(pVertices[lControlPointIndex][0]);
387  lVertices[lVertexCount * VERTEX_STRIDE + 1] = static_cast<float>(pVertices[lControlPointIndex][1]);
388  lVertices[lVertexCount * VERTEX_STRIDE + 2] = static_cast<float>(pVertices[lControlPointIndex][2]);
389  lVertices[lVertexCount * VERTEX_STRIDE + 3] = 1;
390  ++lVertexCount;
391  }
392  }
393  }
394 
395  // Transfer into GPU.
396  if (lVertices)
397  {
398  glBindBuffer(GL_ARRAY_BUFFER, mVBONames[VERTEX_VBO]);
399  glBufferData(GL_ARRAY_BUFFER, lVertexCount * VERTEX_STRIDE * sizeof(float), lVertices, GL_STATIC_DRAW);
400  delete [] lVertices;
401  }

433 void VBOMesh::BeginDraw(ShadingMode pShadingMode) const
434 {
435  // Push OpenGL attributes.
436  glPushClientAttrib(GL_CLIENT_VERTEX_ARRAY_BIT);
437  glPushAttrib(GL_ENABLE_BIT);
438  glPushAttrib(GL_CURRENT_BIT);
439  glPushAttrib(GL_LIGHTING_BIT);
440  glPushAttrib(GL_TEXTURE_BIT);
441 
442  // Set vertex position array.
443  glBindBuffer(GL_ARRAY_BUFFER, mVBONames[VERTEX_VBO]);
444  glVertexPointer(VERTEX_STRIDE, GL_FLOAT, 0, 0);
445  glEnableClientState(GL_VERTEX_ARRAY);
446 
447  // Set normal array.
448  if (mHasNormal && pShadingMode == SHADING_MODE_SHADED)
449  {
450  glBindBuffer(GL_ARRAY_BUFFER, mVBONames[NORMAL_VBO]);
451  glNormalPointer(GL_FLOAT, 0, 0);
452  glEnableClientState(GL_NORMAL_ARRAY);
453  }
454 
455  // Set UV array.
456  if (mHasUV && pShadingMode == SHADING_MODE_SHADED)
457  {
458  glBindBuffer(GL_ARRAY_BUFFER, mVBONames[UV_VBO]);
459  glTexCoordPointer(UV_STRIDE, GL_FLOAT, 0, 0);
460  glEnableClientState(GL_TEXTURE_COORD_ARRAY);
461  }
462 
463  // Set index array.
464  glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, mVBONames[INDEX_VBO]);
465 
466  if (pShadingMode == SHADING_MODE_SHADED)
467  {
468  glEnable(GL_LIGHTING);
469 
470  glEnable(GL_TEXTURE_2D);
471 
472  glEnable(GL_NORMALIZE);
473  }
474  else
475  {
476  glColor4fv(WIREFRAME_COLOR);
477  }
478 }

578 bool LightCache::Initialize(const FbxLight * pLight, FbxAnimLayer * pAnimLayer)
579 {

581 
582  FbxPropertyT<FbxDouble3> lColorProperty = pLight->Color;

606 void LightCache::SetLight(const FbxTime & pTime) const
607 {
608  const GLfloat lLightColor[4] = {mColorRed.Get(pTime), mColorGreen.Get(pTime), mColorBlue.Get(pTime), 1.0f};
609  const GLfloat lConeAngle = mConeAngle.Get(pTime);
610 
611  glColor3fv(lLightColor);
612 
613  glPushAttrib(GL_ENABLE_BIT);
614  glPushAttrib(GL_POLYGON_BIT);
615  // Visible for double side.
616  glDisable(GL_CULL_FACE);
617  // Draw wire-frame geometry.
618  glPolygonMode(GL_FRONT_AND_BACK, GL_LINE);
619  if (mType == FbxLight::eSpot)
620  {
621  // Draw a cone for spot light.
622  glPushMatrix();
623  glScalef(1.0f, 1.0f, -1.0f);
624  const double lRadians = ANGLE_TO_RADIAN * lConeAngle;
625  const double lHeight = 15.0;
626  const double lBase = lHeight * tan(lRadians / 2);
627  GLUquadricObj * lQuadObj = gluNewQuadric();
628  gluCylinder(lQuadObj, 0.0, lBase, lHeight, 18, 1);
629  gluDeleteQuadric(lQuadObj);
630  glPopMatrix();
631  }
632  else
633  {
634  // Draw a sphere for other types.
635  GLUquadricObj * lQuadObj = gluNewQuadric();
636  gluSphere(lQuadObj, 1.0, 10, 10);
637  gluDeleteQuadric(lQuadObj);
638  }
639  glPopAttrib();
640  glPopAttrib();
641 
642  // The transform have been set, so set in local coordinate.
643  if (mType == FbxLight::eDirectional)
644  {
645  glLightfv(mLightIndex, GL_POSITION, DEFAULT_DIRECTION_LIGHT_POSITION);
646  }
647  else
648  {
649  glLightfv(mLightIndex, GL_POSITION, DEFAULT_LIGHT_POSITION);
650  }
651 
652  glLightfv(mLightIndex, GL_DIFFUSE, lLightColor);
653  glLightfv(mLightIndex, GL_SPECULAR, lLightColor);
654 
655  if (mType == FbxLight::eSpot && lConeAngle != 0.0)
656  {
657  glLightfv(mLightIndex, GL_SPOT_DIRECTION, DEFAULT_SPOT_LIGHT_DIRECTION);
658 
659  // If the cone angle is 0, equal to a point light.
660  if (lConeAngle != 0.0f)
661  {
662  // OpenGL use cut off angle, which is half of the cone angle.
663  glLightf(mLightIndex, GL_SPOT_CUTOFF, lConeAngle/2);
664  }
665  }
666  glEnable(mLightIndex);
667 }
