13 //
14 // This example illustrates how to display the content of a FBX or a OBJ file
15 // in a graphical window. This program is based on the OpenGL Utility Toolkit
16 // (GLUT). Start the program on the command line by providing a FBX or a
17 // OBJ file name. A menu is provided to select the current camera and the current
18 // animation stack.
19 //
20 // Among other things, the example illustrates how to:
21 // 1) Use a custom memory allocator
22 // 2) Import a scene from a .FBX, .DAE or .OBJ file;
23 // 3) Convert the nurbs and patch attribute types of a scene into mesh
24 // node attributes; And trianglate all meshes.
25 // 4) Get the list of all the cameras in the scene;
26 // 5) Find the current camera;
27 // 6) Get the relevant settings of a camera depending on it's projection
28 // type and aperture mode;
29 // 7) Compute the local and global positions of a node;
30 // 8) Compute the orientation of a camera;
31 // 9) Compute the orientation of a light;
32 // 10) Compute the shape deformation of mesh vertices;
33 // 11) Compute the link deformation of mesh vertices.
34 // 12) Display the point cache simulation of a mesh.
35 // 13) Get the list of all pose in the scene;
36 // 14) Show the scene using at a specific pose.
37 //

85 class MyMemoryAllocator
86 {
87 public:
88  static void* MyMalloc(size_t pSize)
89  {
90  char *p = (char*)malloc(pSize+1);
91  *p = '#';
92  return p+1;
93  }
94 
95  static void* MyCalloc(size_t pCount, size_t pSize)
96  {
97  char *p = (char*)calloc(pCount, pSize+1);
98  *p = '#';
99  return p+1;
100  }
101 
102  static void* MyRealloc(void* pData, size_t pSize)
103  {
104  if (pData)
105  {
106  FBX_ASSERT(*((char*)pData-1)=='#');
107  if (*((char*)pData-1)=='#')
108  {
109  char *p = (char*)realloc((char*)pData-1, pSize+1);
110  *p = '#';
111  return p+1;
112  }
113  else
114  { // Mismatch
115  char *p = (char*)realloc((char*)pData, pSize+1);
116  *p = '#';
117  return p+1;
118  }
119  }
120  else
121  {
122  char *p = (char*)realloc(NULL, pSize+1);
123  *p = '#';
124  return p+1;
125  }
126  }
127 
128  static void MyFree(void* pData)
129  {
130  if (pData==NULL)
131  return;
132  FBX_ASSERT(*((char*)pData-1)=='#');
133  if (*((char*)pData-1)=='#')
134  {
135  free((char*)pData-1);
136  }
137  else
138  { // Mismatch
139  free(pData);
140  }
141  }
142 };

146 int main(int argc, char** argv)
147 {
148  // Set exit function to destroy objects created by the FBX SDK.
149  atexit(ExitFunction);
150 
151  // Use a custom memory allocator
152  FbxSetMallocHandler(MyMemoryAllocator::MyMalloc);
153  FbxSetReallocHandler(MyMemoryAllocator::MyRealloc);
154  FbxSetFreeHandler(MyMemoryAllocator::MyFree);
155  FbxSetCallocHandler(MyMemoryAllocator::MyCalloc);
156 
157  // glut initialisation
158  glutInit(&argc, argv);
159  glutInitDisplayMode(GLUT_DOUBLE | GLUT_RGBA | GLUT_DEPTH);
160  glutInitWindowSize(DEFAULT_WINDOW_WIDTH, DEFAULT_WINDOW_HEIGHT);
161  glutInitWindowPosition(100, 100);
162  glutCreateWindow("ViewScene");
163 
164  // Initialize OpenGL.
165  const bool lSupportVBO = InitializeOpenGL();
166 
167  // set glut callbacks
168  glutDisplayFunc(DisplayCallback);
169  glutReshapeFunc(ReshapeCallback);
170  glutKeyboardFunc(KeyboardCallback);
171  glutMouseFunc(MouseCallback);
172  glutMotionFunc(MotionCallback);
173 
174  FbxString lFilePath("");
175  for( int i = 1, c = argc; i < c; ++i )
176  {
177  if( FbxString(argv[i]) == "-test" ) gAutoQuit = true;
178  else if( lFilePath.IsEmpty() ) lFilePath = argv[i];
179  }
180 
181  gSceneContext = new SceneContext(!lFilePath.IsEmpty() ? lFilePath.Buffer() : NULL, DEFAULT_WINDOW_WIDTH, DEFAULT_WINDOW_HEIGHT, lSupportVBO);
182 
183  glutMainLoop();
184 
185  return 0;
186 }

195 // Create the menus to select the current camera and the current animation stack.
196 void CreateMenus()
197 {
198  // Create the submenu to select the current camera.
199  int lCameraMenu = glutCreateMenu(CameraSelectionCallback);
200 
201  // Add the producer cameras.
202  glutAddMenuEntry(FBXSDK_CAMERA_PERSPECTIVE, PRODUCER_PERSPECTIVE_ITEM);
203  glutAddMenuEntry(FBXSDK_CAMERA_TOP, PRODUCER_TOP_ITEM);
204  glutAddMenuEntry(FBXSDK_CAMERA_BOTTOM, PRODUCER_BOTTOM_ITEM);
205  glutAddMenuEntry(FBXSDK_CAMERA_FRONT, PRODUCER_FRONT_ITEM);
206  glutAddMenuEntry(FBXSDK_CAMERA_BACK, PRODUCER_BACK_ITEM);
207  glutAddMenuEntry(FBXSDK_CAMERA_RIGHT, PRODUCER_RIGHT_ITEM);
208  glutAddMenuEntry(FBXSDK_CAMERA_LEFT, PRODUCER_LEFT_ITEM);
209 
210  // Add the camera switcher if there is at least one camera in the scene.
211  const FbxArray<FbxNode *> & lCameraArray = gSceneContext->GetCameraArray();
212  if (lCameraArray.GetCount() > 0)
213  {
214  glutAddMenuEntry(FBXSDK_CAMERA_SWITCHER, CAMERA_SWITCHER_ITEM);
215  }
216 
217  // Add the cameras contained in the scene.
218  for (int lCameraIndex = 0; lCameraIndex < lCameraArray.GetCount(); ++lCameraIndex)
219  {
220  glutAddMenuEntry(lCameraArray[lCameraIndex]->GetName(), lCameraIndex);
221  }
222 
223  // Create the submenu to select the current animation stack.
224  int lAnimStackMenu = glutCreateMenu(AnimStackSelectionCallback);
225  int lCurrentAnimStackIndex = 0;
226 
227  // Add the animation stack names.
228  const FbxArray<FbxString *> & lAnimStackNameArray = gSceneContext->GetAnimStackNameArray();
229  for (int lPoseIndex = 0; lPoseIndex < lAnimStackNameArray.GetCount(); ++lPoseIndex)
230  {
231  glutAddMenuEntry(lAnimStackNameArray[lPoseIndex]->Buffer(), lPoseIndex);
232 
233  // Track the current animation stack index.
234  if (lAnimStackNameArray[lPoseIndex]->Compare(gSceneContext->GetScene()->ActiveAnimStackName.Get()) == 0)
235  {
236  lCurrentAnimStackIndex = lPoseIndex;
237  }
238  }
239 
240  // Call the animation stack selection callback immediately to
241  // initialize the start, stop and current time.
242  AnimStackSelectionCallback(lCurrentAnimStackIndex);
243 
244  const int lShadingModeMenu = glutCreateMenu(ShadingModeSelectionCallback);
245  glutAddMenuEntry(MENU_STRING_SHADING_MODE_WIREFRAME, MENU_SHADING_MODE_WIREFRAME);
246  glutAddMenuEntry(MENU_STRING_SHADING_MODE_SHADED, MENU_SHADING_MODE_SHADED);
247 
248  int lBindPoseCount = 0;
249  int lRestPoseCount = 0;
250  // Create a submenu for bind poses
251  int lBindPoseMenu = glutCreateMenu(PoseSelectionCallback);
252 
253  // Add the list of bind poses
254  const FbxArray<FbxPose *> & lPoseArray = gSceneContext->GetPoseArray();
255  for (int lPoseIndex = 0; lPoseIndex < lPoseArray.GetCount(); ++lPoseIndex)
256  {
257  if (lPoseArray[lPoseIndex]->IsBindPose())
258  {
259  glutAddMenuEntry(lPoseArray[lPoseIndex]->GetName(), lPoseIndex);
260  lBindPoseCount++;
261  }
262  }
263 
264  // Create a submenu for rest poses
265  int lRestPoseMenu = glutCreateMenu(PoseSelectionCallback);
266 
267  // Add the list of bind poses
268  for (int lPoseIndex = 0; lPoseIndex < lPoseArray.GetCount(); ++lPoseIndex)
269  {
270  if (lPoseArray[lPoseIndex]->IsRestPose())
271  {
272  glutAddMenuEntry(lPoseArray[lPoseIndex]->GetName(), lPoseIndex);
273  lRestPoseCount++;
274  }
275  }
276 
277  // Create the submenu to go to a specific pose
278  int lPoseMenu = 0;
279  if (lBindPoseCount>0 || lRestPoseCount>0)
280  {
281  lPoseMenu = glutCreateMenu(PoseSelectionCallback);
282  if (lBindPoseCount>0)
283  glutAddSubMenu("Bind Pose", lBindPoseMenu);
284  if (lRestPoseCount>0)
285  glutAddSubMenu("Rest Pose", lRestPoseMenu);
286  }
287 
288  // Create the submenu to zoom mode
289  int lZoomMenu = glutCreateMenu( CameraZoomModeCallback);
290  glutAddMenuEntry( "Zooming lens", MENU_ZOOM_FOCAL_LENGTH);
291  glutAddMenuEntry( "Zooming Position", MENU_ZOOM_POSITION);
292 
293  // Build the main menu.
294  glutCreateMenu(MenuSelectionCallback);
295  glutAddSubMenu("Select Camera", lCameraMenu);
296  glutAddSubMenu("Select Animation Stack", lAnimStackMenu);
297  glutAddSubMenu("Select Shading Mode", lShadingModeMenu);
298  if (lBindPoseCount>0 || lRestPoseCount>0)
299  glutAddSubMenu("Go to Pose", lPoseMenu);
300  glutAddSubMenu( "Zoom Mode", lZoomMenu);
301  glutAddMenuEntry("Play", PLAY_ANIMATION);
302  glutAddMenuEntry("Exit", MENU_EXIT);
303  glutAttachMenu(GLUT_RIGHT_BUTTON);
304 }

307 void CameraSelectionCallback(int pItem)
308 {

310  if (pItem == PRODUCER_PERSPECTIVE_ITEM)

315  {
316  gSceneContext->SetCurrentCamera(FBXSDK_CAMERA_TOP);
317  }
318  else if (pItem == PRODUCER_BOTTOM_ITEM)
319  {
320  gSceneContext->SetCurrentCamera(FBXSDK_CAMERA_BOTTOM);
321  }
322  else if (pItem == PRODUCER_FRONT_ITEM)
323  {
324  gSceneContext->SetCurrentCamera(FBXSDK_CAMERA_FRONT);
325  }
326  else if (pItem == PRODUCER_BACK_ITEM)
327  {
328  gSceneContext->SetCurrentCamera(FBXSDK_CAMERA_BACK);
329  }
330  else if (pItem == PRODUCER_RIGHT_ITEM)
331  {
332  gSceneContext->SetCurrentCamera(FBXSDK_CAMERA_RIGHT);
333  }
334  else if (pItem == PRODUCER_LEFT_ITEM)
335  {
336  gSceneContext->SetCurrentCamera(FBXSDK_CAMERA_LEFT);
337  }
338  else if (pItem == CAMERA_SWITCHER_ITEM)
339  {
340  gSceneContext->SetCurrentCamera(FBXSDK_CAMERA_SWITCHER);
341  }
342  else if (pItem >= 0 && pItem < lCameraArray.GetCount())
343  {
344  gSceneContext->SetCurrentCamera(lCameraArray[pItem]->GetName());
345  }

411 {

415

417  if (gSceneContext->GetStatus() == SceneContext::MUST_BE_LOADED)

422  gSceneContext->LoadFile();
423 
424  CreateMenus();
425 
426  // Call the timer to display the first frame.
427  glutTimerFunc((unsigned int)gSceneContext->GetFrameTime().GetMilliSeconds(), TimerCallback, 0);
428  }
429 
430  if( gAutoQuit ) exit(0);
