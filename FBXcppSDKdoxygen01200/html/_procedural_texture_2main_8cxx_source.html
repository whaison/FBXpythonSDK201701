150 bool DumpProceduralTextureBlobOnDisk(FbxScene* pScene)
151 {
152  FBXSDK_printf("Writing the blob on disk...\n");
153  // Collect all the procedural textures proxy objects in scene
154  int lNbProcTex = pScene->GetSrcObjectCount<FbxProceduralTexture>();
155 
156  if (!lNbProcTex)
157  {
158  return true;
159  }
160 
161  bool lWroteBlob = false; // Bool: Wrote at least one blob on disk
162 
163  // Directory for blob extraction
164  FbxString lDirPath = FbxGetCurrentWorkPath() + "/Blobs/";
165 
166  for(int lIndex = 0; lIndex < lNbProcTex; lIndex++)
167  {
168  FbxProceduralTexture* lProcTex = pScene->GetSrcObject<FbxProceduralTexture>(lIndex);
169  if(!lProcTex)
170  {
171  continue;
172  }
173 
174  FbxProperty lFilenameProp = lProcTex->RootProperty.Find(FILENAME_PROP);
175  if (!lFilenameProp.IsValid())
176  {
177  continue;
178  }
179 
180  // Read binary blob
181  void* lBlobBegin = NULL;
182  size_t lBlobSize = 0;
183  FbxBlob lBinaryBlob = lProcTex->GetBlob();
184  lBlobSize = lBinaryBlob.Size();
185  lBlobBegin = const_cast<void*>(lBinaryBlob.Access());
186 
187  // Get file name to dump the blob to.
188  FbxString lFilename = lFilenameProp.Get<FbxString>();
189  FbxString lFilePath = lDirPath + FbxPathUtils::GetFileName(lFilename, false) + lIndex + "." + FbxPathUtils::GetExtensionName(lFilename);
190 
191  bool lIsWritable = FbxPathUtils::Create(FbxPathUtils::GetFolderName(lFilePath));
192  if (lIsWritable)
193  {
194  std::ofstream lDataStreamOut(lFilePath.Buffer(), std::ofstream::binary);
195  lDataStreamOut.write((const char *)lBlobBegin, lBlobSize);
196  lDataStreamOut.close();
197  // So now we wrote the file!
198  lWroteBlob = true;
199  FBXSDK_printf("Blob is written on disk! File: %s\n", lFilePath.Buffer());
200  }
201  }
202 
203  return lWroteBlob;
204 }

223 // Create a plane.
224 FbxNode* CreatePlane(FbxScene* pScene, const char* pName)
225 {
226  FbxMesh* lMesh = FbxMesh::Create(pScene,pName);
227  FbxVector4 lControlPoint0(-50, 0, 50);
228  FbxVector4 lControlPoint1(50, 0, 50);
229  FbxVector4 lControlPoint2(50, 100, 50);
230  FbxVector4 lControlPoint3(-50, 100, 50);
231  FbxVector4 lNormal(0, 0, 1);
232 
233  // Create control points.
234  lMesh->InitControlPoints(4);
235  FbxVector4* lControlPoints = lMesh->GetControlPoints();
236 
237  lControlPoints[0] = lControlPoint0;
238  lControlPoints[1] = lControlPoint1;
239  lControlPoints[2] = lControlPoint2;
240  lControlPoints[3] = lControlPoint3;
241 
242  // Set the normals.
243  // We want to have one normal for each vertex (or control point),
244  // so we set the mapping mode to eByControlPoint.
245  FbxGeometryElementNormal* lGeometryElementNormal = lMesh->CreateElementNormal();;
246 
247  lGeometryElementNormal->SetMappingMode(FbxGeometryElement::eByControlPoint);
248 
249  // Set the actual normal value for all 4 control points.
250  lGeometryElementNormal->SetReferenceMode(FbxGeometryElement::eDirect);
251  lGeometryElementNormal->GetDirectArray().Add(lNormal);
252  lGeometryElementNormal->GetDirectArray().Add(lNormal);
253  lGeometryElementNormal->GetDirectArray().Add(lNormal);
254  lGeometryElementNormal->GetDirectArray().Add(lNormal);
255 
256  // Array of polygon vertices.
257  int lPolygonVertices[] = { 0, 1, 2, 3 };
258 
259  // Create UV for Diffuse channel
260  FbxGeometryElementUV* lUVDiffuseElement = lMesh->CreateElementUV( "DiffuseUV");
261  FBX_ASSERT( lUVDiffuseElement != NULL);
262  lUVDiffuseElement->SetMappingMode(FbxGeometryElement::eByPolygonVertex);
263  lUVDiffuseElement->SetReferenceMode(FbxGeometryElement::eIndexToDirect);
264 
265  FbxVector2 lVectors0(0, 0);
266  FbxVector2 lVectors1(1, 0);
267  FbxVector2 lVectors2(1, 1);
268  FbxVector2 lVectors3(0, 1);
269 
270  lUVDiffuseElement->GetDirectArray().Add(lVectors0);
271  lUVDiffuseElement->GetDirectArray().Add(lVectors1);
272  lUVDiffuseElement->GetDirectArray().Add(lVectors2);
273  lUVDiffuseElement->GetDirectArray().Add(lVectors3);
274 
275  //Now we have set the UVs as eIndexToDirect reference and in eByPolygonVertex mapping mode
276  //we must update the size of the index array.
277  lUVDiffuseElement->GetIndexArray().SetCount(4);
278 
279  // Create polygon
280  lMesh->BeginPolygon(-1, -1, false);
281  for (int j = 0; j < 4; j++)
282  {
283  //this function points
284  lMesh->AddPolygon(lPolygonVertices[j]);
285  //Now we have to update the index array of the UVs for diffuse
286  lUVDiffuseElement->GetIndexArray().SetAt(j, j);
287  }
288  lMesh->EndPolygon();
289 
290  FbxNode* lNode = FbxNode::Create(pScene,pName);
291 
292  lNode->SetNodeAttribute(lMesh);
293 
294  return lNode;
295 }

297 // Create a procedural texture.
298 FbxProceduralTexture* CreateProceduralTexture(FbxScene* pScene, const char* pName)
299 {
300  FbxProceduralTexture* lProceduralTexture = FbxProceduralTexture::Create(pScene, pName);
301 
302  // For this example we simply dump the binary content of a jpg to the
303  // procedural texture's blob property.
304  // In the general case, dump whatever information needed for the procedural
305  // texture to the blob property.
306 
307  FbxString lFilename = TEXTURE_FILENAME;
308 
309  //check whether the file is readable
310  bool lIsReadable = FbxFileUtils::Exist(lFilename);
311 
312  if (lIsReadable)
313  {
314  // create binary blob from the texture file
315  FbxBlob lBinaryBlob;
316  std::ifstream lDataStreamIn(lFilename, std::ifstream::binary);
317  char* lBlobBegin = (char*)malloc(4096);
318  char* lBlobEnd = lBlobBegin;
319  long long lBlobSize = 0;
320  long long lBlobPointerSize = 4096;
321  std::streamsize lNbRead = 0;
322  while(!lDataStreamIn.eof())
323  {
324  lBlobEnd = lBlobBegin + lBlobSize;
325  lDataStreamIn.read(lBlobEnd, 4096);
326  lNbRead = lDataStreamIn.gcount();
327  lBlobPointerSize += 4096;
328  lBlobBegin = (char *)realloc(lBlobBegin, size_t(lBlobPointerSize));
329  lBlobSize += lNbRead;
330  }
331  lDataStreamIn.close();
332  lBinaryBlob.Assign(lBlobBegin, (int)lBlobSize);
333  free(lBlobBegin);
334 
335  lProceduralTexture->SetBlob(lBinaryBlob);
336  }
337 
338  // Add a property to retain file name
339  FbxProperty lFilenameProp = FbxProperty::Create(lProceduralTexture, FbxStringDT, FILENAME_PROP);
340  if (lFilenameProp.IsValid())
341  {
342  lFilenameProp.Set(lFilename);
343  }
344 
345  return lProceduralTexture;
346 }
