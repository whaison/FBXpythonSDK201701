13 //
14 // The example illustrates how to:
15 // 1) get a stereo camera from a scene
16 // 2) evaluate left/right camera
17 // 3) get local/global matrix of left and right camera
18 // 4) create stereo camera set
19 // 5) connect left/right and stereo
20 //
21 //steps:
22 // 1. initialize FBX sdk object.
23 // 2. load fbx scene form the specified file.
24 // 3. Get root node of the scene.
25 // 4. Recursively traverse each node in the scene.
26 // 5. Detect and get stereo camera by from node attribute type.
27 // 6. Get left and right camera.
28 // 7. Evaluate left/right camera.
29 // 8. Get local/global matrix of left and right camera.
30 // 9. Create stereo camera set after extracting stereo camera.
31 // 10. Connect left/right and stereo.
32 // 11. Get the updated values via connections.
33 // 12. Destroy all objects.
34 //

52 {

59

62  FbxString lFilePath("");
63  for( int i = 1, c = argc; i < c; ++i )
64  {

69 
70  FBXSDK_printf("\n\nFile: %s\n\n", lFilePath.Buffer());
71  lResult = LoadScene(lSdkManager, lScene, lFilePath.Buffer());
72 
73  if(lResult == false)
74  {
75  FBXSDK_printf("\n\nAn error occurred while loading the scene...");
76  }
77  else
78  {
79  if(!lScene)
80  {

99 //This function illustrates how to get stereo camera info from scene.
100 void GetStereoCameraInfo(FbxNode* pNode)
101 {
102  if(!pNode)
103  return;
104 
105  //detect stereo camera by node attribute type
106  if(pNode->GetNodeAttribute() && pNode->GetNodeAttribute()->GetAttributeType() == FbxNodeAttribute::eCameraStereo)
107  {
108  //get stereo camera
109  FbxCameraStereo* lStereoCamera = (FbxCameraStereo*)pNode->GetNodeAttribute();
110 
111  //once we get the FbxCameraStereo, its connection has already been done.
112  //it's easy to get its left and right camera
113  //get left camera
114  FbxCamera* lLeftCamera = lStereoCamera->GetLeftCamera();
115  FbxString lLeftName = lLeftCamera->GetName();
116  FBXSDK_printf("left camera: %s\n", lLeftName.Buffer());
117  //get right camera
118  FbxCamera* lRightCamera = lStereoCamera->GetRightCamera();
119  FbxString lRightName = lRightCamera->GetName();
120  FBXSDK_printf("right camera: %s\n", lRightName.Buffer());
121 
122  //FBX SDK support reevaluation functionality
123  //reevaluate left/right camera
124  double lLeftOffsetX = lStereoCamera->ReevaluateLeftCameraFilmOffsetX();
125  FBXSDK_printf("reevaluated left camera film offset: %f\n", lLeftOffsetX);
126  double lLeftFbxOffsetX = lLeftCamera->FilmOffsetX.Get();
127  FBXSDK_printf("The original FBX left camera film offset: %f\n", lLeftFbxOffsetX);
128  double lRightOffsetX = lStereoCamera->ReevaluateRightCameraFilmOffsetX();
129  FBXSDK_printf("reevaluated right camera film offset: %f\n", lRightOffsetX);
130  double lRightFbxOffsetX = lRightCamera->FilmOffsetX.Get();
131  FBXSDK_printf("The original FBX right camera film offset: %f\n", lRightFbxOffsetX);
132 
133  //get local/global matrix of left and right camera
134  //Then you can get their Translate, Rotate and Scale info.
135  FbxAMatrix lLeft_localMatrix = lStereoCamera->GetLeftCameraLocalMatrix();
136  FbxAMatrix lRight_localMatrix = lStereoCamera->GetRightCameraLocalMatrix();
137  FbxAMatrix lLeft_globalMatrix = lStereoCamera->GetLeftCameraGlobalMatrix();
138  FbxAMatrix lRight_globalMatrix = lStereoCamera->GetRightCameraGlobalMatrix();
139 
140  //print the local and global TRS for left camera
141  FBXSDK_printf("===local TRS of left camera===\n");
142  PrintMatrix(lLeft_localMatrix);
143  FBXSDK_printf("===global TRS of left camera===\n");
144  PrintMatrix(lLeft_globalMatrix);
145 
146  //print the local and global TRS for right camera
147  FBXSDK_printf("===local TRS of right camera===\n");
148  PrintMatrix(lRight_localMatrix);
149  FBXSDK_printf("===global TRS of right camera===\n");
150  PrintMatrix(lRight_globalMatrix);
151  }// end if pNode->GetNodeAttribute()
152 
153 
154  //recursively traverse each node in the scene
155  int i, lCount = pNode->GetChildCount();
156  for (i = 0; i < lCount; i++)
157  {
158  GetStereoCameraInfo(pNode->GetChild(i));
159  }
160 }

187 //This function illustrates how to create and connect stereo camera.
188 void CreateStereoCamera(FbxScene* pScene)
189 {
190  if(!pScene)
191  return;
192 
193  //create a fbx node for stereo camera
194  FbxNode* lMyStereoNode = FbxNode::Create(pScene,"myStereoNode");
195  //create a cameraStereo, it's a node attribute of stereo camera node.
196  FbxCameraStereo* lMyStereoCamera = FbxCameraStereo::Create(pScene,"myStereoCamera");
197  //set stereoCamera as a node attribute of the FBX node.
198  lMyStereoNode->SetNodeAttribute (lMyStereoCamera);
199  //create a camera(node attribute), it will be left camera of stereo
200  FbxCamera* lLeftCamera = FbxCamera::Create(pScene, "leftCamera");
201  //create a camera(node attribute), it will be right camera of stereo
202  FbxCamera* lRightCamera = FbxCamera::Create(pScene, "rightCamera");
203  //add left camera to stereo
204  lMyStereoCamera->SetLeftCamera(lLeftCamera);
205  //add right camera to stereo
206  lMyStereoCamera->SetRightCamera(lRightCamera);
207 
208  //During FBXSDK reevaluating, if ConnectProperties() is called,
209  //left and right camera property will be connected to stereo camera.
210  //It's used to connect the left/right camera property [FocalLength, FarPlane, NearPlane, FilmWidth,
211  //FilmHeight, FilmSqueezeRatio] to stereo camera.
212  //If these properties of stereo camera have been modified by SDK,
213  //FBX will not automatically sync and update the corresponding properties of left/right camera.
214  //However, you could get the newest property of left/right camera since ConnectProperties() is called.
215  //To get the newest property value, please use lLeft_Camera->FocalLength.GetSrcProperty().Get(&lNewValue, ...);
216  //Then you can update your left/right camera properties, for example, lLeftCamera->FocalLength.Set(lNewValue);
217  lMyStereoCamera->ConnectProperties();
218 
219  //test the connection
220  //get the focal length value of left camera.
221  double lFocalLength_Left = lLeftCamera->FocalLength.Get();
222  FBXSDK_printf("FocalLength of left camera: %f\n", lFocalLength_Left);
223  double lFocalLength_Left_src = 0;
224  //get source property of left focal length, it should be stereo focal length.
225  FbxProperty lLeftSrcLengthProperty = lLeftCamera->FocalLength.GetSrcProperty();
226  if(lLeftSrcLengthProperty.IsValid())
227  {
228  lFocalLength_Left_src = lLeftSrcLengthProperty.Get<FbxDouble>();
229  FBXSDK_printf("Initialized FocalLength of left camera source: %f\n", lFocalLength_Left_src);
230  //modify the FocalLength of stereo camera
231  lMyStereoCamera->FocalLength.Set(3.333);
232  //get the FocalLength of left camera. It's 34.89 now.
233  //But it should be updated to 3.33 since stereo FocalLength has changed.
234  //FBX doesn't sync it from stereo camera
235  lFocalLength_Left = lLeftCamera->FocalLength.Get();
236  FBXSDK_printf("FocalLength of left camera: %f\n", lFocalLength_Left);
237  //get the newest value from connected property.
238  lFocalLength_Left_src = lLeftSrcLengthProperty.Get<FbxDouble>();
239  FBXSDK_printf("FocalLength of left camera source: %f\n", lFocalLength_Left_src);
240  }
241 }
