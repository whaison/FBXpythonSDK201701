21 // Get the global position of the node for the current pose.
22 // If the specified node is not part of the pose or no pose is specified, get its
23 // global position at the current time.
24 FbxAMatrix GetGlobalPosition(FbxNode* pNode, const FbxTime& pTime, FbxPose* pPose, FbxAMatrix* pParentGlobalPosition)
25 {
26  FbxAMatrix lGlobalPosition;
27  bool lPositionFound = false;
28 
29  if (pPose)
30  {
31  int lNodeIndex = pPose->Find(pNode);
32 
33  if (lNodeIndex > -1)
34  {
35  // The bind pose is always a global matrix.
36  // If we have a rest pose, we need to check if it is
37  // stored in global or local space.
38  if (pPose->IsBindPose() || !pPose->IsLocalMatrix(lNodeIndex))
39  {
40  lGlobalPosition = GetPoseMatrix(pPose, lNodeIndex);
41  }
42  else
43  {
44  // We have a local matrix, we need to convert it to
45  // a global space matrix.
46  FbxAMatrix lParentGlobalPosition;
47 
48  if (pParentGlobalPosition)
49  {
50  lParentGlobalPosition = *pParentGlobalPosition;
51  }
52  else
53  {
54  if (pNode->GetParent())
55  {
56  lParentGlobalPosition = GetGlobalPosition(pNode->GetParent(), pTime, pPose);
57  }
58  }
59 
60  FbxAMatrix lLocalPosition = GetPoseMatrix(pPose, lNodeIndex);
61  lGlobalPosition = lParentGlobalPosition * lLocalPosition;
62  }
63 
64  lPositionFound = true;
65  }
66  }
67 
68  if (!lPositionFound)
69  {
70  // There is no pose entry for that node, get the current global position instead.
71 
72  // Ideally this would use parent global position and local position to compute the global position.
73  // Unfortunately the equation
74  // lGlobalPosition = pParentGlobalPosition * lLocalPosition
75  // does not hold when inheritance type is other than "Parent" (RSrs).
76  // To compute the parent rotation and scaling is tricky in the RrSs and Rrs cases.
77  lGlobalPosition = pNode->EvaluateGlobalTransform(pTime);
78  }
79 
80  return lGlobalPosition;
81 }

83 // Get the matrix of the given pose
84 FbxAMatrix GetPoseMatrix(FbxPose* pPose, int pNodeIndex)
85 {
86  FbxAMatrix lPoseMatrix;
87  FbxMatrix lMatrix = pPose->GetMatrix(pNodeIndex);
88 
89  memcpy((double*)lPoseMatrix, (double*)lMatrix, sizeof(lMatrix.mData));
90 
91  return lPoseMatrix;
92 }

94 // Get the geometry offset to a node. It is never inherited by the children.
95 FbxAMatrix GetGeometry(FbxNode* pNode)
96 {
97  const FbxVector4 lT = pNode->GetGeometricTranslation(FbxNode::eSourcePivot);
98  const FbxVector4 lR = pNode->GetGeometricRotation(FbxNode::eSourcePivot);
99  const FbxVector4 lS = pNode->GetGeometricScaling(FbxNode::eSourcePivot);
100 
101  return FbxAMatrix(lT, lR, lS);
102 }
