43 /* Targa image and header fields -------------------------------------------*/
44 typedef struct
45 {
46  /* Note that Targa is stored in little-endian order */
47  uint8_t image_id_length;
48 
49  uint8_t color_map_type;
50  /* color map = palette */
51  #define TGA_COLOR_MAP_ABSENT 0
52  #define TGA_COLOR_MAP_PRESENT 1
53 
54  uint8_t image_type;
55  #define TGA_IMAGE_TYPE_NONE 0 /* no image data */
56  #define TGA_IMAGE_TYPE_COLORMAP 1 /* uncompressed, color-mapped */
57  #define TGA_IMAGE_TYPE_BGR 2 /* uncompressed, true-color */
58  #define TGA_IMAGE_TYPE_MONO 3 /* uncompressed, black and white */
59  #define TGA_IMAGE_TYPE_COLORMAP_RLE 9 /* run-length, color-mapped */
60  #define TGA_IMAGE_TYPE_BGR_RLE 10 /* run-length, true-color */
61  #define TGA_IMAGE_TYPE_MONO_RLE 11 /* run-length, black and white */
62 
63  /* color map specification */
64  uint16_t color_map_origin; /* index of first entry */
65  uint16_t color_map_length; /* number of entries included */
66  uint8_t color_map_depth; /* number of bits per entry */
67 
68  /* image specification */
69  uint16_t origin_x;
70  uint16_t origin_y;
71  uint16_t width;
72  uint16_t height;
73  uint8_t pixel_depth;
74 
75  uint8_t image_descriptor;
76  /* bits 0,1,2,3 - attribute bits per pixel
77  * bit 4 - set if image is stored right-to-left
78  * bit 5 - set if image is stored top-to-bottom
79  * bits 6,7 - unused (must be set to zero)
80  */
81  #define TGA_ATTRIB_BITS (uint8_t)(BIT(0)|BIT(1)|BIT(2)|BIT(3))
82  #define TGA_R_TO_L_BIT (uint8_t)BIT(4)
83  #define TGA_T_TO_B_BIT (uint8_t)BIT(5)
84  #define TGA_UNUSED_BITS (uint8_t)(BIT(6)|BIT(7))
85  /* Note: right-to-left order is not honored by some Targa readers */
86 
87  uint8_t *image_id;
88  /* The length of this field is given in image_id_length, it's read raw
89  * from the file so it's not not guaranteed to be zero-terminated. If
90  * it's not NULL, it needs to be deallocated. see: tga_free_buffers()
91  */
92 
93  uint8_t *color_map_data;
94  /* See the "color map specification" fields above. If not NULL, this
95  * field needs to be deallocated. see: tga_free_buffers()
96  */
97 
98  uint8_t *image_data;
99  /* Follows image specification fields (see above) */
100 
101  /* Extension area and developer area are silently ignored. The Targa 2.0
102  * spec says we're not required to read or write them.
103  */
104 
105 } tga_image;
