25 class CustomStreamClass : public FbxStream

27 public:
28  CustomStreamClass( FbxManager* pSdkManager, const char* mode )
29  {
30  mFile = NULL;
31 
32  // expect the mode to contain something
33  if (mode == NULL) return;
34 
35 
36  FBXSDK_strcpy(mFileName, 30, "CustomStreamClass_temp.txt");
37  FBXSDK_strcpy(mMode, 3, (mode) ? mode : "r");
38 
39  if ( mode[0] == 'r' )
40  {
41  const char* format = "FBX (*.fbx)";
42  mReaderID = pSdkManager->GetIOPluginRegistry()->FindReaderIDByDescription( format );
43  mWriterID = -1;
44  }
45  else
46  {
47  const char* format = "FBX ascii (*.fbx)";
48  mWriterID = pSdkManager->GetIOPluginRegistry()->FindWriterIDByDescription( format );
49  mReaderID = -1;
50  }
51  }

58  virtual EState GetState()
59  {

61  }

66  virtual bool Open( void* /*pStreamData*/ )
67  {
68  // This method can be called several times during the
69  // Initialize phase so it is important that it can handle
70  // multiple opens
71  if (mFile == NULL)
72  FBXSDK_fopen(mFile, mFileName, mMode);
73 
74  if (mFile != NULL)

76 
77  return ( mFile != NULL );

185 int testExport()

188  FbxManager* lSdkManager = FbxManager::Create();
189  // create an IOSettings object
190  FbxIOSettings* ios = FbxIOSettings::Create( lSdkManager, IOSROOT );
191  // create an empty scene.
192  FbxScene* lScene = FbxScene::Create( lSdkManager, "" );
193  // create an exporter.
194  FbxExporter* lExporter = FbxExporter::Create( lSdkManager, "" );
195  // create a stream.
196  CustomStreamClass stream( lSdkManager, "wb" );
197  // can pass in a void* data pointer to be passed to the stream on FileCreate
198  void* streamData = NULL;
199  // initialize the exporter.
200  if ( !lExporter->Initialize( &stream, streamData, -1, ios ) )
201  return -1;
202  // export the scene.
203  if ( !lExporter->Export( lScene ) )
204  return -1;
205  // destroy the exporter.
206  lExporter->Destroy();
