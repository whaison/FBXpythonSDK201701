103 bool CreateScene(FbxScene* pScene)
104 {
105  FbxNode* lLightGroup = CreateLightGroup(pScene, "LightGroup");
106  FbxNode* lMarker = CreateMarker(pScene, "Marker");
107  FbxNode* lCamera1 = CreateCamera(pScene, "Camera1");
108  FbxNode* lCamera2 = CreateCamera(pScene, "Camera2");
109 
110  pScene->GetGlobalSettings().SetAmbientColor(FbxColor(1.0, 0.5, 0.2));
111 
112  SetCameraPointOfInterest(lCamera1, lMarker);
113  SetCameraPointOfInterest(lCamera2, lCamera1);
114 
115  SetLightGroupDefaultPosition(lLightGroup);
116  SetMarkerDefaultPosition(lMarker);
117  SetCamera1DefaultPosition(lCamera1);
118  SetCamera2DefaultPosition(lCamera2);
119 
120  // Create the Animation Stack
121  FbxAnimStack* lAnimStack = FbxAnimStack::Create(pScene, "Rotating lights");
122 
123  // The animation nodes can only exist on AnimLayers therefore it is mandatory to
124  // add at least one AnimLayer to the AnimStack. And for the purpose of this example,
125  // one layer is all we need.
126  FbxAnimLayer* lAnimLayer = FbxAnimLayer::Create(pScene, "Base Layer");
127  lAnimStack->AddMember(lAnimLayer);
128 
129  // Build the scene graph.
130  FbxNode* lRootNode = pScene->GetRootNode();
131  lRootNode->AddChild(lLightGroup);
132  lRootNode->AddChild(lMarker);
133  lRootNode->AddChild(lCamera1);
134  lCamera1->AddChild(lCamera2);
135 
136 
137  // Set perspective camera as the default camera.
138  pScene->GetGlobalSettings().SetDefaultCamera(FBXSDK_CAMERA_PERSPECTIVE);
139 
140  AnimateLightGroup(lLightGroup, lAnimLayer);
141  AnimateCamera(lCamera1, lAnimLayer);
142 
143  return true;
144 }

147 FbxNode* CreateLightGroup(FbxScene* pScene, const char* pName)
148 {
149  FbxString lLightName;

153  int i;
154 
155  lGroup = FbxNode::Create(pScene,pName);
156 
157  for(i = 0; i < 6; i++)
158  {
159  lLightName = pName;
160  lLightName += "-Light";
161  lLightName += i;
162 
163  lNode = CreateLight(pScene, lLightName.Buffer());
164  lGroup->AddChild(lNode);
165  }
166 
167  for (i = 0; i < 6; i++)
168  {
169  lLight = (FbxLight*) lGroup->GetChild(i)->GetNodeAttribute();
170  lLight->FileName.Set("gobo.tif");// Resource file is in current directory.
171  lLight->DrawGroundProjection.Set(true);
172  lLight->DrawVolumetricLight.Set(true);
173  lLight->DrawFrontFacingVolumetricLight.Set(false);
174  }
175 
176  return lGroup;

303 // The light group rises and rotates.
304 void AnimateLightGroup(FbxNode* pLightGroup, FbxAnimLayer* pAnimLayer)
305 {
306  FbxAnimCurve* lCurve = NULL;
307  FbxTime lTime;
308  int i;
309  int lKeyIndex = 0;
310 
311  for (i = 0; i < pLightGroup->GetChildCount(); i++)
312  {
313  AnimateLight(pLightGroup->GetChild(i), i, pAnimLayer);
314  }
315 
316  // Create the CurveNodes (they are necessary for the GetCurve to successfully allocate the Animation curve)
317  pLightGroup->LclRotation.GetCurveNode(pAnimLayer, true);
318  pLightGroup->LclTranslation.GetCurveNode(pAnimLayer, true);
319 
320  // Y axis rotation.
321  lCurve = pLightGroup->LclRotation.GetCurve(pAnimLayer, FBXSDK_CURVENODE_COMPONENT_Y, true);
322  if (lCurve)
323  {
324  lCurve->KeyModifyBegin();
325 
326  lTime.SetSecondDouble(0.0);
327  lKeyIndex = lCurve->KeyAdd(lTime);
328  lCurve->KeySetValue(lKeyIndex, 0.0);
329  lCurve->KeySetInterpolation(lKeyIndex, FbxAnimCurveDef::eInterpolationLinear);
330 
331  lTime.SetSecondDouble(10.0);
332  lKeyIndex = lCurve->KeyAdd(lTime);
333  lCurve->KeySetValue(lKeyIndex, 5*360.0);
334 
335  lCurve->KeyModifyEnd();
336  }
337 
338  // Y axis translation.
339  lCurve = pLightGroup->LclTranslation.GetCurve(pAnimLayer, FBXSDK_CURVENODE_COMPONENT_Y, true);
340  if (lCurve)
341  {
342  lCurve->KeyModifyBegin();
343 
344  lTime.SetSecondDouble(0.0);
345  lKeyIndex = lCurve->KeyAdd(lTime);
346  lCurve->KeySetValue(lKeyIndex, 15.0);
347  lCurve->KeySetInterpolation(lKeyIndex, FbxAnimCurveDef::eInterpolationCubic);
348 
349  lTime.SetSecondDouble(5.0);
350  lKeyIndex = lCurve->KeyAdd(lTime);
351  lCurve->KeySetValue(lKeyIndex, 200.0);
352 
353  lCurve->KeyModifyEnd();
354  }
355 }

357 // The lights are changing color, intensity, orientation and cone angle.
358 void AnimateLight(FbxNode* pLight, int pIndex, FbxAnimLayer* pAnimLayer)
359 {
360  FbxAnimCurve* lCurve = NULL;
361  FbxTime lTime;
362  int i, j;
363  int lKeyIndex = 0;
364 
365  FbxLight* light = pLight->GetLight();
366 
367  // Intensity fade in/out.
368  // Create the CurveNode (it is necessary for the GetCurve to successfully allocate the Animation curve)
369  light->Intensity.GetCurveNode(pAnimLayer, true);
370  lCurve = light->Intensity.GetCurve(pAnimLayer, true);
371  if (lCurve)
372  {
373  lCurve->KeyModifyBegin();
374 
375  lTime.SetSecondDouble(0.0);
376  lKeyIndex = lCurve->KeyAdd(lTime);
377  lCurve->KeySetValue(lKeyIndex, 0.0);
378  lCurve->KeySetInterpolation(lKeyIndex, FbxAnimCurveDef::eInterpolationCubic);
379 
380  lTime.SetSecondDouble(3.0);
381  lKeyIndex = lCurve->KeyAdd(lTime);
382  lCurve->KeySetValue(lKeyIndex, 33.0);
383  lCurve->KeySetInterpolation(lKeyIndex, FbxAnimCurveDef::eInterpolationLinear);
384 
385  lTime.SetSecondDouble(7.0);
386  lKeyIndex = lCurve->KeyAdd(lTime);
387  lCurve->KeySetValue(lKeyIndex, 33.0);
388  lCurve->KeySetInterpolation(lKeyIndex, FbxAnimCurveDef::eInterpolationCubic);
389 
390  lTime.SetSecondDouble(10.0);
391  lKeyIndex = lCurve->KeyAdd(lTime);
392  lCurve->KeySetValue(lKeyIndex, 0.0);
393 
394  lCurve->KeyModifyEnd();
395  }
396 
397  // Fog fade in/out
398  // Create the CurveNode (it is necessary for the GetCurve to successfully allocate the Animation curve)
399  light->Fog.GetCurveNode(pAnimLayer, true);
400  lCurve = light->Fog.GetCurve(pAnimLayer, true);
401  if (lCurve)
402  {
403  lCurve->KeyModifyBegin();
404 
405  lTime.SetSecondDouble(0.0);
406  lKeyIndex = lCurve->KeyAdd(lTime);
407  lCurve->KeySetValue(lKeyIndex, 0.0);
408  lCurve->KeySetInterpolation(lKeyIndex, FbxAnimCurveDef::eInterpolationCubic);
409 
410  lTime.SetSecondDouble(3.0);
411  lKeyIndex = lCurve->KeyAdd(lTime);
412  lCurve->KeySetValue(lKeyIndex, 33.0);
413  lCurve->KeySetInterpolation(lKeyIndex, FbxAnimCurveDef::eInterpolationLinear);
414 
415  lTime.SetSecondDouble(7.0);
416  lKeyIndex = lCurve->KeyAdd(lTime);
417  lCurve->KeySetValue(lKeyIndex, 33.0);
418  lCurve->KeySetInterpolation(lKeyIndex, FbxAnimCurveDef::eInterpolationCubic);
419 
420  lTime.SetSecondDouble(10.0);
421  lKeyIndex = lCurve->KeyAdd(lTime);
422  lCurve->KeySetValue(lKeyIndex, 0.0);
423 
424  lCurve->KeyModifyEnd();
425  }
426 
427  // X rotation swoops & cone angle woobles.
428  {
429  // Create the CurveNodes (they are necessary for the GetCurve to successfully allocate the Animation curve)
430  pLight->LclRotation.GetCurveNode(pAnimLayer, true);
431  light->OuterAngle.GetCurveNode(pAnimLayer, true);
432 
433  lCurve = pLight->LclRotation.GetCurve(pAnimLayer, FBXSDK_CURVENODE_COMPONENT_X, true);
434  FbxAnimCurve* lConeCurve = light->OuterAngle.GetCurve(pAnimLayer,true);
435  double lValue;
436 
437  lCurve->KeyModifyBegin();
438  lConeCurve->KeyModifyBegin();
439 
440  for (i = 0; i < 8; i++)
441  {
442  lTime.SetSecondDouble((double)i * 0.833333);
443  lValue = cos((((double)i) + (((double)pIndex) * 60.0)) * 72.0);
444 
445  lKeyIndex = lCurve->KeyAdd(lTime);
446  lCurve->KeySetValue(lKeyIndex, float((lValue - 0.4) * 30.0));
447  lCurve->KeySetInterpolation(lKeyIndex, FbxAnimCurveDef::eInterpolationCubic);
448  lKeyIndex = lConeCurve->KeyAdd(lTime);
449  lConeCurve->KeySetValue(lKeyIndex, float((2.0 - (lValue + 1.0)) * 45.0));
450  lConeCurve->KeySetInterpolation(lKeyIndex, FbxAnimCurveDef::eInterpolationLinear);
451  }
452 
453  // Finally, have the lights spread out and lose focus.
454  lTime.SetSecondDouble(10.0);
455 
456  lKeyIndex = lCurve->KeyAdd(lTime);
457  lCurve->KeySetValue(lKeyIndex, -90.0);
458  lCurve->KeySetInterpolation(lKeyIndex, FbxAnimCurveDef::eInterpolationCubic);
459 
460  lKeyIndex = lConeCurve->KeyAdd(lTime);
461  lConeCurve->KeySetValue(lKeyIndex, 180.0);
462  lConeCurve->KeySetInterpolation(lKeyIndex, FbxAnimCurveDef::eInterpolationLinear);
463 
464  lCurve->KeyModifyEnd();
465  lConeCurve->KeyModifyEnd();
466  }
467 
468  // Color cycling.
469  {
470  FbxDouble3 lColor[6] =
471  {
472  FbxDouble3(1.0, 0.0, 0.0),
473  FbxDouble3(1.0, 1.0, 0.0),
474  FbxDouble3(0.0, 1.0, 0.0),
475  FbxDouble3(0.0, 1.0, 1.0),
476  FbxDouble3(0.0, 0.0, 1.0),
477  FbxDouble3(1.0, 0.0, 1.0)
478  };
479 
480  FbxAnimCurve* lCurveA[3];
481  // Create the CurveNodes (they are necessary for the GetCurve to successfully allocate the Animation curve)
482  light->Color.GetCurveNode(pAnimLayer, true);
483  lCurveA[0] = light->Color.GetCurve(pAnimLayer,FBXSDK_CURVENODE_COLOR_RED, true);
484  lCurveA[1] = light->Color.GetCurve(pAnimLayer,FBXSDK_CURVENODE_COLOR_GREEN, true);
485  lCurveA[2] = light->Color.GetCurve(pAnimLayer,FBXSDK_CURVENODE_COLOR_BLUE, true);
486 
487  if (lCurveA[0] && lCurveA[1] && lCurveA[2])
488  {
489  lCurveA[0]->KeyModifyBegin();
490  lCurveA[1]->KeyModifyBegin();
491  lCurveA[2]->KeyModifyBegin();
492 
493  for (i = 0; i < 24; i++)
494  {
495  j = i + pIndex;
496 
497  while (j > 5)
498  {
499  j -= 6;
500  }
501 
502  lTime.SetSecondDouble((double)i * 0.4166666);
503 
504  lKeyIndex = lCurveA[0]->KeyAdd(lTime);
505  lCurveA[0]->KeySetValue(lKeyIndex, (float)lColor[j][0]);
506  lCurveA[0]->KeySetInterpolation(lKeyIndex, FbxAnimCurveDef::eInterpolationCubic);
507 
508  lKeyIndex = lCurveA[1]->KeyAdd(lTime);
509  lCurveA[1]->KeySetValue(lKeyIndex, (float)lColor[j][1]);
510  lCurveA[1]->KeySetInterpolation(lKeyIndex, FbxAnimCurveDef::eInterpolationCubic);
511 
512  lKeyIndex = lCurveA[2]->KeyAdd(lTime);
513  lCurveA[2]->KeySetValue(lKeyIndex, (float)lColor[j][2]);
514  lCurveA[2]->KeySetInterpolation(lKeyIndex, FbxAnimCurveDef::eInterpolationCubic);
515  }
516 
517  lCurveA[0]->KeyModifyEnd();
518  lCurveA[1]->KeyModifyEnd();
519  lCurveA[2]->KeyModifyEnd();
520  }
521  }
