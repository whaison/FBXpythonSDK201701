16 void DisplayTextureInfo(FbxTexture* pTexture, int pBlendMode)

23  {

79 void FindAndDisplayTextureInfoByProperty(FbxProperty pProperty, bool& pDisplayHeader, int pMaterialIndex){
80 
81  if( pProperty.IsValid() )
82  {
83  int lTextureCount = pProperty.GetSrcObjectCount<FbxTexture>();
84 
85  for (int j = 0; j < lTextureCount; ++j)
86  {
87  //Here we have to check if it's layeredtextures, or just textures:
88  FbxLayeredTexture *lLayeredTexture = pProperty.GetSrcObject<FbxLayeredTexture>(j);
89  if (lLayeredTexture)
90  {
91  DisplayInt(" Layered Texture: ", j);
92  int lNbTextures = lLayeredTexture->GetSrcObjectCount<FbxTexture>();
93  for(int k =0; k<lNbTextures; ++k)
94  {
95  FbxTexture* lTexture = lLayeredTexture->GetSrcObject<FbxTexture>(k);
96  if(lTexture)
97  {
98 
99  if(pDisplayHeader){
100  DisplayInt(" Textures connected to Material ", pMaterialIndex);
101  pDisplayHeader = false;
102  }
103 
104  //NOTE the blend mode is ALWAYS on the LayeredTexture and NOT the one on the texture.
105  //Why is that? because one texture can be shared on different layered textures and might
106  //have different blend modes.
107 
108  FbxLayeredTexture::EBlendMode lBlendMode;
109  lLayeredTexture->GetTextureBlendMode(k, lBlendMode);
110  DisplayString(" Textures for ", pProperty.GetName());
111  DisplayInt(" Texture ", k);
112  DisplayTextureInfo(lTexture, (int) lBlendMode);
113  }
114 
115  }
116  }
117  else
118  {
119  //no layered texture simply get on the property
120  FbxTexture* lTexture = pProperty.GetSrcObject<FbxTexture>(j);
121  if(lTexture)
122  {
123  //display connected Material header only at the first time
124  if(pDisplayHeader){
125  DisplayInt(" Textures connected to Material ", pMaterialIndex);
126  pDisplayHeader = false;
127  }
128 
129  DisplayString(" Textures for ", pProperty.GetName());
130  DisplayInt(" Texture ", j);
131  DisplayTextureInfo(lTexture, -1);
132  }
133  }
134  }
135  }//end if pProperty
136 
137 }

140 void DisplayTexture(FbxGeometry* pGeometry)
141 {
142  int lMaterialIndex;
143  FbxProperty lProperty;
144  if(pGeometry->GetNode()==NULL)
145  return;
146  int lNbMat = pGeometry->GetNode()->GetSrcObjectCount<FbxSurfaceMaterial>();
147  for (lMaterialIndex = 0; lMaterialIndex < lNbMat; lMaterialIndex++){
148  FbxSurfaceMaterial *lMaterial = pGeometry->GetNode()->GetSrcObject<FbxSurfaceMaterial>(lMaterialIndex);
149  bool lDisplayHeader = true;
150 
151  //go through all the possible textures
152  if(lMaterial){
153 
154  int lTextureIndex;
155  FBXSDK_FOR_EACH_TEXTURE(lTextureIndex)
156  {
157  lProperty = lMaterial->FindProperty(FbxLayerElement::sTextureChannelNames[lTextureIndex]);
158  FindAndDisplayTextureInfoByProperty(lProperty, lDisplayHeader, lMaterialIndex);
159  }
160 
161  }//end if(lMaterial)
162 
163  }// end for lMaterialIndex
164 }
